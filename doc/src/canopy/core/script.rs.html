44/// Script host that owns the Rhai engine and scripts.
45pub(crate) struct ScriptHost {
46    /// Rhai engine instance.
47    engine: rhai::Engine,

54/// Argument list passed to Rhai function handlers.
55type FnCallArgs<'a> = [&'a mut rhai::Dynamic];

57/// Result type for script execution helpers.
58type ScriptResult<T> = StdResult<T, Box<rhai::EvalAltResult>>;

60/// Format a Rhai position for error messages.
61fn format_position(pos: rhai::Position) -> String {
62    let line = pos.line();
63    let offset = pos.position();
64    match (line, offset) {
65        (Some(line), Some(offset)) => format!(" (line {line}, offset {offset})"),
66        (Some(line), None) => format!(" (line {line})"),
67        (None, Some(offset)) => format!(" (offset {offset})"),
68        (None, None) => String::new(),
69    }
70}

72/// Convert a Rhai parse error into a Canopy parse error.
73fn format_parse_error(err: &rhai::ParseError) -> error::ParseError {
74    let pos = err.position();
75    error::ParseError::with_position(err.err_type().to_string(), pos.line(), pos.position())
76}

78/// This is a re-implementation of the Module::set_raw_fn from rhai. It turns out that set_raw_fn wants to assume that
79/// the function is a module, which imposes some internal constraints on the number of arguments.
80// Helper function removed - using FuncRegistration API directly instead
81impl ScriptHost {
82    /// Construct a new script host.
83    pub fn new() -> Self {
84        let mut engine = rhai::Engine::new();
85        engine.on_debug(move |s, src, pos| {
86            let src = src.unwrap_or("");
87            tracing::debug!("{} [{}:{}]", s, src, pos)
88        });
89        engine.on_print(move |s| tracing::info!("{}", s));
90
91        Self {
92            engine,
93            scripts: HashMap::new(),
94            current_id: 0,
95        }
96    }

98    /// Load command specs into the script engine.
99    pub fn load_commands(&mut self, cmds: &[CommandSpec]) {
100        // We can't enable this yet - see:
101        //      https://github.com/rhaiscript/rhai/issues/574
102        // engine.set_strict_variables(true);
103        let mut modules: HashMap<NodeName, rhai::Module> = HashMap::new();
104        for i in cmds {
105            let m = modules.entry(i.node.clone()).or_default();
106
107            let node = i.node.clone();
108            let command = i.command.clone();
109
110            let arg_types = i.args.clone();
111            let mut rhai_arg_types = vec![];
112            for a in &arg_types {
113                match a {
114                    ArgTypes::Context => {}
115                    ArgTypes::ISize => {
116                        rhai_arg_types.push(TypeId::of::<i64>());
117                    }
118                }
119            }
120
121            // For dynamic argument handling, we need to use the module's raw function API
122            // Since FuncRegistration doesn't support our use case directly
123            let func = move |context: Option<rhai::NativeCallContext>,
124                             args: &mut FnCallArgs|
125                  -> ScriptResult<rhai::Dynamic> {
126                SCRIPT_GLOBAL.with(|sg| {
127                    // SAFETY: `sg.core` is set from `&mut Core` for the duration of execute.
128                    let core = unsafe { &mut *sg.core };
129                    let node_id = sg.node_id;
130                    let command_label = format!("{node}::{command}");
131                    let call_pos = context
132                        .as_ref()
133                        .map(|ctx| ctx.call_position())
134                        .unwrap_or(rhai::Position::NONE);
135
136                    let make_error = |message: String| {
137                        Box::new(rhai::EvalAltResult::ErrorRuntime(
138                            rhai::Dynamic::from(format!(
139                                "command {command_label} on node {node_id:?}: {message}"
140                            )),
141                            call_pos,
142                        ))
143                    };
144
145                    let mut ciargs = vec![];
146                    let mut arg_types = arg_types.clone();
147                    if !arg_types.is_empty() && arg_types[0] == ArgTypes::Context {
148                        ciargs.push(Args::Context);
149                        arg_types.remove(0);
150                    }
151                    if args.len() != arg_types.len() {
152                        return Err(make_error(format!(
153                            "expected {} arguments, got {}",
154                            arg_types.len(),
155                            args.len()
156                        )));
157                    }
158                    for (i, a) in arg_types.iter().enumerate() {
159                        match a {
160                            ArgTypes::Context => {
161                                return Err(make_error(
162                                    "unexpected context argument in command signature".into(),
163                                ));
164                            }
165                            ArgTypes::ISize => {
166                                let val = args[i].as_int().map_err(|err| {
167                                    make_error(format!("argument {i} expected integer: {err}"))
168                                })?;
169                                ciargs.push(Args::ISize(val as isize));
170                            }
171                        }
172                    }
173
174                    let ci = CommandInvocation {
175                        node: node.clone(),
176                        command: command.clone(),
177                        args: ciargs,
178                    };
179                    match dispatch(core, sg.node_id, &ci) {
180                        Ok(Some(ret)) => Ok(match ret {
181                            ReturnValue::Void => rhai::Dynamic::UNIT,
182                            ReturnValue::String(s) => rhai::Dynamic::from(s),
183                        }),
184                        Ok(None) => Ok(rhai::Dynamic::UNIT),
185                        Err(err) => Err(make_error(err.to_string())),
186                    }
187                })
188            };
189
190            // Use the deprecated set_fn API since we need dynamic type handling
191            // We'll suppress the warning for now
192            #[allow(deprecated)]
193            m.set_fn(
194                i.command.clone(),
195                rhai::FnNamespace::Internal,
196                rhai::FnAccess::Public,
197                None,
198                &rhai_arg_types,
199                rhai::RhaiFunc::Pure {
200                    func: rhai::Shared::new(func),
201                    has_context: true,
202                    is_pure: true,
203                    is_volatile: false,
204                },
205            );
206        }
207        for (n, m) in modules {
208            self.engine.register_static_module(n.to_string(), m.into());
209        }
210    }

212    /// Compile a script and store it. Returns a ScriptId that can be used to
213    /// execute the script later.
214    pub fn compile(&mut self, source: &str) -> Result<ScriptId> {
215        self.current_id += 1;
216        let ast = self
217            .engine
218            .compile(source)
219            .map_err(|err| error::Error::Parse(format_parse_error(&err)))?;
220        let s = Script {
221            ast,
222            source: source.into(),
223        };
224        self.scripts.insert(self.current_id, s);
225        Ok(self.current_id)
226    }
227
228    /// Execute a script by id for the given node.
229    pub fn execute(&self, core: &mut Core, node_id: NodeId, sid: ScriptId) -> Result<()> {
230        self.execute_value(core, node_id, sid).map(|_| ())
231    }
232
233    /// Execute a script by id for the given node and return its value.
234    pub fn execute_value(
235        &self,
236        core: &mut Core,
237        node_id: NodeId,
238        sid: ScriptId,
239    ) -> Result<rhai::Dynamic> {
240        let s = self.scripts.get(&sid).ok_or_else(|| {
241            error::Error::Script(format!("script {sid} not found for node {node_id:?}"))
242        })?;
243        let sg = ScriptGlobal {
244            core: core as *mut Core,
245            node_id,
246        };
247        SCRIPT_GLOBAL.set(&sg, || {
248            self.engine.eval_ast::<rhai::Dynamic>(&s.ast).map_err(|e| {
249                let location = format_position(e.position());
250                error::Error::Script(format!(
251                    "script {sid} on node {node_id:?} failed{location}: {e}"
252                ))
253            })
254        })
255    }
