27/// Script execution context shared via thread-local pointer.
28struct ScriptGlobal<'a> {
29    /// Core context handle.
30    core: &'a mut dyn Context,
31    /// Root node handle.
32    root: &'a mut dyn Node,
33    /// Node identifier for dispatch.
34    node_id: NodeId,
35}
36
37scoped_thread_local!(static SCRIPT_GLOBAL: *const ());
38
39#[derive(Debug)]
40/// Script host that owns the Rhai engine and scripts.
41pub(crate) struct ScriptHost {
42    /// Rhai engine instance.
43    engine: rhai::Engine,
44    /// Loaded scripts by ID.
45    scripts: HashMap<ScriptId, Script>,
46    /// Next script ID.
47    current_id: u64,
48}
49
50/// Argument list passed to Rhai function handlers.
51type FnCallArgs<'a> = [&'a mut rhai::Dynamic];
52
53/// Result type for script execution helpers.
54type ScriptResult<T> = StdResult<T, Box<rhai::EvalAltResult>>;
55
56/// This is a re-implementation of the Module::set_raw_fn from rhai. It turns out that set_raw_fn wants to assume that
57/// the function is a module, which imposes some internal constraints on the number of arguments.
58// Helper function removed - using FuncRegistration API directly instead
59impl ScriptHost {
60    /// Construct a new script host.
61    pub fn new() -> Self {
62        let mut engine = rhai::Engine::new();
63        engine.on_debug(move |s, src, pos| {
64            let src = src.unwrap_or("");
65            tracing::debug!("{} [{}:{}]", s, src, pos)
66        });
67        engine.on_print(move |s| tracing::info!("{}", s));
68
69        Self {
70            engine,
71            scripts: HashMap::new(),
72            current_id: 0,
73        }
74    }
75
76    /// Load command specs into the script engine.
77    pub fn load_commands(&mut self, cmds: &[CommandSpec]) {
78        // We can't enable this yet - see:
79        //      https://github.com/rhaiscript/rhai/issues/574
80        // engine.set_strict_variables(true);
81        let mut modules: HashMap<NodeName, rhai::Module> = HashMap::new();
82        for i in cmds {
83            if !modules.contains_key(&i.node) {
84                let m = rhai::Module::new();
85                modules.insert(i.node.clone(), m);
86            }
87            let m = modules.get_mut(&i.node).unwrap();
88
89            let node = i.node.clone();
90            let command = i.command.clone();
91
92            let arg_types = i.args.clone();
93            let mut rhai_arg_types = vec![];
94            for a in &arg_types {
95                match a {
96                    ArgTypes::Context => {}
97                    ArgTypes::ISize => {
98                        rhai_arg_types.push(TypeId::of::<i64>());
99                    }
100                }
101            }
102
103            // For dynamic argument handling, we need to use the module's raw function API
104            // Since FuncRegistration doesn't support our use case directly
105            let func = move |_context: Option<rhai::NativeCallContext>,
106                             args: &mut FnCallArgs|
107                  -> ScriptResult<rhai::Dynamic> {
108                SCRIPT_GLOBAL.with(|ptr| {
109                    // SAFETY: `ptr` was created from a pointer to `ScriptGlobal`
110                    // which lives for the duration of this closure.
111                    let sg = unsafe { &mut *(*ptr as *mut ScriptGlobal) };
112                    let core = &mut *sg.core;
113                    let root = &mut *sg.root;
114
115                    let mut ciargs = vec![];
116                    let mut arg_types = arg_types.clone();
117                    if !arg_types.is_empty() && arg_types[0] == ArgTypes::Context {
118                        ciargs.push(Args::Context);
119                        arg_types.remove(0);
120                    }
121                    // I believe this is guaranteed by rhai
122                    assert!(args.len() == arg_types.len());
123                    for (i, a) in arg_types.iter().enumerate() {
124                        match a {
125                            ArgTypes::Context => {
126                                panic!("unexpected")
127                            }
128                            ArgTypes::ISize => {
129                                // The type here should be guaranteed by rhai
130                                ciargs.push(Args::ISize(args[i].as_int().unwrap() as isize));
131                            }
132                        }
133                    }
134
135                    let ci = CommandInvocation {
136                        node: node.clone(),
137                        command: command.clone(),
138                        args: ciargs,
139                    };
140                    if let Some(ret) = dispatch(core, sg.node_id.clone(), root, &ci).unwrap() {
141                        Ok(match ret {
142                            ReturnValue::Void => rhai::Dynamic::UNIT,
143                            ReturnValue::String(s) => rhai::Dynamic::from(s),
144                        })
145                    } else {
146                        Ok(rhai::Dynamic::UNIT)
147                    }
148                })
149            };
150
151            // Use the deprecated set_fn API since we need dynamic type handling
152            // We'll suppress the warning for now
153            #[allow(deprecated)]
154            m.set_fn(
155                i.command.clone(),
156                rhai::FnNamespace::Internal,
157                rhai::FnAccess::Public,
158                None,
159                &rhai_arg_types,
160                rhai::RhaiFunc::Pure {
161                    func: rhai::Shared::new(func),
162                    has_context: true,
163                    is_pure: true,
164                    is_volatile: false,
165                },
166            );
167        }
168        for (n, m) in modules {
169            self.engine.register_static_module(n.to_string(), m.into());
170        }
171    }
172
173    /// Compile a script and store it. Returns a ScriptId that can be used to
174    /// execute the script later.
175    pub fn compile(&mut self, source: &str) -> Result<ScriptId> {
176        self.current_id += 1;
177        let ast = self
178            .engine
179            .compile(source)
180            .map_err(|_e| error::Error::Parse(error::ParseError {}))?;
181        let s = Script {
182            ast,
183            source: source.into(),
184        };
185        self.scripts.insert(self.current_id, s);
186        Ok(self.current_id)
187    }
188
189    /// Execute a script by id for the given node.
190    pub fn execute(
191        &self,
192        core: &mut dyn Context,
193        root: &mut dyn Node,
194        node_id: NodeId,
195        sid: ScriptId,
196    ) -> Result<()> {
197        let s = self.scripts.get(&sid).unwrap();
198        let sg = ScriptGlobal {
199            core,
200            root,
201            node_id,
202        };
203        let ptr = &sg as *const _ as *const ();
204        SCRIPT_GLOBAL.set(&ptr, || {
205            self.engine
206                .run_ast(&s.ast)
207                .map_err(|e| error::Error::Script(e.to_string()))
208        })?;
209        Ok(())
210    }
211}
