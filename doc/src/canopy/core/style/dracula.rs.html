1//! Dracula theme - a dark theme with vibrant colors.
2//!
3//! Based on the Dracula theme: <https://draculatheme.com>
4
5use super::{Attr, Color, StyleMap};
6use crate::{rgb, style::AttrSet};
7
8// Dracula background colors
9/// Background.
10pub const BACKGROUND: Color = rgb!("#282a36");
11/// Current line / selection background.
12pub const CURRENT_LINE: Color = rgb!("#44475a");
13/// Selection.
14pub const SELECTION: Color = rgb!("#44475a");
15
16// Dracula foreground colors
17/// Foreground.
18pub const FOREGROUND: Color = rgb!("#f8f8f2");
19/// Comment color (also used for subtle elements).
20pub const COMMENT: Color = rgb!("#6272a4");
21
22// Dracula accent colors
23/// Red.
24pub const RED: Color = rgb!("#ff5555");
25/// Orange.
26pub const ORANGE: Color = rgb!("#ffb86c");
27/// Yellow.
28pub const YELLOW: Color = rgb!("#f1fa8c");
29/// Green.
30pub const GREEN: Color = rgb!("#50fa7b");
31/// Cyan.
32pub const CYAN: Color = rgb!("#8be9fd");
33/// Purple.
34pub const PURPLE: Color = rgb!("#bd93f9");
35/// Pink.
36pub const PINK: Color = rgb!("#ff79c6");
37
38// ANSI colors for terminal compatibility
39/// ANSI black.
40pub const ANSI_BLACK: Color = rgb!("#21222c");
41
42/// Build a Dracula style map.
43pub fn dracula() -> StyleMap {
44    use super::StyleBuilder;
45
46    let mut c = StyleMap::new();
47    c.rules()
48        .style(
49            "/",
50            StyleBuilder::new()
51                .fg(FOREGROUND)
52                .bg(BACKGROUND)
53                .attrs(AttrSet::default()),
54        )
55        .fg("/frame", COMMENT)
56        .fg("/frame/focused", PURPLE)
57        .fg("/frame/active", CYAN)
58        .fg("/frame/title", FOREGROUND)
59        .fg("/tab", COMMENT)
60        .style(
61            "/tab/inactive",
62            StyleBuilder::new().fg(FOREGROUND).bg(CURRENT_LINE),
63        )
64        .style("/tab/active", StyleBuilder::new().fg(BACKGROUND).bg(PURPLE))
65        .fg("/blue", CYAN)
66        .fg("/red", RED)
67        .fg("/magenta", PINK)
68        .fg("/violet", PURPLE)
69        .fg("/cyan", CYAN)
70        .fg("/green", GREEN)
71        .fg("/yellow", YELLOW)
72        .fg("/orange", ORANGE)
73        .fg("/black", ANSI_BLACK)
74        .attr("/text/bold", Attr::Bold)
75        .attr("/text/italic", Attr::Italic)
76        .attr("/text/underline", Attr::Underline)
77        .fg("/selector", FOREGROUND)
78        .fg("/selector/selected", PURPLE)
79        .style(
80            "/selector/focus",
81            StyleBuilder::new().fg(BACKGROUND).bg(PURPLE),
82        )
83        .style(
84            "/selector/focus/selected",
85            StyleBuilder::new().fg(BACKGROUND).bg(CYAN),
86        )
87        .fg("/dropdown", FOREGROUND)
88        .fg("/dropdown/selected", PURPLE)
89        .style(
90            "/dropdown/highlight",
91            StyleBuilder::new().fg(BACKGROUND).bg(PURPLE),
92        )
93        .style(
94            "/editor/text",
95            StyleBuilder::new().fg(FOREGROUND).bg(BACKGROUND),
96        )
97        .style(
98            "/editor/selection",
99            StyleBuilder::new().fg(FOREGROUND).bg(SELECTION),
100        )
101        .style(
102            "/editor/search/match",
103            StyleBuilder::new().fg(BACKGROUND).bg(YELLOW),
104        )
105        .style(
106            "/editor/search/current",
107            StyleBuilder::new().fg(BACKGROUND).bg(ORANGE),
108        )
109        .fg("/editor/line-number", COMMENT)
110        .fg("/editor/line-number/current", PURPLE)
111        .style(
112            "/editor/prompt",
113            StyleBuilder::new().fg(FOREGROUND).bg(CURRENT_LINE),
114        )
115        .apply();
116    c
117}
