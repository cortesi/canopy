1//! Dracula theme - a dark theme with vibrant colors.
2//!
3//! Based on the Dracula theme: <https://draculatheme.com>
4
5use super::{Attr, Color, StyleMap};
6use crate::{rgb, style::AttrSet};
7
8// Dracula background colors
9/// Background.
10pub const BACKGROUND: Color = rgb!("#282a36");
11/// Current line / selection background.
12pub const CURRENT_LINE: Color = rgb!("#44475a");
13/// Selection.
14pub const SELECTION: Color = rgb!("#44475a");
15
16// Dracula foreground colors
17/// Foreground.
18pub const FOREGROUND: Color = rgb!("#f8f8f2");
19/// Comment color (also used for subtle elements).
20pub const COMMENT: Color = rgb!("#6272a4");
21
22// Dracula accent colors
23/// Red.
24pub const RED: Color = rgb!("#ff5555");
25/// Orange.
26pub const ORANGE: Color = rgb!("#ffb86c");
27/// Yellow.
28pub const YELLOW: Color = rgb!("#f1fa8c");
29/// Green.
30pub const GREEN: Color = rgb!("#50fa7b");
31/// Cyan.
32pub const CYAN: Color = rgb!("#8be9fd");
33/// Purple.
34pub const PURPLE: Color = rgb!("#bd93f9");
35/// Pink.
36pub const PINK: Color = rgb!("#ff79c6");
37
38// ANSI colors for terminal compatibility
39/// ANSI black.
40pub const ANSI_BLACK: Color = rgb!("#21222c");
41
42/// Build a Dracula style map.
43pub fn dracula() -> StyleMap {
44    let mut c = StyleMap::new();
45    c.add(
46        "/",
47        Some(FOREGROUND),
48        Some(BACKGROUND),
49        Some(AttrSet::default()),
50    );
51    c.add_fg("/frame", COMMENT);
52    c.add_fg("/frame/focused", PURPLE);
53    c.add_fg("/frame/active", CYAN);
54    c.add_fg("/frame/title", FOREGROUND);
55    c.add_fg("/tab", COMMENT);
56    c.add_fg("/tab/inactive", FOREGROUND);
57    c.add_bg("/tab/inactive", CURRENT_LINE);
58    c.add_fg("/tab/active", BACKGROUND);
59    c.add_bg("/tab/active", PURPLE);
60
61    c.add_fg("/blue", CYAN);
62    c.add_fg("/red", RED);
63    c.add_fg("/magenta", PINK);
64    c.add_fg("/violet", PURPLE);
65    c.add_fg("/cyan", CYAN);
66    c.add_fg("/green", GREEN);
67    c.add_fg("/yellow", YELLOW);
68    c.add_fg("/orange", ORANGE);
69    c.add_fg("/black", ANSI_BLACK);
70
71    // Text style variants
72    c.add("/text/bold", None, None, Some(AttrSet::new(Attr::Bold)));
73    c.add("/text/italic", None, None, Some(AttrSet::new(Attr::Italic)));
74    c.add(
75        "/text/underline",
76        None,
77        None,
78        Some(AttrSet::new(Attr::Underline)),
79    );
80
81    // Selector widget styles
82    c.add_fg("/selector", FOREGROUND);
83    c.add_fg("/selector/selected", PURPLE);
84    c.add("/selector/focus", Some(BACKGROUND), Some(PURPLE), None);
85    c.add(
86        "/selector/focus/selected",
87        Some(BACKGROUND),
88        Some(CYAN),
89        None,
90    );
91
92    // Dropdown widget styles
93    c.add_fg("/dropdown", FOREGROUND);
94    c.add_fg("/dropdown/selected", PURPLE);
95    c.add("/dropdown/highlight", Some(BACKGROUND), Some(PURPLE), None);
96
97    c
98}
