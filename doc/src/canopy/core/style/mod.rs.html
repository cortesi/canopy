10/// A text attribute.
11#[derive(Debug, PartialEq, Eq, Clone, Copy)]
12pub enum Attr {
13    /// Bold text.
14    Bold,
15    /// Crossed out text.
16    CrossedOut,
17    /// Dim text.
18    Dim,
19    /// Italic text.
20    Italic,
21    /// Overlined text.
22    Overline,
23    /// Underlined text.
24    Underline,
25}
26
27/// A set of active text attributes.
28#[derive(Debug, PartialEq, Eq, Clone, Copy)]
29pub struct AttrSet {
30    /// Bold flag.
31    pub bold: bool,
32    /// Crossed out flag.
33    pub crossedout: bool,
34    /// Dim flag.
35    pub dim: bool,
36    /// Italic flag.
37    pub italic: bool,
38    /// Overline flag.
39    pub overline: bool,
40    /// Underline flag.
41    pub underline: bool,
42}
43
44impl Default for AttrSet {
45    /// Construct an empty set of text attributes.
46    fn default() -> Self {
47        Self {
48            bold: false,
49            crossedout: false,
50            dim: false,
51            italic: false,
52            overline: false,
53            underline: false,
54        }
55    }
56}
57
58impl AttrSet {
59    /// Construct a set of text attributes with a single attribute turned on.
60    pub fn new(attr: Attr) -> Self {
61        Self::default().with(attr)
62    }
63    /// Is this attribute set empty?
64    pub fn is_empty(&self) -> bool {
65        !(self.bold
66            || self.dim
67            || self.italic
68            || self.crossedout
69            || self.overline
70            || self.underline)
71    }
72    /// A helper for progressive construction of attribute sets.
73    pub fn with(mut self, attr: Attr) -> Self {
74        match attr {
75            Attr::Bold => self.bold = true,
76            Attr::Dim => self.dim = true,
77            Attr::Italic => self.italic = true,
78            Attr::CrossedOut => self.crossedout = true,
79            Attr::Underline => self.underline = true,
80            Attr::Overline => self.overline = true,
81        };
82        self
83    }
84}

88pub struct Style {

90    pub fg: Color,
91    /// Background color.
92    pub bg: Color,
93    /// Text attributes.
94    pub attrs: AttrSet,

97/// A possibly partial style specification, which is stored in a StyleManager.
98/// Partial styles are completely resolved during the style resolution process.
99#[derive(Default, Debug, PartialEq, Eq, Clone)]
100pub struct PartialStyle {
101    /// Optional foreground color.
102    pub fg: Option<Color>,
103    /// Optional background color.
104    pub bg: Option<Color>,
105    /// Optional attributes.
106    pub attrs: Option<AttrSet>,
107}
108
109impl PartialStyle {
110    /// Create a new PartialStyle with only a foreground color.
111    pub fn fg(fg: Color) -> Self {
112        Self {
113            fg: Some(fg),
114            bg: None,
115            attrs: None,
116        }
117    }
118
119    /// Create a new PartialStyle with only a background color.
120    pub fn bg(bg: Color) -> Self {
121        Self {
122            fg: None,
123            bg: Some(bg),
124            attrs: None,
125        }
126    }
127
128    /// Create a new PartialStyle with only attributes.
129    pub fn attrs(attrs: AttrSet) -> Self {
130        Self {
131            fg: None,
132            bg: None,
133            attrs: Some(attrs),
134        }
135    }
136
137    /// Resolve the partial style into a full style.
138    pub fn resolve(&self) -> Style {
139        Style {
140            fg: self.fg.unwrap(),
141            bg: self.bg.unwrap(),
142            attrs: self.attrs.unwrap(),
143        }
144    }
145
146    /// Set the foreground color.
147    pub fn with_fg(mut self, fg: Color) -> Self {
148        self.fg = Some(fg);
149        self
150    }
151
152    /// Set the background color.
153    pub fn with_bg(mut self, bg: Color) -> Self {
154        self.bg = Some(bg);
155        self
156    }
157
158    /// Add a single attribute.
159    pub fn with_attr(mut self, attr: Attr) -> Self {
160        if let Some(attrs) = self.attrs {
161            self.attrs = Some(attrs.with(attr));
162        } else {
163            self.attrs = Some(AttrSet::new(attr));
164        }
165        self
166    }
167
168    /// Replace the attributes set.
169    pub fn with_attrs(mut self, attrs: AttrSet) -> Self {
170        self.attrs = Some(attrs);
171        self
172    }
173
174    /// Merge two partial styles.
175    pub fn join(&self, other: &Self) -> Self {
176        Self {
177            fg: if self.fg.is_some() { self.fg } else { other.fg },
178            bg: if self.bg.is_some() { self.bg } else { other.bg },
179            attrs: if self.attrs.is_some() {
180                self.attrs
181            } else {
182                other.attrs
183            },
184        }
185    }
186
187    /// Return true if all components are set.
188    pub fn is_complete(&self) -> bool {
189        self.fg.is_some() && self.bg.is_some() && self.attrs.is_some()
190    }
191}

193/// Split a style path into components.
194fn parse_path(path: &str) -> Vec<String> {
195    path.split('/')
196        .filter_map(|s| {
197            if !s.is_empty() {
198                Some(s.to_owned())
199            } else {
200                None
201            }
202        })
203        .collect()
204}

206/// Map of style paths to partial styles.
207#[derive(Debug, Default)]
208pub struct StyleMap {
209    /// Path-to-style map.
210    styles: HashMap<Vec<String>, PartialStyle>,
211}
212
213impl StyleMap {
214    /// Construct a style map with defaults.
215    pub fn new() -> Self {
216        let mut cs = Self {
217            styles: HashMap::new(),
218        };
219        cs.add(
220            "/",
221            Some(Color::White),
222            Some(Color::Black),
223            Some(AttrSet::default()),
224        );
225        cs
226    }
227
228    /// Insert a foreground color at a specified path.
229    pub fn add_fg(&mut self, path: &str, fg: Color) {
230        let parsed = parse_path(path);
231        if let Some(ps) = self.styles.get_mut(&parsed) {
232            ps.fg = Some(fg);
233        } else {
234            self.styles
235                .insert(parsed, PartialStyle::default().with_fg(fg));
236        }
237    }
238
239    /// Insert a background color at a specified path.
240    pub fn add_bg(&mut self, path: &str, bg: Color) {
241        let parsed = parse_path(path);
242        if let Some(ps) = self.styles.get_mut(&parsed) {
243            ps.bg = Some(bg);
244        } else {
245            self.styles
246                .insert(parsed, PartialStyle::default().with_bg(bg));
247        }
248    }
249
250    /// Insert a style attribute at a specified path.
251    pub fn add_attr(&mut self, path: &str, attr: Attr) {
252        let parsed = parse_path(path);
253        if let Some(ps) = self.styles.get_mut(&parsed) {
254            if let Some(attrs) = ps.attrs {
255                ps.attrs = Some(attrs.with(attr));
256            } else {
257                ps.attrs = Some(AttrSet::default().with(attr));
258            }
259        } else {
260            self.styles
261                .insert(parsed, PartialStyle::default().with_attr(attr));
262        }
263    }
264
265    /// Add a style at a specified path.
266    pub fn add(
267        &mut self,
268        path: &str,
269        fg: Option<Color>,
270        bg: Option<Color>,
271        attrs: Option<AttrSet>,
272    ) {
273        self.styles
274            .insert(parse_path(path), PartialStyle { fg, bg, attrs });
275    }
276}

278/// A hierarchical style manager.
279///
280/// `Style` objects are entered into the manager with '/'-separated paths. For
281/// example:
282///
283///   / white, black
284///   /frame -> grey, None
285///   /frame/selected -> blue, None
286///
287/// The first entry with the empty path is the global default. Every
288/// `StyleManager` is guaranteed to have a default Style object with non-None
289/// foreground and background colors, so style resolution always succeeds.
290///
291/// `Style` objects also contain text attributes.
292///
293/// During rendering, a node may push a name onto the stack of layers tracked by
294/// the `Style` object. Layers are maintained for a node and all its
295/// descendants, and `Canopy` manages poppping layers back off the stack at the
296/// appropriate time during rendering.
297///
298/// When a colour is resolved, we first try to find the specified path under
299/// each layer to the root; failing that we look up the default colours for each
300/// layer to the root.
301///
302/// So given a layer stack ["foo"], and an attempt to look up "frame/selected",
303/// we try the following lookups in order: ["foo/frame/selected",
304/// "/frame/selected", "foo", ""].
305#[derive(Debug, PartialEq, Eq, Clone)]
306pub struct StyleManager {
307    /// Current render level.
308    level: usize,
309    /// Active layer names.
310    layers: Vec<String>,
311    /// Render levels corresponding to layers.
312    layer_levels: Vec<usize>,
313}
314
315impl Default for StyleManager {
316    fn default() -> Self {
317        Self::new()
318    }
319}
320
321impl StyleManager {
322    /// Construct a new style manager.
323    pub fn new() -> Self {
324        Self {
325            level: 0,
326            layers: vec![],
327            layer_levels: vec![],
328        }
329    }
330
331    /// Reset all layers and levels.
332    pub fn reset(&mut self) {
333        self.level = 0;
334        self.layers = vec![];
335        self.layer_levels = vec![0];
336    }
337
338    /// Increment the render level.
339    pub fn push(&mut self) {
340        self.level += 1
341    }
342
343    /// Decrement the render level and pop any layers at this level.
344    pub fn pop(&mut self) {
345        if self.level != 0 {
346            if self.layer_levels.last() == Some(&self.level) {
347                self.layers.pop();
348                self.layer_levels.pop();
349            }
350            self.level -= 1
351        }
352    }
353
354    /// Push onto the layer stack with the current render level.
355    pub fn push_layer(&mut self, name: &str) {
356        self.layers.push(name.to_owned());
357        self.layer_levels.push(self.level);
358    }
359
360    /// Resolve a style path.
361    pub fn get(&self, smap: &StyleMap, path: &str) -> Style {
362        self.resolve(smap, &self.layers, &parse_path(path))
363    }
364
365    /// Look up one suffix along a layer chain.
366    fn lookup(&self, smap: &StyleMap, layers: &[String], suffix: &[String]) -> PartialStyle {
367        let mut ret = PartialStyle::default();
368        // Look up the path on all layers to the root.
369        for i in 0..layers.len() + 1 {
370            let mut v = layers[0..layers.len() - i].to_vec();
371            v.extend(suffix.to_vec());
372            if let Some(c) = smap.styles.get(&v) {
373                ret = ret.join(c);
374                if ret.is_complete() {
375                    break;
376                }
377            }
378        }
379        ret
380    }
381
382    /// Directly resolve a style using a path and a layer specification,
383    /// ignoring `self.layers`.
384    pub(crate) fn resolve(&self, smap: &StyleMap, layers: &[String], path: &[String]) -> Style {
385        let mut ret = PartialStyle::default();
386        for i in 0..path.len() + 1 {
387            ret = ret.join(&self.lookup(smap, layers, &path[0..path.len() - i]));
388            if ret.is_complete() {
389                break;
390            }
391        }
392        ret.resolve()
393    }
394}
