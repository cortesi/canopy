36/// A set of active text attributes.
37#[derive(Debug, PartialEq, Eq, Clone, Copy)]
38pub struct AttrSet {
39    /// Bold flag.
40    pub bold: bool,
41    /// Crossed out flag.
42    pub crossedout: bool,
43    /// Dim flag.
44    pub dim: bool,
45    /// Italic flag.
46    pub italic: bool,
47    /// Overline flag.
48    pub overline: bool,
49    /// Underline flag.
50    pub underline: bool,

53impl Default for AttrSet {
54    /// Construct an empty set of text attributes.
55    fn default() -> Self {
56        Self {
57            bold: false,
58            crossedout: false,
59            dim: false,
60            italic: false,
61            overline: false,
62            underline: false,
63        }
64    }
65}
66
67impl AttrSet {
68    /// Construct a set of text attributes with a single attribute turned on.
69    pub fn new(attr: Attr) -> Self {
70        Self::default().with(attr)
71    }
72    /// Is this attribute set empty?
73    pub fn is_empty(&self) -> bool {
74        !(self.bold
75            || self.dim
76            || self.italic
77            || self.crossedout
78            || self.overline
79            || self.underline)
80    }
81    /// A helper for progressive construction of attribute sets.
82    pub fn with(mut self, attr: Attr) -> Self {
83        match attr {
84            Attr::Bold => self.bold = true,
85            Attr::Dim => self.dim = true,
86            Attr::Italic => self.italic = true,
87            Attr::CrossedOut => self.crossedout = true,
88            Attr::Underline => self.underline = true,
89            Attr::Overline => self.overline = true,
90        };
91        self
92    }
93}

114/// A gradient paint specification.
115#[derive(Debug, Clone, PartialEq)]
116pub struct GradientSpec {
117    /// Gradient angle in degrees (0 = left to right, 90 = top to bottom).
118    pub angle_deg: f32,
119    /// Ordered list of gradient stops.
120    pub stops: Vec<GradientStop>,
121}
122
123impl GradientSpec {
124    /// Construct a two-stop gradient.
125    pub fn new(angle_deg: f32, start: Color, end: Color) -> Self {
126        Self::with_stops(
127            angle_deg,
128            vec![GradientStop::new(0.0, start), GradientStop::new(1.0, end)],
129        )
130    }
131
132    /// Construct a gradient from explicit stops.
133    pub fn with_stops(angle_deg: f32, mut stops: Vec<GradientStop>) -> Self {
134        if stops.is_empty() {
135            stops.push(GradientStop::new(0.0, Color::White));
136            stops.push(GradientStop::new(1.0, Color::White));
137        }
138        stops.sort_by(|a, b| a.offset.total_cmp(&b.offset));
139        Self { angle_deg, stops }
140    }
141
142    /// Map all colors in this gradient through a transform.
143    pub fn map_colors(&self, f: impl Fn(Color) -> Color) -> Self {
144        let stops = self
145            .stops
146            .iter()
147            .map(|stop| GradientStop::new(stop.offset, f(stop.color)))
148            .collect();
149        Self {
150            angle_deg: self.angle_deg,
151            stops,
152        }
153    }
154
155    /// Resolve a gradient color at a point within a rectangle.
156    pub fn color_at(&self, rect: geom::Rect, point: geom::Point) -> Color {
157        if rect.w == 0 || rect.h == 0 {
158            return self.stops[0].color;
159        }
160
161        let width = rect.w as f32;
162        let height = rect.h as f32;
163        let angle = self.angle_deg.to_radians();
164        let dir_x = angle.cos();
165        let dir_y = angle.sin();
166        let corners = [(0.0, 0.0), (width, 0.0), (0.0, height), (width, height)];
167        let (min_dot, max_dot) = corners.iter().fold(
168            (f32::INFINITY, f32::NEG_INFINITY),
169            |(min_dot, max_dot), (x, y)| {
170                let dot = dir_x * x + dir_y * y;
171                (min_dot.min(dot), max_dot.max(dot))
172            },
173        );
174
175        let local_x = point.x.saturating_sub(rect.tl.x) as f32 + 0.5;
176        let local_y = point.y.saturating_sub(rect.tl.y) as f32 + 0.5;
177        let dot = dir_x * local_x + dir_y * local_y;
178        let ratio = if (max_dot - min_dot).abs() < f32::EPSILON {
179            0.0
180        } else {
181            ((dot - min_dot) / (max_dot - min_dot)).clamp(0.0, 1.0)
182        };
183
184        self.color_for_ratio(ratio)
185    }
186
187    /// Blend between gradient stops for a normalized ratio.
188    fn color_for_ratio(&self, ratio: f32) -> Color {
189        if self.stops.len() == 1 {
190            return self.stops[0].color;
191        }
192        let ratio = ratio.clamp(0.0, 1.0);
193        let mut prev = &self.stops[0];
194        for stop in &self.stops[1..] {
195            if ratio <= stop.offset {
196                let span = (stop.offset - prev.offset).max(f32::EPSILON);
197                let local = (ratio - prev.offset) / span;
198                return prev.color.blend(stop.color, local);
199            }
200            prev = stop;
201        }
202        self.stops.last().expect("gradient stops exist").color
203    }
204}

206/// A paint definition for a style channel.
207#[derive(Debug, Clone, PartialEq)]
208pub enum Paint {
209    /// Solid color fill.
210    Solid(Color),
211    /// Gradient fill.
212    Gradient(GradientSpec),
213}
214
215impl Paint {
216    /// Construct a solid paint.
217    pub fn solid(color: Color) -> Self {
218        Self::Solid(color)
219    }
220
221    /// Construct a gradient paint.
222    pub fn gradient(spec: GradientSpec) -> Self {
223        Self::Gradient(spec)
224    }
225
226    /// Return the solid color if this paint is solid.
227    pub fn solid_color(&self) -> Option<Color> {
228        match self {
229            Self::Solid(color) => Some(*color),
230            Self::Gradient(_) => None,
231        }
232    }
233
234    /// Resolve the paint at a location.
235    pub fn resolve(&self, rect: geom::Rect, point: geom::Point) -> Color {
236        match self {
237            Self::Solid(color) => *color,
238            Self::Gradient(spec) => spec.color_at(rect, point),
239        }
240    }
241
242    /// Map colors within this paint.
243    pub fn map_colors(&self, f: impl Fn(Color) -> Color) -> Self {
244        match self {
245            Self::Solid(color) => Self::Solid(f(*color)),
246            Self::Gradient(spec) => Self::Gradient(spec.map_colors(f)),
247        }
248    }
249}
250
251impl From<Color> for Paint {
252    fn from(color: Color) -> Self {
253        Self::Solid(color)
254    }
255}
256
257impl From<GradientSpec> for Paint {
258    fn from(spec: GradientSpec) -> Self {
259        Self::Gradient(spec)
260    }

479/// Split a style path into components.
480fn parse_path(path: &str) -> Vec<String> {
481    path.split('/')
482        .filter_map(|s| {
483            if !s.is_empty() {
484                Some(s.to_owned())
485            } else {
486                None
487            }
488        })
489        .collect()
490}

492/// Map of style paths to partial styles.
493#[derive(Debug)]
494pub struct StyleMap {
495    /// Path-to-style map.
496    styles: HashMap<Vec<String>, PartialStyle>,
497}
498
499impl StyleMap {
500    /// Construct a style map with defaults.
501    pub fn new() -> Self {
502        let mut cs = Self {
503            styles: HashMap::new(),
504        };
505        cs.insert_style(
506            "/",
507            PartialStyle {
508                fg: Some(Paint::Solid(Color::White)),
509                bg: Some(Paint::Solid(Color::Black)),
510                attrs: Some(AttrSet::default()),
511            },
512        );
513        cs
514    }
515
516    /// Begin a fluent rule-building chain.
517    ///
518    /// # Example
519    ///
520    /// ```ignore
521    /// style_map.rules()
522    ///     .fg("red/text", solarized::RED)
523    ///     .fg("blue/text", solarized::BLUE)
524    ///     .apply();
525    /// ```
526    pub fn rules(&mut self) -> StyleRules<'_> {
527        StyleRules {
528            map: self,
529            prefix: None,
530            pending: Vec::new(),
531        }
532    }
533
534    /// Insert a style attribute at a specified path.
535    pub fn add_attr(&mut self, path: &str, attr: Attr) {
536        let parsed = parse_path(path);
537        if let Some(ps) = self.styles.get_mut(&parsed) {
538            if let Some(attrs) = ps.attrs {
539                ps.attrs = Some(attrs.with(attr));
540            } else {
541                ps.attrs = Some(AttrSet::default().with(attr));
542            }
543        } else {
544            self.styles
545                .insert(parsed, PartialStyle::default().with_attr(attr));
546        }
547    }
548
549    /// Insert a partial style at a path.
550    fn insert_style(&mut self, path: &str, style: PartialStyle) {
551        self.styles.insert(parse_path(path), style);
552    }
553}
554
555impl Default for StyleMap {
556    fn default() -> Self {
557        Self::new()
558    }
559}

561/// A fluent builder for adding style rules to a StyleMap.
562///
563/// Created via [`StyleMap::rules()`]. Collects path/style pairs and commits
564/// them on [`.apply()`](StyleRules::apply).
565#[must_use = "call .apply() to commit rules"]
566pub struct StyleRules<'a> {
567    /// The target style map.
568    map: &'a mut StyleMap,
569    /// Optional path prefix for subsequent rules.
570    prefix: Option<String>,
571    /// Accumulated rules to be committed.
572    pending: Vec<(String, PartialStyle)>,
573}
574
575impl<'a> StyleRules<'a> {
576    /// Set the foreground paint for a path.
577    ///
578    /// If a rule already exists for this path, the foreground paint is merged
579    /// with the existing style.
580    pub fn fg(mut self, path: &str, paint: impl Into<Paint>) -> Self {
581        let full_path = self.make_path(path);
582        self.merge_pending(full_path, PartialStyle::fg(paint));
583        self
584    }
585
586    /// Set the background paint for a path.
587    ///
588    /// If a rule already exists for this path, the background paint is merged
589    /// with the existing style.
590    pub fn bg(mut self, path: &str, paint: impl Into<Paint>) -> Self {
591        let full_path = self.make_path(path);
592        self.merge_pending(full_path, PartialStyle::bg(paint));
593        self
594    }
595
596    /// Add a single attribute for a path.
597    ///
598    /// If a rule already exists for this path, the attribute is merged
599    /// with the existing style.
600    pub fn attr(mut self, path: &str, attr: Attr) -> Self {
601        let full_path = self.make_path(path);
602        self.merge_pending(full_path, PartialStyle::attrs(AttrSet::new(attr)));
603        self
604    }
605
606    /// Set all attributes for a path.
607    ///
608    /// If a rule already exists for this path, the attributes are merged
609    /// with the existing style.
610    pub fn attrs(mut self, path: &str, attrs: AttrSet) -> Self {
611        let full_path = self.make_path(path);
612        self.merge_pending(full_path, PartialStyle::attrs(attrs));
613        self
614    }
615
616    /// Apply a complete style to a path.
617    ///
618    /// If a rule already exists for this path, the style is merged
619    /// with the existing style (new values take precedence).
620    pub fn style(mut self, path: &str, style: impl Into<PartialStyle>) -> Self {
621        let full_path = self.make_path(path);
622        self.merge_pending(full_path, style.into());
623        self
624    }
625
626    /// Set the foreground paint for multiple paths.
627    ///
628    /// If a rule already exists for any path, the foreground paint is merged
629    /// with the existing style.
630    pub fn fg_all<P>(mut self, paths: &[&str], paint: P) -> Self
631    where
632        P: Into<Paint>,
633    {
634        let paint: Paint = paint.into();
635        for path in paths {
636            let full_path = self.make_path(path);
637            self.merge_pending(full_path, PartialStyle::fg(paint.clone()));
638        }
639        self
640    }
641
642    /// Set the background paint for multiple paths.
643    ///
644    /// If a rule already exists for any path, the background paint is merged
645    /// with the existing style.
646    pub fn bg_all<P>(mut self, paths: &[&str], paint: P) -> Self
647    where
648        P: Into<Paint>,
649    {
650        let paint: Paint = paint.into();
651        for path in paths {
652            let full_path = self.make_path(path);
653            self.merge_pending(full_path, PartialStyle::bg(paint.clone()));
654        }
655        self
656    }
657
658    /// Add a single attribute to multiple paths.
659    ///
660    /// If a rule already exists for any path, the attribute is merged
661    /// with the existing style.
662    pub fn attr_all(mut self, paths: &[&str], attr: Attr) -> Self {
663        for path in paths {
664            let full_path = self.make_path(path);
665            self.merge_pending(full_path, PartialStyle::attrs(AttrSet::new(attr)));
666        }
667        self
668    }
669
670    /// Set all attributes for multiple paths.
671    ///
672    /// If a rule already exists for any path, the attributes are merged
673    /// with the existing style.
674    pub fn attrs_all(mut self, paths: &[&str], attrs: AttrSet) -> Self {
675        for path in paths {
676            let full_path = self.make_path(path);
677            self.merge_pending(full_path, PartialStyle::attrs(attrs));
678        }
679        self
680    }
681
682    /// Apply a complete style to multiple paths.
683    ///
684    /// If a rule already exists for any path, the style is merged
685    /// with the existing style (new values take precedence).
686    pub fn style_all(mut self, paths: &[&str], style: impl Into<PartialStyle>) -> Self {
687        let partial = style.into();
688        for path in paths {
689            let full_path = self.make_path(path);
690            self.merge_pending(full_path, partial.clone());
691        }
692        self
693    }
694
695    /// Merge a style into the pending rules.
696    ///
697    /// If a rule with the same path exists, merge the new style into it.
698    /// Otherwise, add a new pending rule.
699    fn merge_pending(&mut self, path: String, style: PartialStyle) {
700        if let Some((_, existing)) = self.pending.iter_mut().find(|(p, _)| p == &path) {
701            *existing = style.join(existing);
702        } else {
703            self.pending.push((path, style));
704        }
705    }
706
707    /// Set a path prefix for all subsequent rules.
708    ///
709    /// Can be called multiple times; each call replaces the previous prefix.
710    pub fn prefix(mut self, prefix: &str) -> Self {
711        self.prefix = Some(prefix.to_string());
712        self
713    }
714
715    /// Clear the current prefix.
716    pub fn no_prefix(mut self) -> Self {
717        self.prefix = None;
718        self
719    }
720
721    /// Commit all pending rules to the StyleMap.
722    pub fn apply(self) {
723        for (path, style) in self.pending {
724            self.map.insert_style(&path, style);
725        }
726    }
727
728    /// Combine the current prefix with a path suffix.
729    fn make_path(&self, path: &str) -> String {
730        match &self.prefix {
731            Some(prefix) if !prefix.is_empty() && !path.is_empty() => {
732                format!("{}/{}", prefix, path)
733            }
734            Some(prefix) if !prefix.is_empty() => prefix.clone(),
735            _ => path.to_string(),
736        }
737    }
738}

740/// A hierarchical style manager.
741///
742/// `Style` objects are entered into the manager with '/'-separated paths. For
743/// example:
744///
745///   / white, black
746///   /frame -> grey, None
747///   /frame/selected -> blue, None
748///
749/// The first entry with the empty path is the global default. Every
750/// `StyleManager` is guaranteed to have a default Style object with non-None
751/// foreground and background colors, so style resolution always succeeds.
752///
753/// `Style` objects also contain text attributes.
754///
755/// During rendering, a node may push a name onto the stack of layers tracked by
756/// the `Style` object. Layers are maintained for a node and all its
757/// descendants, and `Canopy` manages poppping layers back off the stack at the
758/// appropriate time during rendering.
759///
760/// When a colour is resolved, we first try to find the specified path under
761/// each layer to the root; failing that we look up the default colours for each
762/// layer to the root.
763///
764/// So given a layer stack ["foo"], and an attempt to look up "frame/selected",
765/// we try the following lookups in order: ["foo/frame/selected",
766/// "/frame/selected", "foo", ""].
767#[derive(Debug, PartialEq, Eq, Clone)]
768pub struct StyleManager {
769    /// Current render level.
770    level: usize,
771    /// Active layer names.
772    layers: Vec<String>,
773    /// Render levels corresponding to layers.
774    layer_levels: Vec<usize>,
775}
776
777impl Default for StyleManager {
778    fn default() -> Self {
779        Self::new()
780    }
781}
782
783impl StyleManager {
784    /// Construct a new style manager.
785    pub fn new() -> Self {
786        Self {
787            level: 0,
788            layers: vec![],
789            layer_levels: vec![],
790        }
791    }
792
793    /// Reset all layers and levels.
794    pub fn reset(&mut self) {
795        self.level = 0;
796        self.layers = vec![];
797        self.layer_levels = vec![0];
798    }
799
800    /// Increment the render level.
801    pub fn push(&mut self) {
802        self.level += 1
803    }
804
805    /// Decrement the render level and pop any layers at this level.
806    pub fn pop(&mut self) {
807        if self.level != 0 {
808            while self.layer_levels.last() == Some(&self.level) {
809                self.layers.pop();
810                self.layer_levels.pop();
811            }
812            self.level -= 1;
813        }
814    }
815
816    /// Push onto the layer stack with the current render level.
817    pub fn push_layer(&mut self, name: &str) {
818        self.layers.push(name.to_owned());
819        self.layer_levels.push(self.level);
820    }
821
822    /// Resolve a style path.
823    pub fn get(&self, smap: &StyleMap, path: &str) -> Style {
824        self.resolve(smap, &self.layers, &parse_path(path))
825    }
826
827    /// Look up one suffix along a layer chain.
828    fn lookup(&self, smap: &StyleMap, layers: &[String], suffix: &[String]) -> PartialStyle {
829        let mut ret = PartialStyle::default();
830        // Look up the path on all layers to the root.
831        for i in 0..layers.len() + 1 {
832            let mut v = layers[0..layers.len() - i].to_vec();
833            v.extend(suffix.to_vec());
834            if let Some(c) = smap.styles.get(&v) {
835                ret = ret.join(c);
836                if ret.is_complete() {
837                    break;
838                }
839            }
840        }
841        ret
842    }
843
844    /// Directly resolve a style using a path and a layer specification,
845    /// ignoring `self.layers`.
846    pub(crate) fn resolve(&self, smap: &StyleMap, layers: &[String], path: &[String]) -> Style {
847        let mut ret = PartialStyle::default();
848        for i in 0..path.len() + 1 {
849            ret = ret.join(&self.lookup(smap, layers, &path[0..path.len() - i]));
850            if ret.is_complete() {
851                break;
852            }
853        }
854        ret.resolve()
855    }
856}

1126    #[test]
1127    fn gradient_interpolates_multiple_stops() {
1128        let red = Color::Rgb { r: 255, g: 0, b: 0 };
1129        let green = Color::Rgb { r: 0, g: 255, b: 0 };
1130        let blue = Color::Rgb { r: 0, g: 0, b: 255 };
1131        let spec = GradientSpec::with_stops(
1132            0.0,
1133            vec![
1134                GradientStop::new(1.0, blue),
1135                GradientStop::new(0.0, red),
1136                GradientStop::new(0.5, green),
1137            ],
1138        );
1139        let rect = geom::Rect::new(0, 0, 10, 1);
1140        let point = geom::Point { x: 4, y: 0 };
1141
1142        let width = rect.w as f32;
1143        let angle = spec.angle_deg.to_radians();
1144        let dir_x = angle.cos();
1145        let dir_y = angle.sin();
1146        let corners = [(0.0, 0.0), (width, 0.0), (0.0, 1.0), (width, 1.0)];
1147        let (min_dot, max_dot) = corners.iter().fold(
1148            (f32::INFINITY, f32::NEG_INFINITY),
1149            |(min_dot, max_dot), (x, y)| {
1150                let dot = dir_x * x + dir_y * y;
1151                (min_dot.min(dot), max_dot.max(dot))
1152            },
1153        );
1154        let local_x = point.x as f32 + 0.5;
1155        let dot = dir_x * local_x + dir_y * 0.5;
1156        let ratio = ((dot - min_dot) / (max_dot - min_dot)).clamp(0.0, 1.0);
1157        let expected = if ratio <= 0.5 {
1158            red.blend(green, ratio / 0.5)
1159        } else {
1160            green.blend(blue, (ratio - 0.5) / 0.5)
1161        };
1162
1163        assert_eq!(spec.color_at(rect, point), expected);
1164    }
