22/// A terminal cell with glyph and style.
23#[derive(Clone, Debug, PartialEq)]
24pub struct Cell {
25    /// Base glyph character.
26    pub ch: char,
27    /// Additional grapheme characters stored with the base glyph.
28    pub suffix: String,
29    /// Style applied to the cell.
30    pub style: Style,
31    /// True when this cell continues a wide glyph from the previous column.
32    pub continuation: bool,
33}
34
35impl Cell {
36    /// Construct a cell containing a single glyph.
37    fn new(ch: char, style: Style) -> Self {
38        Self {
39            ch,
40            suffix: String::new(),
41            style,
42            continuation: false,
43        }
44    }
45
46    /// Construct an empty cell.
47    fn empty(style: Style) -> Self {
48        Self {
49            ch: NULL,
50            suffix: String::new(),
51            style,
52            continuation: false,
53        }
54    }
55
56    /// Construct a continuation cell for a wide glyph.
57    fn continuation(style: Style) -> Self {
58        Self {
59            ch: NULL,
60            suffix: String::new(),
61            style,
62            continuation: true,
63        }
64    }
65
66    /// Return true when the cell is empty.
67    pub(crate) fn is_empty(&self) -> bool {
68        self.ch == NULL && self.suffix.is_empty() && !self.continuation
69    }
70
71    /// Return a display character for tests and debugging.
72    #[cfg(any(test, feature = "testing"))]
73    pub(crate) fn display_char(&self) -> char {
74        if self.is_empty() || self.continuation {
75            NULL
76        } else {
77            self.ch
78        }
79    }
80
81    /// Append this cell's renderable text to the output buffer.
82    fn push_text(&self, out: &mut String) {
83        if self.continuation {
84            return;
85        }
86        if self.is_empty() {
87            out.push(' ');
88            return;
89        }
90        out.push(self.ch);
91        out.push_str(&self.suffix);
92    }
93}

105    /// Construct a buffer filled with the given character and style.
106    pub fn new(size: impl Into<Expanse>, ch: char, style: Style) -> Self {
107        let size = size.into();
108        let cell = Cell::new(ch, style);

114    /// Create an empty TermBuf filled with NULL characters.

116        let size = size.into();

119            size,

124    /// Create an empty TermBuf filled with NULL characters.
125    pub fn empty(size: impl Into<Expanse>) -> Self {
126        let default_style = Style {
127            fg: Color::White,
128            bg: Color::Black,
129            attrs: AttrSet::default(),
130        };
131        Self::empty_with_style(size, default_style)
132    }

134    /// Copy non-empty cells from a rectangle of another TermBuf into this one
135    pub fn copy(&mut self, src: &Self, rect: Rect) {
136        if src.size != self.size {
137            return;
138        }
139
140        // Intersect the rectangle with our bounds
141        if let Some(isec) = self.rect().intersect(&rect) {
142            for y in isec.tl.y..isec.tl.y + isec.h {
143                for x in isec.tl.x..isec.tl.x + isec.w {
144                    let p = Point { x, y };
145                    if let Some(cell) = src.get(p)
146                        && !cell.is_empty()
147                        && let Some(i) = self.idx(p)
148                    {
149                        self.cells[i] = cell.clone();
150                    }
151                }
152            }
153        }
154    }
155
156    /// Copy non-empty cells from a source TermBuf into a destination rectangle
157    pub fn copy_to_rect(&mut self, src: &Self, dest_rect: Rect) {
158        // The source buffer represents content to be placed at dest_rect
159        // We need to map from source coordinates to destination coordinates
160
161        // Intersect the destination rectangle with our bounds
162        if let Some(clipped_dest) = self.rect().intersect(&dest_rect) {
163            // Calculate the offset into the source buffer based on clipping
164            let src_offset_x = (clipped_dest.tl.x - dest_rect.tl.x) as i32;
165            let src_offset_y = (clipped_dest.tl.y - dest_rect.tl.y) as i32;
166
167            // Copy the visible portion
168            for dy in 0..clipped_dest.h {
169                for dx in 0..clipped_dest.w {
170                    let src_x = (dx as i32 + src_offset_x) as u32;
171                    let src_y = (dy as i32 + src_offset_y) as u32;
172                    let src_p = Point { x: src_x, y: src_y };
173
174                    if let Some(cell) = src.get(src_p)
175                        && !cell.is_empty()
176                    {
177                        let dest_p = Point {
178                            x: clipped_dest.tl.x + dx,
179                            y: clipped_dest.tl.y + dy,
180                        };
181                        if let Some(i) = self.idx(dest_p) {
182                            self.cells[i] = cell.clone();
183                        }
184                    }
185                }
186            }
187        }
188    }

200    /// Convert a point into a cell index.
201    fn idx(&self, p: Point) -> Option<usize> {
202        if self.rect().contains_point(p) {
203            Some(p.y as usize * self.size.w as usize + p.x as usize)
204        } else {
205            None

209    /// Write a cell at a specific point.
210    pub(crate) fn put(&mut self, p: Point, ch: char, style: Style) {
211        if let Some(i) = self.idx(p) {
212            self.cells[i] = Cell::new(ch, style);
213        }
214    }

216    /// Write a grapheme cluster at a specific point.
217    fn put_grapheme(&mut self, p: Point, grapheme: &str, style: Style) {
218        if let Some(i) = self.idx(p) {
219            let mut chars = grapheme.chars();
220            let ch = chars.next().unwrap_or(' ');
221            let suffix: String = chars.collect();
222            self.cells[i] = Cell {
223                ch,
224                suffix,
225                style,
226                continuation: false,
227            };
228        }
229    }

232    fn put_continuation(&mut self, p: Point, style: Style) {
233        if let Some(i) = self.idx(p) {
234            self.cells[i] = Cell::continuation(style);
235        }
236    }

238    /// Fill a rectangle with a glyph and style.
239    pub fn fill(&mut self, style: &Style, r: Rect, ch: char) {
240        if let Some(isec) = self.rect().intersect(&r) {
241            for y in isec.tl.y..isec.tl.y + isec.h {
242                for x in isec.tl.x..isec.tl.x + isec.w {
243                    self.put(Point { x, y }, ch, style.clone());
244                }
245            }
246        }
247    }

250    pub fn fill_empty(&mut self, ch: char, style: &Style) {
251        for i in 0..self.cells.len() {
252            if self.cells[i].is_empty() {
253                self.cells[i] = Cell::new(ch, style.clone());
254            }
255        }

258    /// Overlay a cursor on a cell by adjusting its style.
259    pub fn overlay_cursor(&mut self, location: Point, shape: cursor::CursorShape) {
260        let Some(idx) = self.idx(location) else {
261            return;
262        };
263        let mut cell = self.cells[idx].clone();
264        match shape {
265            cursor::CursorShape::Underscore => {
266                cell.style.attrs = cell.style.attrs.with(Attr::Underline);
267            }
268            cursor::CursorShape::Block | cursor::CursorShape::Line => {
269                mem::swap(&mut cell.style.fg, &mut cell.style.bg);
270            }
271        }
272        if cell.is_empty() || cell.continuation {
273            cell.ch = ' ';
274            cell.suffix.clear();
275            cell.continuation = false;
276        }
277        self.cells[idx] = cell;
278    }

292    /// Draw text clipped to the given line.
293    pub fn text(&mut self, style: &Style, l: Line, txt: &str) {
294        if let Some(isec) = self.rect().intersect(&l.rect()) {
295            let offset = isec.tl.x.saturating_sub(l.tl.x) as usize;
296            let max = isec.w as usize;
297            let (out, _) = text::slice_by_columns(txt, offset, max);
298            let mut col = 0usize;
299            let mut x = isec.tl.x;
300
301            for grapheme in out.graphemes(true) {
302                let width = text::grapheme_width(grapheme);
303                if width == 0 {
304                    continue;
305                }
306                if col + width > max {
307                    break;
308                }
309
310                self.put_grapheme(Point { x, y: isec.tl.y }, grapheme, style.clone());
311                for i in 1..width {
312                    self.put_continuation(
313                        Point {
314                            x: x + i as u32,
315                            y: isec.tl.y,
316                        },
317                        style.clone(),
318                    );
319                }
320                x += width as u32;
321                col += width;
322            }
323
324            for i in col..max {
325                self.put(
326                    Point {
327                        x: isec.tl.x + i as u32,
328                        y: isec.tl.y,
329                    },
330                    ' ',
331                    style.clone(),
332                );
333            }
334        }
335    }

341    /// Diff this terminal buffer against a previous state, emitting changes
342    /// to the provided render backend.
343    pub fn diff<R: RenderBackend>(&self, prev: &Self, backend: &mut R) -> Result<()> {
344        let mut wrote = false;
345        if self.size != prev.size {
346            return self.render(backend);
347        }
348        if backend.supports_line_shift()
349            && let Some(shift) = detect_row_shift(self, prev, MAX_ROW_SHIFT)
350        {
351            let last_row = self.size.h.saturating_sub(1);
352            backend.shift_lines(0, last_row, shift)?;
353            let width = self.size.w as usize;
354            let count = shift.unsigned_abs();
355            if shift > 0 {
356                for y in 0..count {
357                    let row_start = y as usize * width;
358                    let row_end = row_start + width;
359                    let row = &self.cells[row_start..row_end];
360                    render_line_range(backend, row, y, 0, width)?;
361                }
362            } else if shift < 0 {
363                let start = self.size.h.saturating_sub(count);
364                for y in start..self.size.h {
365                    let row_start = y as usize * width;
366                    let row_end = row_start + width;
367                    let row = &self.cells[row_start..row_end];
368                    render_line_range(backend, row, y, 0, width)?;
369                }
370            }
371            backend.flush()?;
372            return Ok(());
373        }
374        let width = self.size.w as usize;
375        let can_shift = backend.supports_char_shift();
376        for y in 0..self.size.h {
377            let row_start = y as usize * width;
378            let row_end = row_start + width;
379            let current_row = &self.cells[row_start..row_end];
380            let prev_row = &prev.cells[row_start..row_end];
381
382            if current_row == prev_row {
383                continue;
384            }
385
386            if can_shift
387                && let Some(shift) = detect_line_shift(current_row, prev_row, MAX_LINE_SHIFT)
388            {
389                let gap = if shift > 0 {
390                    shift as usize
391                } else {
392                    (-shift) as usize
393                };
394                if gap > 0 && gap < width {
395                    backend.shift_chars(Point { x: 0, y }, shift)?;
396                    if shift > 0 {
397                        render_line_range(backend, current_row, y, 0, gap)?;
398                    } else {
399                        let start = width.saturating_sub(gap);
400                        render_line_range(backend, current_row, y, start, gap)?;
401                    }
402                    wrote = true;
403                    continue;
404                }
405            }
406
407            let mut x = 0usize;
408            while x < width {
409                if current_row[x] == prev_row[x] {
410                    x += 1;
411                    continue;
412                }
413
414                let style = &current_row[x].style;
415                let start_x = x;
416                let mut text = String::new();
417                while x < width {
418                    let cell = &current_row[x];
419                    if cell == &prev_row[x] || cell.style != *style {
420                        break;
421                    }
422                    cell.push_text(&mut text);
423                    x += 1;
424                }
425                backend.style(style)?;
426                backend.text(
427                    Point {
428                        x: start_x as u32,
429                        y,
430                    },
431                    &text,
432                )?;
433                wrote = true;
434            }
435        }
436        if wrote {
437            backend.flush()?;
438        }
439        Ok(())
440    }

442    /// Render this terminal buffer in full using the provided backend,
443    /// batching runs of text with the same style.
444    pub fn render<R: RenderBackend>(&self, backend: &mut R) -> Result<()> {
445        let mut wrote = false;
446        for y in 0..self.size.h {
447            let mut x = 0;
448            while x < self.size.w {
449                let idx = y as usize * self.size.w as usize + x as usize;
450                let cell = &self.cells[idx];
451                let style = cell.style.clone();
452                let start_x = x;
453                let mut text = String::new();
454                while x < self.size.w {
455                    let idx2 = y as usize * self.size.w as usize + x as usize;
456                    let ccell = &self.cells[idx2];
457                    if ccell.style == style {
458                        ccell.push_text(&mut text);
459                        x += 1;
460                    } else {
461                        break;
462                    }
463                }
464                backend.style(&style)?;
465                backend.text(Point { x: start_x, y }, &text)?;
466                wrote = true;
467            }
468        }
469        if wrote {
470            backend.flush()?;
471        }
472        Ok(())
473    }

476/// Check whether two lines are identical up to a horizontal shift.
477fn detect_line_shift(current: &[Cell], prev: &[Cell], max_shift: usize) -> Option<i32> {
478    let width = current.len();
479    if width == 0 || width != prev.len() {
480        return None;
481    }
482
483    let max = max_shift.min(width.saturating_sub(1));
484    if max == 0 {
485        return None;
486    }
487
488    for shift in 1..=max {
489        if line_matches_shift(current, prev, shift as i32) {
490            return Some(shift as i32);
491        }
492        if line_matches_shift(current, prev, -(shift as i32)) {
493            return Some(-(shift as i32));
494        }
495    }
496    None
497}

499/// Check whether two buffers are identical up to a vertical shift.
500fn detect_row_shift(current: &TermBuf, prev: &TermBuf, max_shift: usize) -> Option<i32> {
501    let height = current.size.h as i32;
502    if height == 0 || height != prev.size.h as i32 {
503        return None;
504    }
505
506    let max = max_shift.min(height.saturating_sub(1) as usize);
507    if max == 0 {
508        return None;
509    }
510
511    for shift in 1..=max {
512        let shift = shift as i32;
513        if buffer_matches_shift(current, prev, shift) {
514            return Some(shift);
515        }
516        if buffer_matches_shift(current, prev, -shift) {
517            return Some(-shift);
518        }
519    }
520    None
521}

523/// Determine whether two buffers match for a given vertical shift.
524fn buffer_matches_shift(current: &TermBuf, prev: &TermBuf, shift: i32) -> bool {
525    let height = current.size.h as i32;
526    let width = current.size.w as usize;
527    if shift == 0 || shift.unsigned_abs() as i32 >= height {
528        return false;
529    }
530
531    if shift > 0 {
532        for y in shift..height {
533            let row = y as usize * width;
534            let prev_row = (y - shift) as usize * width;
535            if current.cells[row..row + width] != prev.cells[prev_row..prev_row + width] {
536                return false;
537            }
538        }
539    } else {
540        let limit = height + shift;
541        for y in 0..limit {
542            let row = y as usize * width;
543            let prev_row = (y - shift) as usize * width;
544            if current.cells[row..row + width] != prev.cells[prev_row..prev_row + width] {
545                return false;
546            }
547        }
548    }
549    true
550}

552/// Determine whether the current line matches the previous line shifted by `shift`.
553fn line_matches_shift(current: &[Cell], prev: &[Cell], shift: i32) -> bool {
554    let width = current.len();
555    if width == 0 || width != prev.len() || shift == 0 {
556        return false;
557    }
558
559    if shift > 0 {
560        let shift = shift as usize;
561        if shift >= width {
562            return false;
563        }
564        current[shift..] == prev[..width - shift]
565    } else {
566        let shift = (-shift) as usize;
567        if shift >= width {
568            return false;
569        }
570        current[..width - shift] == prev[shift..]
571    }
572}

574/// Render a slice of a single line using style runs from the current buffer.
575fn render_line_range<R: RenderBackend>(
576    backend: &mut R,
577    row: &[Cell],
578    y: u32,
579    start: usize,
580    len: usize,
581) -> Result<()> {
582    if len == 0 || start >= row.len() {
583        return Ok(());
584    }
585
586    let end = start.saturating_add(len).min(row.len());
587    let mut x = start;
588    while x < end {
589        let style = &row[x].style;
590        let run_start = x;
591        let mut text = String::new();
592        while x < end && row[x].style == *style {
593            row[x].push_text(&mut text);
594            x += 1;
595        }
596        backend.style(style)?;
597        backend.text(
598            Point {
599                x: run_start as u32,
600                y,
601            },
602            &text,
603        )?;
604    }
605    Ok(())
606}
