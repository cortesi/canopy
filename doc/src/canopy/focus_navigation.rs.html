3/// Information about a focusable node
4#[derive(Debug, Clone)]
5pub(crate) struct FocusableNode {
6    pub(crate) id: NodeId,
7    pub(crate) rect: Rect,
8}
9
10/// Collect all focusable nodes with their screen rectangles
11pub fn collect_focusable_nodes(
12    ctx: &dyn Context,
13    root: &mut dyn Node,
14) -> Result<(Vec<FocusableNode>, Option<(NodeId, Rect)>)> {
15    let mut nodes = Vec::new();
16    let mut focused = None;
17
18    fn recurse(
19        ctx: &dyn Context,
20        node: &mut dyn Node,
21        offset: (i16, i16),
22        acc: &mut Vec<FocusableNode>,
23        focused: &mut Option<(NodeId, Rect)>,
24    ) -> Result<()> {
25        if node.is_hidden() {
26            return Ok(());
27        }
28
29        let vp = node.vp();
30        if vp.view().is_zero() {
31            return Ok(());
32        }
33
34        let pos = (
35            offset.0 + vp.position().x as i16,
36            offset.1 + vp.position().y as i16,
37        );
38        let screen = vp.view().shift(pos.0, pos.1);
39
40        if node.accept_focus() {
41            if focused.is_none() && ctx.is_focused(node) {
42                *focused = Some((node.id(), screen));
43            }
44            acc.push(FocusableNode {
45                id: node.id(),
46                rect: screen,
47            });
48        }
49
50        node.children(&mut |child| {
51            recurse(ctx, child, pos, acc, focused)?;
52            Ok(())
53        })?;
54
55        Ok(())
56    }
57
58    recurse(ctx, root, (0, 0), &mut nodes, &mut focused)?;
59    Ok((nodes, focused))
60}

62/// Find the best focus target in the specified direction
63pub fn find_focus_target(
64    current_rect: Rect,
65    direction: Direction,
66    candidates: &[FocusableNode],
67    current_id: &NodeId,
68) -> Option<NodeId> {
69    // Debug: print current rect
70    #[cfg(test)]
71    {
72        eprintln!("find_focus_target: current_rect = {current_rect:?}, direction = {direction:?}");
73        eprintln!("Total candidates: {}", candidates.len());
74    }
75
76    // Filter out the current node and nodes that don't make sense for the direction
77    let valid_candidates: Vec<&FocusableNode> = candidates
78        .iter()
79        .filter(|n| &n.id != current_id)
80        .filter(|n| {
81            let in_dir = is_in_direction(&current_rect, &n.rect, direction);
82            #[cfg(test)]
83            {
84                if !in_dir
85                    && matches!(direction, Direction::Right)
86                    && n.rect.tl.x > current_rect.tl.x
87                {
88                    eprintln!("  Candidate at {:?} rejected by is_in_direction", n.rect);
89                }
90            }
91            in_dir
92        })
93        .collect();
94
95    #[cfg(test)]
96    eprintln!(
97        "Valid candidates after filtering: {}",
98        valid_candidates.len()
99    );
100
101    if valid_candidates.is_empty() {
102        return None;
103    }
104
105    // Find the best candidate based on direction
106
107    valid_candidates
108        .into_iter()
109        .min_by_key(|n| distance_score(&current_rect, &n.rect, direction))
110        .map(|n| n.id.clone())
111}

113/// Check if target is in the specified direction from source
114fn is_in_direction(source: &Rect, target: &Rect, direction: Direction) -> bool {
115    // Don't consider the exact same rectangle
116    if source == target {
117        return false;
118    }
119
120    match direction {
121        Direction::Right => {
122            // Accept nodes that are:
123            // 1. Strictly to the right (no overlap)
124            // 2. On the same row (overlapping Y) but extending further right
125            if target.tl.x >= source.tl.x + source.w {
126                // Strictly to the right
127                true
128            } else {
129                // Check if on same row and extends further right
130                let same_row = (target.tl.y < source.tl.y + source.h)
131                    && (target.tl.y + target.h > source.tl.y);
132                let extends_right = target.tl.x + target.w > source.tl.x + source.w;
133                same_row && extends_right
134            }
135        }
136        Direction::Left => {
137            // Accept nodes that are:
138            // 1. Strictly to the left (no overlap)
139            // 2. On the same row (overlapping Y) but extending further left
140            if target.tl.x + target.w <= source.tl.x {
141                // Strictly to the left
142                true
143            } else {
144                // Check if on same row and extends further left
145                let same_row = (target.tl.y < source.tl.y + source.h)
146                    && (target.tl.y + target.h > source.tl.y);
147                let extends_left = target.tl.x < source.tl.x;
148                same_row && extends_left
149            }
150        }
151        Direction::Down => {
152            // Accept nodes that are:
153            // 1. Strictly below (no overlap)
154            // 2. On the same column (overlapping X) but extending further down
155            if target.tl.y >= source.tl.y + source.h {
156                // Strictly below
157                true
158            } else {
159                // Check if in same column and extends further down
160                let same_col = (target.tl.x < source.tl.x + source.w)
161                    && (target.tl.x + target.w > source.tl.x);
162                let extends_down = target.tl.y + target.h > source.tl.y + source.h;
163                same_col && extends_down
164            }
165        }
166        Direction::Up => {
167            // Accept nodes that are:
168            // 1. Strictly above (no overlap)
169            // 2. On the same column (overlapping X) but extending further up
170            if target.tl.y + target.h <= source.tl.y {
171                // Strictly above
172                true
173            } else {
174                // Check if in same column and extends further up
175                let same_col = (target.tl.x < source.tl.x + source.w)
176                    && (target.tl.x + target.w > source.tl.x);
177                let extends_up = target.tl.y < source.tl.y;
178                same_col && extends_up
179            }
180        }
181    }
182}

184/// Calculate a distance score for ranking candidates
185/// Lower scores are better
186fn distance_score(source: &Rect, target: &Rect, direction: Direction) -> u32 {
187    match direction {
188        Direction::Right => {
189            // For rightward movement in a grid:
190            // 1. Prefer nodes on the same row (small y difference)
191            // 2. Among those, prefer the closest one to the right
192            let y_diff = source.tl.y.abs_diff(target.tl.y) as u32;
193            let x_dist = target.tl.x.saturating_sub(source.tl.x) as u32;
194
195            // If on same row (y_diff == 0), prioritize by x distance
196            // Otherwise, prioritize by row difference, then x distance
197            if y_diff == 0 {
198                x_dist
199            } else {
200                // Not on same row - add a large penalty
201                100000 + y_diff * 1000 + x_dist
202            }
203        }
204        Direction::Left => {
205            // For leftward movement in a grid:
206            // 1. Prefer nodes on the same row (small y difference)
207            // 2. Among those, prefer the closest one to the left
208            let y_diff = source.tl.y.abs_diff(target.tl.y) as u32;
209            let x_dist = source.tl.x.saturating_sub(target.tl.x) as u32;
210
211            // If on same row (y_diff == 0), prioritize by x distance
212            // Otherwise, prioritize by row difference, then x distance
213            if y_diff == 0 {
214                x_dist
215            } else {
216                // Not on same row - add a large penalty
217                100000 + y_diff * 1000 + x_dist
218            }
219        }
220        Direction::Down => {
221            // For downward movement in a grid:
222            // 1. Prefer nodes in the same column (small x difference)
223            // 2. Among those, prefer the closest one below
224            let x_diff = source.tl.x.abs_diff(target.tl.x) as u32;
225            let y_dist = target.tl.y.saturating_sub(source.tl.y) as u32;
226
227            // If in same column (x_diff == 0), prioritize by y distance
228            // Otherwise, prioritize by column difference, then y distance
229            if x_diff == 0 {
230                y_dist
231            } else {
232                // Not in same column - add a penalty, but less than for horizontal movement
233                // since we often want to find the closest node below even if not perfectly aligned
234                10000 + x_diff * 100 + y_dist
235            }
236        }
237        Direction::Up => {
238            // For upward movement in a grid:
239            // 1. Prefer nodes in the same column (small x difference)
240            // 2. Among those, prefer the closest one above
241            let x_diff = source.tl.x.abs_diff(target.tl.x) as u32;
242            let y_dist = source.tl.y.saturating_sub(target.tl.y) as u32;
243
244            // If in same column (x_diff == 0), prioritize by y distance
245            // Otherwise, prioritize by column difference, then y distance
246            if x_diff == 0 {
247                y_dist
248            } else {
249                // Not in same column - add a penalty
250                10000 + x_diff * 100 + y_dist
251            }
252        }
253    }
