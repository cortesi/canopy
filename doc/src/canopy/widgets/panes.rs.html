8/// Panes manages a set of child nodes arranged in a 2d grid.
9#[derive(StatefulNode)]
10pub struct Panes<N: Node> {
11    pub children: Vec<Vec<N>>,
12    pub state: NodeState,
13}
14
15#[derive_commands]
16impl<N> Panes<N>
17where
18    N: Node,
19{
20    pub fn new(n: N) -> Self {
21        Panes {
22            children: vec![vec![n]],
23            state: NodeState::default(),
24        }
25    }
26
27    /// Get the offset of the current focus in the children vector.
28    pub fn focus_coords(&mut self, c: &Canopy) -> Option<(usize, usize)> {
29        for (x, col) in self.children.iter_mut().enumerate() {
30            for (y, row) in col.iter_mut().enumerate() {
31                if c.is_on_focus_path(row) {
32                    return Some((x, y));
33                }
34            }
35        }
36        None
37    }
38
39    /// Delete the focus node. If a column ends up empty, it is removed.
40    pub fn delete_focus(&mut self, c: &mut Canopy) -> Result<()> {
41        if let Some((x, y)) = self.focus_coords(c) {
42            c.focus_next(self);
43            self.children[x].remove(y);
44            if self.children[x].is_empty() {
45                self.children.remove(x);
46            }
47            c.taint_tree(self);
48        }
49        Ok(())
50    }
51
52    /// Insert a node, splitting vertically. If we have a focused node, the new
53    /// node is inserted in a row beneath it. If not, a new column is added.
54    pub fn insert_row(&mut self, c: &mut Canopy, n: N)
55    where
56        N: Node,
57    {
58        if let Some((x, y)) = self.focus_coords(c) {
59            self.children[x].insert(y, n);
60        } else {
61            self.children.push(vec![n]);
62        }
63        c.taint_tree(self);
64    }
65
66    /// Insert a node in a new column. If we have a focused node, the new node
67    /// is added in a new column to the right.
68    pub fn insert_col(&mut self, c: &mut Canopy, mut n: N) -> Result<()>
69    where
70        N: Node,
71    {
72        let coords = self.focus_coords(c);
73        c.focus_next(&mut n);
74        if let Some((x, _)) = coords {
75            self.children.insert(x + 1, vec![n])
76        } else {
77            self.children.push(vec![n])
78        }
79        c.taint_tree(self);
80        Ok(())
81    }
82
83    /// Returns the shape of the current child grid
84    fn shape(&self) -> Vec<u16> {
85        let mut ret = vec![];
86        for i in &self.children {
87            ret.push(i.len() as u16)
88        }
89        ret
90    }
91}
92
93impl<N: Node> Node for Panes<N> {
94    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
95        for col in &mut self.children {
96            for row in col {
97                f(row)?
98            }
99        }
100        Ok(())
101    }
102
103    fn layout(&mut self, l: &Layout, _: Expanse) -> Result<()> {
104        let lst = self.vp().screen_rect().split_panes(&self.shape())?;
105        for (ci, col) in self.children.iter_mut().enumerate() {
106            for (ri, row) in col.iter_mut().enumerate() {
107                l.place(row, lst[ci][ri])?;
108            }
109        }
110        Ok(())
111    }
112}

119    #[test]
120    fn tlayout() -> Result<()> {
121        let mut c = Canopy::new();
122        let tn = Ba::new();
123        let mut p: Panes<Ba> = Panes::new(tn);
124        let l = Layout {};
125        let e = Expanse { w: 100, h: 100 };
126
127        p.layout(&l, e)?;
128
129        assert_eq!(p.shape(), vec![1]);
130        let tn = Ba::new();
131        p.insert_col(&mut c, tn)?;
132        p.layout(&l, e)?;
133
134        assert_eq!(p.shape(), vec![1, 1]);
135        c.set_focus(&mut p.children[0][0].a);
136        p.layout(&l, e)?;
137
138        let tn = Ba::new();
139        assert_eq!(p.focus_coords(&c), Some((0, 0)));
140        p.insert_row(&mut c, tn);
141        p.layout(&l, e)?;
142
143        assert_eq!(p.shape(), vec![2, 1]);
144
145        c.set_focus(&mut p.children[1][0].a);
146        assert_eq!(p.focus_coords(&c), Some((1, 0)));
147        Ok(())
