9#[derive(crate::core::StatefulNode)]
10/// Internal editor view node.
11pub struct EditorView {
12    /// Node state.
13    state: NodeState,
14    /// Core editor state and logic.
15    core: core::Core,
16}
17
18#[derive_commands]
19impl EditorView {
20    /// Construct a new editor view.
21    pub fn new(txt: &str) -> Self {
22        Self {
23            state: NodeState::default(),
24            core: core::Core::new(txt),
25        }
26    }
27}
28
29impl Node for EditorView {
30    fn cursor(&self) -> Option<cursor::Cursor> {
31        let p = self.core.cursor_position();
32        p.map(|p| cursor::Cursor {
33            location: p,
34            shape: cursor::CursorShape::Block,
35            blink: true,
36        })
37    }
38
39    fn accept_focus(&mut self) -> bool {
40        true
41    }
42
43    fn render(&mut self, _: &dyn Context, r: &mut Render) -> Result<()> {
44        let vo = self.vp().view();
45        let sr = self.vp().screen_rect();
46        self.core.resize_window(sr.w as usize, sr.h as usize);
47        for (i, s) in self.core.window_text().iter().enumerate() {
48            if let Some(t) = s {
49                r.text("text", vo.line(i as u32), t)?;
50            }
51        }
52        Ok(())
53    }
54
55    fn layout(&mut self, _l: &Layout, sz: Expanse) -> Result<()> {
56        let outer = Expanse::new(sz.w, self.core.wrapped_height() as u32);
57        self.fit_size(outer, sz);
58        Ok(())
59    }
60}

62/// A simple editor
63#[derive(crate::core::StatefulNode)]
64pub struct Editor {
65    /// Node state.
66    state: NodeState,
67    /// Editor view node.
68    view: EditorView,
69}
70
71#[derive_commands]
72impl Editor {
73    /// Construct a new editor with the provided text.
74    pub fn new(txt: &str) -> Self {
75        Self {
76            state: NodeState::default(),
77            view: EditorView::new(txt),
78        }
79    }
80
81    /// Move the cursor left or right.
82    #[command]
83    fn cursor_shift(&mut self, _: &dyn Context, n: isize) {
84        self.view.core.cursor_shift(n);
85    }
86
87    /// Move the cursor up or down in the chunk list.
88    #[command]
89    fn cursor_shift_chunk(&mut self, _: &dyn Context, n: isize) {
90        self.view.core.cursor_shift_chunk(n);
91    }
92
93    /// Move the cursor up or down by visual line.
94    #[command]
95    fn cursor_shift_lines(&mut self, _: &dyn Context, n: isize) {
96        self.view.core.cursor_shift_lines(n);
97    }
98}
