65    /// Build an activation invocation that includes the row index.
66    fn invocation_with_index(&self, index: usize) -> CommandInvocation {
67        let args = match &self.command.args {
68            CommandArgs::Positional(values) => {
69                let mut out = values.clone();
70                out.push(index.to_arg_value());
71                CommandArgs::Positional(out)
72            }
73            CommandArgs::Named(values) => {
74                let mut out = values.clone();
75                out.insert("index".to_string(), index.to_arg_value());
76                CommandArgs::Named(out)
77            }
78        };
79        CommandInvocation {
80            id: self.command.id,
81            args,
82        }
83    }

97/// Trait for widgets that can be selected in a list.
98///
99/// Items in a `List` must implement this trait so the list can manage
100/// their selection state. Selection is independent of focus - an item
101/// remains selected even when the list loses focus.
102pub trait Selectable: Widget {
103    /// Set the selection state of this item.
104    fn set_selected(&mut self, selected: bool);
105}

107/// A typed list container for widget items.
108///
109/// List items are actual widgets in the tree, enabling composition and focus management.
110/// The list arranges items vertically and supports scrolling.
111///
112/// Items must implement the [`Selectable`] trait so the list can manage their
113/// selection state independently of focus.
114pub struct List<W: Selectable> {
115    /// Item widget node IDs in order.
116    items: Vec<TypedId<W>>,
117    /// Currently selected item index.
118    selected: Option<usize>,
119    /// Optional list-level selection indicator.
120    selection_indicator: Option<SelectionIndicator>,
121    /// Optional activation command configuration.
122    on_activate: Option<ListActivateConfig>,
123    /// Pending activation state while handling clicks.
124    pending_activate: Option<PendingActivate>,
125    /// Marker for the widget type.
126    _marker: PhantomData<W>,
127}
128
129impl<W: Selectable> Default for List<W> {
130    fn default() -> Self {
131        Self::new()
132    }

220    /// Append an item widget to the end of the list.
221    pub fn append(&mut self, ctx: &mut dyn Context, widget: W) -> Result<TypedId<W>>
222    where
223        W: 'static,
224    {
225        let id = TypedId::new(ctx.add_child(widget)?);
226        self.items.push(id);
227
228        // Auto-select and focus if this is the first item
229        if self.selected.is_none() {
230            self.update_selection(ctx, Some(self.items.len() - 1));
231            ctx.set_focus(id.into());
232        }
233
234        Ok(id)
235    }

237    /// Insert an item widget at the specified index.
238    pub fn insert(&mut self, ctx: &mut dyn Context, index: usize, widget: W) -> Result<TypedId<W>>
239    where
240        W: 'static,
241    {
242        let clamped = index.min(self.items.len());
243        let was_empty = self.selected.is_none();
244        let id = TypedId::new(ctx.add_child(widget)?);
245        self.items.insert(clamped, id);
246
247        // Adjust selection if inserting before current selection
248        if let Some(sel) = self.selected {
249            if clamped <= sel {
250                // Just update index, don't change which item is selected
251                self.selected = Some(sel + 1);
252            }
253        } else if !self.items.is_empty() {
254            self.update_selection(ctx, Some(0));
255        }
256
257        // Focus first item if this was an empty list
258        if was_empty && let Some(first_id) = self.items.first() {
259            ctx.set_focus((*first_id).into());
260        }
261
262        // Sync children order with the arena
263        self.sync_children(ctx)?;
264
265        Ok(id)
266    }

270        if index >= self.items.len() {
271            return Ok(false);
272        }

274        let id = self.items.remove(index);

276        self.repair_selection_after_remove(ctx, index);

320    /// Update selection to a new index, managing item selection states.
321    fn update_selection(&mut self, ctx: &mut dyn Context, new_selected: Option<usize>) {
322        // Clear old selection
323        if let Some(old_idx) = self.selected
324            && let Some(old_id) = self.items.get(old_idx).copied()
325        {
326            ctx.with_widget(old_id.into(), |w: &mut W, _| {
327                w.set_selected(false);
328                Ok(())
329            })
330            .ok();
331        }
332
333        // Set new selection
334        if let Some(new_idx) = new_selected
335            && let Some(new_id) = self.items.get(new_idx).copied()
336        {
337            ctx.with_widget(new_id.into(), |w: &mut W, _| {
338                w.set_selected(true);
339                Ok(())
340            })
341            .ok();
342        }
343
344        self.selected = new_selected;
345    }

347    /// Repair selection and focus after removing an item.
348    fn repair_selection_after_remove(&mut self, ctx: &mut dyn Context, index: usize) {
349        if let Some(sel) = self.selected {
350            if index < sel {
351                self.selected = Some(sel - 1);
352                return;
353            }
354            if index == sel {
355                let new_sel = if self.items.is_empty() {
356                    None
357                } else {
358                    Some(sel.min(self.items.len() - 1))
359                };
360                self.update_selection(ctx, new_sel);
361                if new_sel.is_some() {
362                    self.focus_selected(ctx);
363                }
364            }
365        }
366    }

390    /// Move selection by a signed offset.
391    #[command]
392    pub fn select_by(&mut self, c: &mut dyn Context, delta: i32) {
393        if self.items.is_empty() {
394            return;
395        }
396        let current = self.selected.unwrap_or(0);
397        let next = if delta.is_negative() {
398            current.saturating_sub(delta.unsigned_abs() as usize)
399        } else {
400            current.saturating_add(delta as usize)
401        };
402        self.update_selection(c, Some(next.min(self.items.len() - 1)));
403        self.focus_selected(c);
404        self.ensure_selected_visible(c);
405    }

419    /// Handle a mouse click within the list.
420    fn handle_click(&mut self, c: &mut dyn Context, event: mouse::MouseEvent) -> bool {
421        match event.action {
422            mouse::Action::Down if event.button == mouse::Button::Left => {
423                let Some(index) = self.index_at_location(c, event.location) else {
424                    return false;
425                };
426                self.select(c, index);
427                self.focus_selected(c);
428                self.ensure_selected_visible(c);
429                if self.on_activate.is_some() {
430                    self.pending_activate = Some(PendingActivate {
431                        index,
432                        origin: event.location,
433                        dragged: false,
434                    });
435                    c.capture_mouse();
436                }
437                true
438            }
439            mouse::Action::Drag if event.button == mouse::Button::Left => {
440                if let Some(pending) = self.pending_activate.as_mut()
441                    && let Some(config) = self.on_activate.as_ref()
442                {
443                    if drag_exceeded(pending.origin, event.location, config.drag_threshold) {
444                        pending.dragged = true;
445                    }
446                    return true;
447                }
448                false
449            }
450            mouse::Action::Up if event.button == mouse::Button::Left => {
451                let pending = self.pending_activate.take();
452                if let Some(pending) = pending {
453                    c.release_mouse();
454                    if !pending.dragged {
455                        let index = self.index_at_location(c, event.location);
456                        if index == Some(pending.index) {
457                            self.dispatch_activate(c, pending.index);
458                        }
459                    }
460                    return true;
461                }
462                false
463            }
464            _ => false,
465        }
466    }

475    /// Dispatch the activation command for a selected row.
476    fn dispatch_activate(&self, c: &mut dyn Context, index: usize) -> bool {
477        let Some(config) = self.on_activate.as_ref() else {
478            return false;
479        };
480        let frame = CommandScopeFrame {
481            event: c.current_event().cloned(),
482            mouse: c.current_mouse_event(),
483            list_row: Some(ListRowContext {
484                list: c.node_id(),
485                index,
486            }),
487        };
488        let invocation = config.invocation_with_index(index);
489        c.dispatch_command_scoped(frame, &invocation).is_ok()
490    }

557    /// Ensure the selected item is visible in the view.
558    fn ensure_selected_visible(&self, c: &mut dyn Context) {
559        let Some(selected_idx) = self.selected else {
560            return;
561        };
562
563        let view = c.view();
564        let view_rect = view.view_rect();
565
566        // Compute item positions by measuring each child
567        let metrics = self.item_metrics(c, view_rect.w.max(1));
568        let Some((start, height)) = metrics.get(selected_idx).copied() else {
569            return;
570        };
571
572        if start < view_rect.tl.y {
573            let delta = view_rect.tl.y - start;
574            let _ = c.scroll_by(0, -(delta as i32));
575        } else if start.saturating_add(height) > view_rect.tl.y.saturating_add(view_rect.h) {
576            let delta = start.saturating_add(height) - (view_rect.tl.y + view_rect.h);
577            let _ = c.scroll_by(0, delta as i32);
578        }
579    }

581    /// Move selection by one page and keep it visible.
582    fn page_shift(&mut self, c: &mut dyn Context, forward: bool) {
583        if self.items.is_empty() {
584            return;
585        }
586
587        let view = c.view();
588        let view_rect = view.view_rect();
589        if view_rect.h == 0 {
590            return;
591        }
592
593        let metrics = self.item_metrics(c, view_rect.w.max(1));
594        let selected_idx = self.selected.unwrap_or(0).min(self.items.len() - 1);
595        let Some((start, _height)) = metrics.get(selected_idx).copied() else {
596            return;
597        };
598
599        let page = view_rect.h.max(1);
600        let target_y = if forward {
601            start.saturating_add(page)
602        } else {
603            start.saturating_sub(page)
604        };
605
606        if let Some(target_idx) = Self::index_at_y(&metrics, target_y) {
607            self.select(c, target_idx);
608            self.focus_selected(c);
609            self.ensure_selected_visible(c);
610        }
611    }

613    /// Find the item index at a local content-space location.
614    fn index_at_location(&self, c: &dyn Context, location: Point) -> Option<usize> {
615        let view = c.view();
616        let view_rect = view.view_rect();
617        let content_y = view_rect.tl.y.saturating_add(location.y);
618        let metrics = self.item_metrics(c, view_rect.w.max(1));
619        Self::index_at_y(&metrics, content_y)
620    }
621
622    /// Build (start_y, height) tuples for each item.
623    fn item_metrics(&self, c: &dyn ViewContext, available_width: u32) -> Vec<(u32, u32)> {
624        let mut metrics = Vec::with_capacity(self.items.len());
625        let mut y_offset = 0u32;
626
627        for id in &self.items {
628            // Get the child's layout and compute its height
629            let height = c.node_view((*id).into()).map(|v| v.outer.h).unwrap_or(1);
630
631            metrics.push((y_offset, height));
632            y_offset = y_offset.saturating_add(height);
633        }
634
635        // If no layout data yet, estimate with 1-height items
636        if metrics.is_empty() && !self.items.is_empty() {
637            for (i, _) in self.items.iter().enumerate() {
638                metrics.push((i as u32, 1));
639            }
640        }
641
642        let _ = available_width; // Future: could use for responsive layouts
643        metrics
644    }
645
646    /// Find the item index covering a y coordinate.
647    fn index_at_y(metrics: &[(u32, u32)], y: u32) -> Option<usize> {
648        for (idx, (start, height)) in metrics.iter().enumerate() {
649            if y < start.saturating_add(*height) {
650                return Some(idx);
651            }
652        }
653        if metrics.is_empty() {
654            None
655        } else {
656            Some(metrics.len() - 1)
657        }
658    }

661impl<W: Selectable + Send + 'static> Widget for List<W> {
662    fn layout(&self) -> Layout {
663        let mut layout = Layout::column()
664            .flex_vertical(1)
665            .flex_horizontal(1)
666            .overflow_x();
667        if let Some(indicator) = &self.selection_indicator
668            && indicator.width > 0
669        {
670            layout = layout.padding(Edges::new(0, 0, 0, indicator.width));
671        }
672        layout
673    }

675    fn on_event(&mut self, event: &Event, ctx: &mut dyn Context) -> EventOutcome {
676        if let Event::Mouse(mouse_event) = event
677            && self.handle_click(ctx, *mouse_event)
678        {
679            return EventOutcome::Handle;
680        }
681        EventOutcome::Ignore
682    }

684    fn render(&mut self, rndr: &mut Render, ctx: &dyn ViewContext) -> Result<()> {
685        let view = ctx.view();
686        let area = view.outer_rect_local();
687
688        // Fill background using list style.
689        rndr.fill("list", area, ' ')?;
690
691        if let Some(indicator) = &self.selection_indicator
692            && let Some(selected_idx) = self.selected
693            && indicator.width > 0
694        {
695            let metrics = self.item_metrics(ctx, view.view_rect().w.max(1));
696            if let Some((start, height)) = metrics.get(selected_idx).copied() {
697                let view_rect = view.view_rect();
698                let visible_start = start.max(view_rect.tl.y);
699                let visible_end = start
700                    .saturating_add(height)
701                    .min(view_rect.tl.y.saturating_add(view_rect.h));
702
703                if visible_start < visible_end {
704                    let content_origin = view.content_origin();
705                    let local_y = content_origin
706                        .y
707                        .saturating_add(visible_start - view_rect.tl.y);
708                    let width = indicator.width.min(area.w);
709
710                    if width > 0 {
711                        if indicator.repeat {
712                            for offset in 0..(visible_end - visible_start) {
713                                let line = Line::new(0, local_y.saturating_add(offset), width);
714                                rndr.text(&indicator.style, line, &indicator.text)?;
715                            }
716                        } else {
717                            let line = Line::new(0, local_y, width);
718                            rndr.text(&indicator.style, line, &indicator.text)?;
719                        }
720                    }
721                }
722            }
723        }
724
725        Ok(())
726    }

728    fn measure(&self, c: MeasureConstraints) -> Measurement {
729        // For now, defer to intrinsic content sizing
730        // The actual layout will be handled by the column layout
731        let available_width = match c.width {
732            Constraint::Exact(n) | Constraint::AtMost(n) => n.max(1),
733            Constraint::Unbounded => 100,
734        };
735
736        // Estimate based on item count (items will self-measure)
737        let height = self.items.len() as u32;
738        c.clamp(Size::new(available_width, height.max(1)))
739    }

741    fn canvas(&self, view: Size<u32>, ctx: &CanvasContext<'_>) -> Size<u32> {
742        // Sum child canvas heights and find max canvas width for scrolling
743        let mut total_height = 0u32;
744        let mut max_width = view.width;
745
746        for child in ctx.children() {
747            // Use canvas dimensions for proper scroll support
748            total_height = total_height.saturating_add(child.canvas.height);
749            max_width = max_width.max(child.canvas.width);
750        }
751
752        Size::new(max_width, total_height.max(1))
753    }

765/// Compute the indicator width in cells from a multi-line string.
766fn indicator_width(text: &str) -> u32 {
767    text.lines()
768        .map(UnicodeWidthStr::width)
769        .max()
770        .unwrap_or(0)
771        .try_into()
772        .unwrap_or(0)
773}

775/// Return true when drag distance exceeds the configured threshold.
776fn drag_exceeded(origin: Point, current: Point, threshold: u32) -> bool {
777    let dx = origin.x.abs_diff(current.x);
778    let dy = origin.y.abs_diff(current.y);
779    dx.max(dy) > threshold
780}
