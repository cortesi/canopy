6/// Information about a focusable node
7#[derive(Debug, Clone)]
8pub struct FocusableNode {
9    pub id: NodeId,
10    pub rect: Rect,
11}

13/// Collect all focusable nodes with their screen rectangles
14pub fn collect_focusable_nodes(root: &mut dyn Node) -> Result<Vec<FocusableNode>> {
15    let mut nodes = Vec::new();
16
17    // Create initial ViewStack - same as render_traversal
18    let root_vp = root.vp();
19    let screen_vp = ViewPort::new(root_vp.canvas(), root_vp.canvas().rect(), (0, 0))?;
20    let mut view_stack = ViewStack::new(screen_vp);
21
22    collect_focusable_recursive(root, &mut view_stack, &mut nodes, Point::zero())?;
23
24    Ok(nodes)
25}
26
27fn collect_focusable_recursive(
28    node: &mut dyn Node,
29    view_stack: &mut ViewStack,
30    nodes: &mut Vec<FocusableNode>,
31    _parent_screen_pos: Point,
32) -> Result<()> {
33    if node.is_hidden() {
34        return Ok(());
35    }
36
37    let node_vp = node.vp();
38    if node_vp.view().is_zero() {
39        return Ok(());
40    }
41
42    // Push the node's viewport onto the stack
43    // The node's position is already parent-relative
44    view_stack.push(node_vp);
45
46    // Get screen rect from projection
47    if let Some((_, screen_rect)) = view_stack.projection() {
48        if node.accept_focus() {
49            nodes.push(FocusableNode {
50                id: node.id(),
51                rect: screen_rect,
52            });
53        }
54
55        // Process children
56        node.children(&mut |child| {
57            collect_focusable_recursive(child, view_stack, nodes, _parent_screen_pos)?;
58            Ok(())
59        })?;
60    }
61
62    // Pop viewport
63    view_stack.pop()?;
64
65    Ok(())
66}

68/// Find the best focus target in the specified direction
69pub fn find_focus_target(
70    current_rect: Rect,
71    direction: Direction,
72    candidates: &[FocusableNode],
73    current_id: &NodeId,
74) -> Option<NodeId> {
75    // Find the center point of the current rectangle
76    let current_center = Point {
77        x: current_rect.tl.x + current_rect.w / 2,
78        y: current_rect.tl.y + current_rect.h / 2,
79    };
80
81    // Filter candidates based on direction
82    // A candidate is valid if:
83    // 1. Its center is in the correct direction from current's center
84    // 2. It has overlap in the perpendicular axis (to prevent diagonal movement)
85    let mut valid_candidates: Vec<&FocusableNode> = candidates
86        .iter()
87        .filter(|n| &n.id != current_id)
88        .filter(|n| {
89            let candidate_center = Point {
90                x: n.rect.tl.x + n.rect.w / 2,
91                y: n.rect.tl.y + n.rect.h / 2,
92            };
93
94            match direction {
95                Direction::Right | Direction::Left => {
96                    // For horizontal movement, check:
97                    // 1. Center is in the right direction
98                    let center_ok = match direction {
99                        Direction::Right => candidate_center.x > current_center.x,
100                        Direction::Left => candidate_center.x < current_center.x,
101                        _ => unreachable!(),
102                    };
103
104                    // 2. There's vertical overlap between the rectangles
105                    let vertical_overlap = n.rect.tl.y < current_rect.tl.y + current_rect.h
106                        && n.rect.tl.y + n.rect.h > current_rect.tl.y;
107
108                    center_ok && vertical_overlap
109                }
110                Direction::Down | Direction::Up => {
111                    // For vertical movement, check:
112                    // 1. Center is in the right direction
113                    let center_ok = match direction {
114                        Direction::Down => candidate_center.y > current_center.y,
115                        Direction::Up => candidate_center.y < current_center.y,
116                        _ => unreachable!(),
117                    };
118
119                    // 2. There's horizontal overlap between the rectangles
120                    let horizontal_overlap = n.rect.tl.x < current_rect.tl.x + current_rect.w
121                        && n.rect.tl.x + n.rect.w > current_rect.tl.x;
122
123                    center_ok && horizontal_overlap
124                }
125            }
126        })
127        .collect();
128
129    // Remove current node if it somehow got through
130    valid_candidates.retain(|n| &n.id != current_id);
131
132    if valid_candidates.is_empty() {
133        return None;
134    }
135
136    // Sort candidates by a score that considers:
137    // 1. Edge distance in the primary direction
138    // 2. Overlap amount in the secondary direction
139    // 3. Center-to-center distance as a tiebreaker
140    valid_candidates.sort_by_key(|n| {
141        match direction {
142            Direction::Right => {
143                // Primary: distance from current's right edge to candidate's left edge
144                let edge_dist = n
145                    .rect
146                    .tl
147                    .x
148                    .saturating_sub(current_rect.tl.x + current_rect.w);
149
150                // Secondary: vertical alignment (less distance = better alignment)
151                let vert_center_dist = current_center.y.abs_diff(n.rect.tl.y + n.rect.h / 2);
152
153                // Score: prioritize edge distance, then vertical alignment
154                (edge_dist as u64) * 10000 + (vert_center_dist as u64)
155            }
156            Direction::Left => {
157                // Primary: distance from candidate's right edge to current's left edge
158                let edge_dist = current_rect.tl.x.saturating_sub(n.rect.tl.x + n.rect.w);
159
160                // Secondary: vertical alignment
161                let vert_center_dist = current_center.y.abs_diff(n.rect.tl.y + n.rect.h / 2);
162
163                (edge_dist as u64) * 10000 + (vert_center_dist as u64)
164            }
165            Direction::Down => {
166                // Primary: distance from current's bottom edge to candidate's top edge
167                let edge_dist = n
168                    .rect
169                    .tl
170                    .y
171                    .saturating_sub(current_rect.tl.y + current_rect.h);
172
173                // Secondary: horizontal alignment
174                let horiz_center_dist = current_center.x.abs_diff(n.rect.tl.x + n.rect.w / 2);
175
176                (edge_dist as u64) * 10000 + (horiz_center_dist as u64)
177            }
178            Direction::Up => {
179                // Primary: distance from candidate's bottom edge to current's top edge
180                let edge_dist = current_rect.tl.y.saturating_sub(n.rect.tl.y + n.rect.h);
181
182                // Secondary: horizontal alignment
183                let horiz_center_dist = current_center.x.abs_diff(n.rect.tl.x + n.rect.w / 2);
184
185                (edge_dist as u64) * 10000 + (horiz_center_dist as u64)
186            }
187        }
188    });
189
190    valid_candidates.first().map(|n| n.id.clone())
191}

193/// Find the currently focused node and its screen rectangle
194pub fn find_focused_node(
195    ctx: &dyn Context,
196    root: &mut dyn Node,
197    focusable_nodes: &[FocusableNode],
198) -> Option<(NodeId, Rect)> {
199    // Use preorder traversal to find the focused node
200    let mut focused_id = None;
201    preorder(root, &mut |node| -> Result<Walk<()>> {
202        if ctx.is_focused(node) {
203            focused_id = Some(node.id());
204            return Ok(Walk::Handle(()));
205        }
206        Ok(Walk::Continue)
207    })
208    .ok()?;
209
210    // Find the corresponding focusable node info
211    if let Some(id) = focused_id {
212        for node in focusable_nodes {
213            if node.id == id {
214                return Some((node.id.clone(), node.rect));
215            }
216        }
217    }
218
219    None
220}
