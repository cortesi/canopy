13/// Collect all focusable nodes with their screen rectangles
14pub fn collect_focusable_nodes(root: &mut dyn Node) -> Result<Vec<FocusableNode>> {
15    let mut nodes = Vec::new();
16
17    // Create initial ViewStack - same as render_traversal
18    let root_vp = root.vp();
19    let screen_vp = ViewPort::new(root_vp.canvas(), root_vp.canvas().rect(), (0, 0))?;
20    let mut view_stack = ViewStack::new(screen_vp);
21
22    collect_focusable_recursive(root, &mut view_stack, &mut nodes, Point::zero())?;
23
24    Ok(nodes)
25}

27fn collect_focusable_recursive(
28    node: &mut dyn Node,
29    view_stack: &mut ViewStack,
30    nodes: &mut Vec<FocusableNode>,
31    parent_screen_pos: Point,
32) -> Result<()> {
33    if node.is_hidden() {
34        return Ok(());
35    }
36
37    let node_vp = node.vp();
38    if node_vp.view().is_zero() {
39        return Ok(());
40    }
41
42    // Convert node position to parent-relative coordinates
43    // If the position is already parent-relative (from place_()), it will be small
44    // If the position is absolute (from place()), we need to subtract parent position
45    let relative_pos = if node_vp.position().x < parent_screen_pos.x
46        || node_vp.position().y < parent_screen_pos.y
47    {
48        // Position is smaller than parent position, so it's already parent-relative
49        node_vp.position()
50    } else {
51        // Position might be absolute, convert to parent-relative
52        Point {
53            x: node_vp.position().x.saturating_sub(parent_screen_pos.x),
54            y: node_vp.position().y.saturating_sub(parent_screen_pos.y),
55        }
56    };
57
58    // Create a new viewport with parent-relative position
59    let relative_vp = ViewPort::new(node_vp.canvas(), node_vp.view(), relative_pos)?;
60
61    // Push the viewport onto the stack
62    view_stack.push(relative_vp);
63
64    // Get screen rect from projection
65    if let Some((_, screen_rect)) = view_stack.projection() {
66        if node.accept_focus() {
67            nodes.push(FocusableNode {
68                id: node.id(),
69                rect: screen_rect,
70            });
71        }
72
73        // Process children
74        // The screen_rect gives us the absolute screen position of this node
75        // We should use this as the parent position for children
76        let node_screen_pos = screen_rect.tl;
77        node.children(&mut |child| {
78            collect_focusable_recursive(child, view_stack, nodes, node_screen_pos)?;
79            Ok(())
80        })?;
81    }
82
83    // Pop viewport
84    view_stack.pop()?;
85
86    Ok(())
87}

89/// Find the best focus target in the specified direction
90pub fn find_focus_target(
91    current_rect: Rect,
92    direction: Direction,
93    candidates: &[FocusableNode],
94    current_id: &NodeId,
95) -> Option<NodeId> {
96    // Find the center point of the current rectangle
97    let current_center = Point {
98        x: current_rect.tl.x + current_rect.w / 2,
99        y: current_rect.tl.y + current_rect.h / 2,
100    };
101
102    // Filter candidates based on direction
103    // A candidate is valid if:
104    // 1. Its center is in the correct direction from current's center
105    // 2. It has overlap in the perpendicular axis (to prevent diagonal movement)
106    let mut valid_candidates: Vec<&FocusableNode> = candidates
107        .iter()
108        .filter(|n| &n.id != current_id)
109        .filter(|n| {
110            let candidate_center = Point {
111                x: n.rect.tl.x + n.rect.w / 2,
112                y: n.rect.tl.y + n.rect.h / 2,
113            };
114
115            match direction {
116                Direction::Right | Direction::Left => {
117                    // For horizontal movement, check:
118                    // 1. Center is in the right direction
119                    let center_ok = match direction {
120                        Direction::Right => candidate_center.x > current_center.x,
121                        Direction::Left => candidate_center.x < current_center.x,
122                        _ => unreachable!(),
123                    };
124
125                    // 2. There's vertical overlap between the rectangles
126                    let vertical_overlap = n.rect.tl.y < current_rect.tl.y + current_rect.h
127                        && n.rect.tl.y + n.rect.h > current_rect.tl.y;
128
129                    center_ok && vertical_overlap
130                }
131                Direction::Down | Direction::Up => {
132                    // For vertical movement, check:
133                    // 1. Center is in the right direction
134                    let center_ok = match direction {
135                        Direction::Down => candidate_center.y > current_center.y,
136                        Direction::Up => candidate_center.y < current_center.y,
137                        _ => unreachable!(),
138                    };
139
140                    // 2. There's horizontal overlap between the rectangles
141                    let horizontal_overlap = n.rect.tl.x < current_rect.tl.x + current_rect.w
142                        && n.rect.tl.x + n.rect.w > current_rect.tl.x;
143
144                    center_ok && horizontal_overlap
145                }
146            }
147        })
148        .collect();
149
150    // Remove current node if it somehow got through
151    valid_candidates.retain(|n| &n.id != current_id);
152
153    if valid_candidates.is_empty() {
154        return None;
155    }
156
157    // Sort candidates by a score that considers:
158    // 1. Edge distance in the primary direction
159    // 2. Overlap amount in the secondary direction
160    // 3. Center-to-center distance as a tiebreaker
161    valid_candidates.sort_by_key(|n| {
162        match direction {
163            Direction::Right => {
164                // Primary: distance from current's right edge to candidate's left edge
165                let edge_dist = n
166                    .rect
167                    .tl
168                    .x
169                    .saturating_sub(current_rect.tl.x + current_rect.w);
170
171                // Secondary: vertical alignment (less distance = better alignment)
172                let vert_center_dist = current_center.y.abs_diff(n.rect.tl.y + n.rect.h / 2);
173
174                // Score: prioritize edge distance, then vertical alignment
175                (edge_dist as u64) * 10000 + (vert_center_dist as u64)
176            }
177            Direction::Left => {
178                // Primary: distance from candidate's right edge to current's left edge
179                let edge_dist = current_rect.tl.x.saturating_sub(n.rect.tl.x + n.rect.w);
180
181                // Secondary: vertical alignment
182                let vert_center_dist = current_center.y.abs_diff(n.rect.tl.y + n.rect.h / 2);
183
184                (edge_dist as u64) * 10000 + (vert_center_dist as u64)
185            }
186            Direction::Down => {
187                // Primary: distance from current's bottom edge to candidate's top edge
188                let edge_dist = n
189                    .rect
190                    .tl
191                    .y
192                    .saturating_sub(current_rect.tl.y + current_rect.h);
193
194                // Secondary: horizontal alignment
195                let horiz_center_dist = current_center.x.abs_diff(n.rect.tl.x + n.rect.w / 2);
196
197                (edge_dist as u64) * 10000 + (horiz_center_dist as u64)
198            }
199            Direction::Up => {
200                // Primary: distance from candidate's bottom edge to current's top edge
201                let edge_dist = current_rect.tl.y.saturating_sub(n.rect.tl.y + n.rect.h);
202
203                // Secondary: horizontal alignment
204                let horiz_center_dist = current_center.x.abs_diff(n.rect.tl.x + n.rect.w / 2);
205
206                (edge_dist as u64) * 10000 + (horiz_center_dist as u64)
207            }
208        }
209    });
210
211    valid_candidates.first().map(|n| n.id.clone())
212}

214/// Find the currently focused node and its screen rectangle
215pub fn find_focused_node(
216    ctx: &dyn Context,
217    root: &mut dyn Node,
218    focusable_nodes: &[FocusableNode],
219) -> Option<(NodeId, Rect)> {
220    // Use preorder traversal to find the focused node
221    let mut focused_id = None;
222    preorder(root, &mut |node| -> Result<Walk<()>> {
223        if ctx.is_focused(node) {
224            focused_id = Some(node.id());
225            return Ok(Walk::Handle(()));
226        }
227        Ok(Walk::Continue)
228    })
229    .ok()?;
230
231    // Find the corresponding focusable node info
232    if let Some(id) = focused_id {
233        for node in focusable_nodes {
234            if node.id == id {
235                return Some((node.id.clone(), node.rect));
236            }
237        }
238    }
239
240    None
241}
