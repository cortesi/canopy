5/// A projection from the virtual space of a node to the screen.
6#[derive(Default, Debug, Clone, Copy, Hash, PartialEq, Eq)]
7struct Projection {
8    /// A region in the node's canvas. This is always a sub-rectangle of the node's view.
9    region: Rect,
10    /// The absolute point on the screen to which the region is projected.
11    screen: Point,
12}
13
14impl Projection {
15    fn new<R, S>(region: R, screen: S) -> Projection
16    where
17        R: Into<Rect>,
18        S: Into<Point>,
19    {
20        Projection {
21            region: region.into(),
22            screen: screen.into(),
23        }
24    }
25}

27/// A ViewPort manages the size of a node and its projection onto the screen.
28#[derive(Default, Debug, Clone, Copy, Hash, PartialEq, Eq)]
29pub struct ViewPort {
30    /// The location of the node in the parent's canvas. Must only be changed by the parent node.
31    pub position: Point,
32    /// The portion of this node that is displayed - a sub-rectangle of the canvas. Must only be changed by the node
33    /// itself.
34    pub view: Rect,
35    /// The canvas on which children are positioned, and to which rendering occurs. Must only be changed by the node
36    /// itself.
37    pub canvas: Expanse,
38}
39
40impl ViewPort {
41    /// Create a new View with the given outer and inner rectangles. The view
42    /// rectangle must be fully contained within the outer rectangle.
43    pub fn new(
44        canvas: impl Into<Expanse>,
45        view: impl Into<Rect>,
46        position: impl Into<Point>,
47    ) -> Result<ViewPort> {
48        let view = view.into();
49        let size = canvas.into();
50        if !size.rect().contains_rect(&view) {
51            Err(error::Error::Geometry(format!(
52                "view {:?} not contained in size {:?}",
53                view, size,
54            )))
55        } else {
56            Ok(ViewPort {
57                canvas: size,
58                view,
59                position: position.into(),
60            })
61        }
62    }
63
64    /// Scroll the view to the specified position. The view is clamped within
65    /// the outer rectangle.
66    pub(crate) fn scroll_to(&mut self, x: u16, y: u16) {
67        let r = Rect::new(x, y, self.view.w, self.view.h);
68        // We unwrap here, because this can only be an error if view is larger
69        // than outer, which we ensure is not the case.
70        self.view = r.clamp_within(self.canvas.rect()).unwrap();
71    }
72
73    /// Scroll the view by the given offsets. The view rectangle is clamped
74    /// within the outer rectangle.
75    pub(crate) fn scroll_by(&mut self, x: i16, y: i16) {
76        self.view = self.view.shift_within(x, y, self.canvas.rect());
77    }
78
79    /// Scroll the view up by the height of the view rectangle.
80    pub(crate) fn page_up(&mut self) {
81        self.scroll_by(0, -(self.view.h as i16))
82    }
83
84    /// Scroll the view down by the height of the view rectangle.
85    pub(crate) fn page_down(&mut self) {
86        self.scroll_by(0, self.view.h as i16)
87    }
88
89    /// Scroll the view up by one line.
90    pub(crate) fn scroll_up(&mut self) {
91        self.scroll_by(0, -1)
92    }
93
94    /// Scroll the view down by one line.
95    pub(crate) fn scroll_down(&mut self) {
96        self.scroll_by(0, 1)
97    }
98
99    /// Scroll the view left by one line.
100    pub(crate) fn scroll_left(&mut self) {
101        self.scroll_by(-1, 0)
102    }
103
104    /// Scroll the view right by one line.
105    pub fn scroll_right(&mut self) {
106        self.scroll_by(1, 0)
107    }
108
109    /// Absolute rectangle for the screen region the node is being projected
110    /// onto.
111    pub fn screen_rect(&self) -> Rect {
112        self.view.at(self.position)
113    }
114
115    /// Set the screen, view and outer rects all to the same size. This is
116    /// useful for nodes that fill whatever space they're given.
117    pub fn set_fill(&self, screen: Rect) -> Self {
118        let mut vp = *self;
119        vp.view = screen;
120        vp.canvas = screen.into();
121        vp
122    }
123
124    /// Set the node size and the target view size at the same time. We try to retain the old view position, but shift
125    /// and resize it to be within the view if necessary.
126    pub fn fit_size(&mut self, size: Expanse, view_size: Expanse) {
127        let w = size.w.min(view_size.w);
128        let h = size.h.min(view_size.h);
129        self.canvas = size;
130        // Now we just clamp the rect into the view.
131        self.view = Rect {
132            tl: self.view.tl,
133            w,
134            h,
135        }
136        .clamp_within(self.canvas.rect())
137        // Safe to unwrap because of w, h computation above.
138        .unwrap();
139    }
140
141    /// Calculates the (pre, active, post) rectangles needed to draw a vertical
142    /// scroll bar for this viewport in the specified margin rect (usually a
143    /// right or left vertical margin). Returns None if no scroll bar is needed.
144    pub fn vactive(&self, margin: Rect) -> Result<Option<(Rect, Rect, Rect)>> {
145        if self.view.h == self.canvas.h {
146            Ok(None)
147        } else {
148            let (pre, active, post) = margin
149                .vextent()
150                .split_active(self.view.vextent(), self.canvas.rect().vextent())?;
151            Ok(Some((
152                margin.vslice(&pre)?,
153                margin.vslice(&active)?,
154                margin.vslice(&post)?,
155            )))
156        }
157    }
158
159    /// Calculates the (pre, active, post) rectangles needed to draw a
160    /// horizontal scroll bar for this viewport in the specified margin rect
161    /// (usually a bottom horizontal margin). Returns None if no scroll bar is
162    /// needed.
163    pub fn hactive(&self, margin: Rect) -> Result<Option<(Rect, Rect, Rect)>> {
164        if self.view.w == self.canvas.w {
165            Ok(None)
166        } else {
167            let (pre, active, post) = margin
168                .hextent()
169                .split_active(self.view.hextent(), self.canvas.rect().hextent())?;
170            Ok(Some((
171                margin.hslice(&pre)?,
172                margin.hslice(&active)?,
173                margin.hslice(&post)?,
174            )))
175        }
176    }
177
178    /// Project a point in virtual space to the screen. If the point is not
179    /// on-screen, return None.
180    pub fn project_point(&self, p: impl Into<Point>) -> Option<Point> {
181        let p = p.into();
182        if self.view.contains_point(p) {
183            let rp = self.view.rebase_point(p).unwrap();
184            // We know view is not larger than screen, so we can unwrap.
185            Some(Point {
186                x: self.position.x + rp.x,
187                y: self.position.y + rp.y,
188            })
189        } else {
190            None
191        }
192    }
193
194    /// Take a rectangle on the physical screen, and calculate the matching portion of the view rectangle.
195    pub fn unproject(&self, r: Rect) -> Result<Rect> {
196        self.screen_rect().rebase_rect(&r)
197    }
198
199    /// Project a rect in virtual space to the screen. If the virtual rect and
200    /// the screen rect partially overlap, just the overlap is returned.
201    pub fn project_rect(&self, r: Rect) -> Option<Rect> {
202        if let Some(o) = self.view.intersect(&r) {
203            let r = self.view.rebase_rect(&o).unwrap();
204            Some(Rect {
205                tl: self.position.scroll(r.tl.x as i16, r.tl.y as i16),
206                w: r.w,
207                h: r.h,
208            })
209        } else {
210            None
211        }
212    }
213
214    /// Project a line in virtual space to the screen. Returns an offset from
215    /// the start of the input line, plus a Line that is the projected region.
216    pub fn project_line(&self, l: Line) -> Option<(u16, Line)> {
217        if let Some(o) = self.view.intersect(&l.rect()) {
218            let rebase = self.view.rebase_rect(&o).unwrap();
219            Some((
220                o.tl.x - l.tl.x,
221                Line {
222                    tl: self.position.scroll(rebase.tl.x as i16, rebase.tl.y as i16),
223                    w: rebase.w,
224                },
225            ))
226        } else {
227            None
228        }
229    }
230
231    /// Given a rectangle within our outer, calculate the intersection with our
232    /// current view, and generate a ViewPort that would correctly display the
233    /// child on our screen.
234    pub fn map<T: Into<Rect>>(&self, child: T) -> Result<Option<ViewPort>> {
235        let child = child.into();
236        if let Some(i) = self.view.intersect(&child) {
237            let view_relative = self.view.rebase_rect(&i).unwrap();
238            Ok(Some(ViewPort {
239                canvas: child.expanse(),
240                // The view is the intersection relative to the child's outer
241                view: Rect::new(i.tl.x - child.tl.x, i.tl.y - child.tl.y, i.w, i.h),
242                position: Point {
243                    x: self.position.x + view_relative.tl.x,
244                    y: self.position.y + view_relative.tl.y,
245                },
246            }))
247        } else {
248            Ok(None)
249        }
250    }
251
252    /// Calculate this node's projection, given a parent projection. If there is no screen overlap, return None.
253    fn projection(&self, parent_projection: Projection) -> Option<Projection> {
254        let view_in_parent = self.view.at(self.position);
255        if let Some(overlap) = parent_projection.region.intersect(&view_in_parent) {
256            let region = view_in_parent.rebase_rect(&overlap).ok()?;
257            // Now, to calculate the screen offset, we take the relative position in the parent's projection, then add
258            // that to the screen offset.
259            let p = parent_projection.region.rebase_point(region.tl).ok()?;
260            Some(Projection::new(
261                region,
262                (
263                    p.x + parent_projection.screen.x,
264                    p.y + parent_projection.screen.y,
265                ),
266            ))
267        } else {
268            None
269        }
270    }

277    fn view_projection() -> Result<()> {
278        let v = ViewPort::new((30, 30), (0, 0, 30, 30), (0, 0))?;
279        // Bottom-right subrect
280        assert_eq!(
281            v.projection(Projection::new((20, 20, 10, 10), (0, 0))),
282            Some(Projection::new((20, 20, 10, 10), (0, 0))),
283        );
284        assert_eq!(
285            v.projection(Projection::new((20, 20, 10, 10), (10, 10))),
286            Some(Projection::new((20, 20, 10, 10), (10, 10))),
287        );
288        // Top-left subrect
289        assert_eq!(
290            v.projection(Projection::new((0, 0, 10, 10), (0, 0))),
291            Some(Projection::new((0, 0, 10, 10), (0, 0))),
292        );
293        assert_eq!(
294            v.projection(Projection::new((0, 0, 10, 10), (10, 10))),
295            Some(Projection::new((0, 0, 10, 10), (10, 10))),
296        );
297        // Perfectly wrapping the view
298        assert_eq!(
299            v.projection(Projection::new((0, 0, 30, 30), (0, 0))),
300            Some(Projection::new((0, 0, 30, 30), (0, 0))),
301        );
302        assert_eq!(
303            v.projection(Projection::new((0, 0, 30, 30), (10, 10))),
304            Some(Projection::new((0, 0, 30, 30), (10, 10))),
305        );
306
307        // No overlap
308        let v = ViewPort::new((30, 30), (10, 10, 10, 10), (0, 0))?;
309        assert_eq!(
310            v.projection(Projection::new((10, 10, 10, 10), (0, 0))),
311            None
312        );
313
314        // No overlap
315        let v = ViewPort::new((30, 30), (10, 10, 10, 10), (5, 5))?;
316        assert_eq!(
317            v.projection(Projection::new((10, 10, 10, 10), (0, 0))),
318            Some(Projection::new((0, 0, 30, 30), (10, 10))),
319        );
320
321        Ok(())
322    }
