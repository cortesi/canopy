8/// Wrap a single child node - fit the child into the size and project it to the point (0, 0) in the parent view. We
9/// then mirror the child's size and view into the parent.
10#[macro_export]
11macro_rules! fit_wrap {
12    ($self: ident, $child:expr, $sz:expr) => {
13        $child.layout($sz)?;
14        $child.vp_mut().position = (0, 0).into();
15
16        let cvp = $child.vp();
17        // Mirror the child's size and view
18        $self.vp_mut().canvas = cvp.canvas;
19        $self.vp_mut().view = cvp.view;
20    };
21}
22
23/// Frame a single child node. First, we calculate the inner size after subtracting the frame. We then fit the child
24/// into this inner size, and project it appropriately in the parent view.
25#[macro_export]
26macro_rules! fit_frame {
27    ($self: ident, $child:expr, $sz:expr, $border:expr) => {{
28        $child.layout(Expanse {
29            w: $sz.w - ($border * 2),
30            h: $sz.h - ($border * 2),
31        })?;
32        $child.vp_mut().position = crate::geom::Point {
33            x: $border,
34            y: $border,
35        };
36        $self.vp_mut().canvas = $sz;
37        $self.vp_mut().view = $sz.rect();
38        crate::geom::Frame::new($sz.rect(), $border)
39    }};
40}
41
42/// Place a child in a given sub-rectangle of our view.
43#[macro_export]
44macro_rules! fit_place {
45    ($self: ident, $child:expr, $loc:expr) => {
46        $child.layout($loc.expanse())?;
47        $child.vp_mut().position = $loc.tl;
48    };
49}

51/// Adjust a node so that it fits a viewport. This fits the node to the viewport's screen rectangle, then adjusts the
52/// node's view to place as much of it within the viewport's screen rectangle as possible.
53pub fn fit(n: &mut dyn Node, parent_vp: ViewPort) -> Result<()> {
54    n.layout(parent_vp.screen_rect().into())?;
55    n.vp_mut().position = parent_vp.position;
56    Ok(())
57}
58
59/// Adjust a node so that viewport's screen rectangle frames it with a given
60/// margin. Fits the child to the viewport screen rect minus the border margin,
61/// then adjusts the node's view to place as much of of it on screen as
62/// possible. This function returns a `Frame` object that can be used to draw a
63/// border around the node.
64pub fn frame(n: &mut dyn Node, parent_vp: ViewPort, border: u16) -> Result<Frame> {
65    let inner = parent_vp.screen_rect().inner(border);
66    n.layout(inner.into())?;
67    n.vp_mut().position = inner.tl;
68
69    // Return a frame for drawing the screen boundary, but in the view
70    // rect's co-ordinate system.
71    Ok(Frame::new(
72        parent_vp.screen_rect().at(parent_vp.view.tl),
73        border,
74    ))
75}
76
77/// Place a node in a given screen rectangle. This fits the node to the
78/// region, and updates its viewport.
79pub fn place(n: &mut dyn Node, screen: Rect) -> Result<()> {
80    n.layout(screen.expanse())?;
81    n.vp_mut().position = screen.tl;
82    Ok(())
83}

94    #[test]
95    fn node_fit() -> Result<()> {
96        // If the child is the same size as the parent, then wrap just produces
97        // the same viewport
98        let mut n = TFixed::new(10, 10);
99        let vp = ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 10, 10), (10, 10))?;
100        fit(&mut n, vp)?;
101        assert_eq!(n.state().viewport, vp);
102
103        // If the child is smaller than parent, then wrap places the viewport at (0, 0)
104        let mut n = TFixed::new(5, 5);
105        let vp = ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 10, 10), (10, 10))?;
106        let expected = ViewPort::new(Expanse::new(5, 5), Rect::new(0, 0, 5, 5), (10, 10))?;
107        fit(&mut n, vp)?;
108        assert_eq!(n.state().viewport, expected,);
109        n.vp_mut().scroll_right();
110        n.vp_mut().scroll_down();
111        assert_eq!(n.state().viewport, expected,);
112
113        // If the child is larger than parent, then wrap places the viewport at (0, 0).
114        let mut n = TFixed::new(20, 20);
115        let vp = ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 10, 10), (10, 10))?;
116        fit(&mut n, vp)?;
117        assert_eq!(
118            n.state().viewport,
119            ViewPort::new(Expanse::new(20, 20), Rect::new(0, 0, 10, 10), (10, 10))?
120        );
121
122        // The child can shift its view freely
123        n.vp_mut().scroll_right();
124        n.vp_mut().scroll_down();
125        assert_eq!(
126            n.state().viewport,
127            ViewPort::new(Expanse::new(20, 20), Rect::new(1, 1, 10, 10), (10, 10))?
128        );
129
130        // And subsequent wraps maintain the child view position, if possible
131        fit(&mut n, vp)?;
132        assert_eq!(
133            n.state().viewport,
134            ViewPort::new(Expanse::new(20, 20), Rect::new(1, 1, 10, 10), (10, 10))?
135        );
136
137        // When the parent viewport shrinks, we maintain position and resize
138        let shrink = ViewPort::new(Expanse::new(3, 3), Rect::new(0, 0, 2, 2), (10, 10))?;
139        fit(&mut n, shrink)?;
140        assert_eq!(
141            n.state().viewport,
142            ViewPort::new(Expanse::new(20, 20), Rect::new(1, 1, 2, 2), (10, 10))?
143        );
144
145        Ok(())
146    }

148    #[test]
149    fn node_frame() -> Result<()> {
150        // If we have room, the adjustment just shifts the child node relative to the screen.
151        let mut n = TFixed::new(5, 5);
152        let vp = ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 10, 10), (10, 10))?;
153        frame(&mut n, vp, 1)?;
154        assert_eq!(
155            n.state().viewport,
156            ViewPort::new(Expanse::new(5, 5), Rect::new(0, 0, 5, 5), (11, 11))?
157        );
158
159        // If if the child node is too large, it is clipped to the bottom and left
160        let mut n = TFixed::new(10, 10);
161        let vp = ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 10, 10), (10, 10))?;
162        frame(&mut n, vp, 1)?;
163        assert_eq!(
164            n.state().viewport,
165            ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 8, 8), (11, 11))?
166        );
167
168        // If if the parent is smaller than the frame would require, we get a zero view
169        let mut n = TFixed::new(10, 10);
170        let vp = ViewPort::new(Expanse::new(0, 0), Rect::new(0, 0, 0, 0), (10, 10))?;
171        frame(&mut n, vp, 1)?;
172        assert_eq!(
173            n.state().viewport,
174            ViewPort::new(Expanse::new(10, 10), Rect::new(0, 0, 0, 0), (0, 0))?
175        );
176
177        Ok(())
178    }
