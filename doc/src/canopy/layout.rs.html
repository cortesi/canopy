1//! Helper functions for `Node::layout` implementations.
2
3use crate::{
4    geom::{Expanse, Frame, Rect},
5    Node, Result, ViewPort,
6};
7
8pub struct Layout {}
9
10impl Layout {
11    /// Wrap a single child node, mirroring the child's size and view.
12    pub fn wrap(&self, parent: &mut dyn Node, vp: ViewPort) -> Result<()> {
13        // Mirror the child's size and view
14        parent.__vp_mut().canvas = vp.canvas;
15        parent.__vp_mut().view = vp.view;
16        Ok(())
17    }
18
19    /// Frame a single child node. First, we calculate the inner size after subtracting the frame. We then fit the child
20    /// into this inner size, and project it appropriately in the parent view.
21    pub fn frame(&self, child: &mut dyn Node, sz: Expanse, border: u16) -> Result<Frame> {
22        child.layout(
23            self,
24            Expanse {
25                w: sz.w - (border * 2),
26                h: sz.h - (border * 2),
27            },
28        )?;
29        child.__vp_mut().position = crate::geom::Point {
30            x: border,
31            y: border,
32        };
33        Ok(crate::geom::Frame::new(sz.rect(), border))
34    }
35
36    /// Place a node in a given sub-rectangle of a parent's view.
37    pub fn fill(&self, n: &mut dyn Node, sz: Expanse) -> Result<()> {
38        let vp = n.__vp_mut();
39        vp.canvas = sz;
40        vp.view = sz.rect();
41        Ok(())
42    }
43
44    /// Place a child in a given sub-rectangle of a parent's view.
45    pub fn place(&self, child: &mut dyn Node, loc: Rect) -> Result<()> {
46        child.layout(self, loc.expanse())?;
47        child.__vp_mut().position = loc.tl;
48        Ok(())
49    }
50
51    pub fn size(&self, n: &mut dyn Node, sz: Expanse, view_size: Expanse) -> Result<()> {
52        n.__vp_mut().fit_size(sz, view_size);
53        Ok(())
54    }
55
56    /// Adjust a child node so that it fits a viewport. This lays the node out to the viewport's screen rectangle, then
57    /// adjusts the node's view to place as much of it within the viewport's screen rectangle as possible.
58    pub fn fit(&self, n: &mut dyn Node, parent_vp: ViewPort) -> Result<()> {
59        n.layout(self, parent_vp.screen_rect().into())?;
60        n.__vp_mut().position = parent_vp.position;
61        Ok(())
62    }
63}
