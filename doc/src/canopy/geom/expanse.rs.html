3/// An `Expanse` is a rectangle that has a width and height but no location.
4/// This is useful when we want to deal with `Rect`s abstractly, or when we want
5/// to madate that the location of a `Rect` is (0, 0).
6#[derive(Debug, Clone, Copy, Hash, PartialEq, Eq)]
7pub struct Expanse {
8    pub w: u16,
9    pub h: u16,
10}
11
12impl Default for Expanse {
13    /// Constructs a zero-valued size.
14    fn default() -> Expanse {
15        Expanse { w: 0, h: 0 }
16    }
17}
18
19impl Expanse {
20    pub fn new(w: u16, h: u16) -> Expanse {
21        Expanse { w, h }
22    }
23
24    /// The area of this expanse.
25    pub fn area(&self) -> u32 {
26        self.w as u32 * self.h as u32
27    }
28
29    /// Return a `Rect` with the same dimensions as the `Expanse`, but a location at (0, 0).
30    pub fn rect(&self) -> Rect {
31        Rect {
32            tl: Point::default(),
33            w: self.w,
34            h: self.h,
35        }
36    }
37    /// True if this Size can completely enclose the target size in both dimensions.
38    pub fn contains(&self, other: &Expanse) -> bool {
39        self.w >= other.w && self.h >= other.h
40    }
41}
42
43impl From<Rect> for Expanse {
44    fn from(r: Rect) -> Expanse {
45        Expanse { w: r.w, h: r.h }
46    }
47}
48
49impl From<(u16, u16)> for Expanse {
50    fn from(v: (u16, u16)) -> Expanse {
51        Expanse { w: v.0, h: v.1 }
52    }
53}
