1use crate::Canopy;
2use canopy_core::{Result, event::key::Key, event::mouse::Mouse};

4/// Binder provides an ergonomic way to specify a set of key bindings using a
5/// builder patttern.
6pub struct Binder<'a> {
7    mode: String,
8    path_filter: String,
9    cnpy: &'a mut Canopy,
10}
11
12impl<'a> Binder<'a> {
13    pub fn new(cnpy: &'a mut Canopy) -> Self {
14        Binder {
15            mode: "".into(),
16            path_filter: "".into(),
17            cnpy,
18        }
19    }
20
21    /// Add the default bindings for a widget.
22    pub fn defaults<T>(self) -> Self
23    where
24        T: DefaultBindings,
25    {
26        T::defaults(self)
27    }
28
29    /// Set the mode for subsequent bindings.
30    pub fn with_mode(mut self, m: &str) -> Self {
31        self.mode = m.to_string();
32        self
33    }
34
35    /// Set the path filter for subsequent bindings.
36    pub fn with_path(mut self, m: &str) -> Self {
37        self.path_filter = m.into();
38        self
39    }
40
41    /// Bind a key to a script fallibly.
42    pub fn try_key<K>(self, key: K, script: &str) -> Result<Self>
43    where
44        Key: From<K>,
45    {
46        self.cnpy
47            .bind_mode_key(key, &self.mode, &self.path_filter, script)?;
48        Ok(self)
49    }
50
51    /// Bind a key to a script, panicing if there is any error (usually in
52    /// compilation of the script). This is often acceptable for initial default
53    /// key bindings where scripts don't come from user input.
54    pub fn key<K>(self, key: K, script: &str) -> Self
55    where
56        Key: From<K>,
57    {
58        self.try_key(key, script).unwrap()
59    }
60
61    /// Bind a mouse action to a script fallibly.
62    pub fn try_mouse<K>(self, m: K, script: &str) -> Result<Self>
63    where
64        Mouse: From<K>,
65    {
66        self.cnpy
67            .bind_mode_mouse(m, &self.mode, &self.path_filter, script)?;
68        Ok(self)
69    }
70
71    /// Bind a mouse action to a script, panicing if there is any error (usually
72    /// in compilation of the script). This is often acceptable for initial
73    /// default key bindings where scripts don't come from user input.
74    pub fn mouse<K>(self, m: K, script: &str) -> Self
75    where
76        Mouse: From<K>,
77    {
78        self.try_mouse(m, script).unwrap()
79    }
80}
81
82pub trait DefaultBindings {
83    fn defaults(b: Binder) -> Binder;
84}
