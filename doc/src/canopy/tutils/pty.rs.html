1use expectrl::{spawn, Eof, Session};
2use std::time::Duration;
3
4use crate::{Error, Result};

6/// A handle to a process running under a pseudo terminal.
7pub struct PtyApp {
8    sess: Session,
9}
10
11impl PtyApp {
12    /// Spawn a command with the given arguments.
13    pub fn spawn_cmd(cmd: &str, args: &[&str]) -> Result<Self> {
14        let mut c = cmd.to_string();
15        for a in args {
16            c.push(' ');
17            c.push_str(a);
18        }
19        let sess = spawn(c).map_err(|e| Error::Internal(e.to_string()))?;
20        Ok(PtyApp { sess })
21    }
22
23    /// Expect the supplied pattern within `timeout`.
24    pub fn expect(&mut self, pat: &str, timeout: Duration) -> Result<()> {
25        self.sess.set_expect_timeout(Some(timeout));
26        self.sess
27            .expect(pat)
28            .map(|_| ())
29            .map_err(|e| Error::Internal(e.to_string()))
30    }
31
32    /// Send raw text to the running process.
33    pub fn send(&mut self, s: &str) -> Result<()> {
34        self.sess
35            .send(s)
36            .map_err(|e| Error::Internal(e.to_string()))
37    }
38
39    /// Send text followed by a newline.
40    pub fn send_line(&mut self, s: &str) -> Result<()> {
41        self.sess
42            .send_line(s)
43            .map_err(|e| Error::Internal(e.to_string()))
44    }
45
46    /// Wait for the process to exit.
47    pub fn wait_eof(&mut self, timeout: Duration) -> Result<()> {
48        self.sess.set_expect_timeout(Some(timeout));
49        self.sess
50            .expect(Eof)
51            .map(|_| ())
52            .map_err(|e| Error::Internal(e.to_string()))
53    }
54}
55
56/// Spawn a workspace binary from `target/debug` with the provided arguments.
57pub fn spawn_workspace_bin(name: &str, args: &[&str]) -> Result<PtyApp> {
58    let bin = format!(
59        concat!(env!("CARGO_MANIFEST_DIR"), "/../target/debug/{}"),
60        name
61    );
62    PtyApp::spawn_cmd(&bin, args)
63}
