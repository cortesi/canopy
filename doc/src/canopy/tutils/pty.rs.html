4/// Spawn a binary built by Cargo using `expectrl`.
5///
6/// `name` should be the crate name as listed in `Cargo.toml`.
7/// Any additional arguments are passed to the binary.
8/// The returned [`Session`] can be used to drive the process.
9pub fn spawn_bin(name: &str, args: &[&str]) -> std::result::Result<Session, expectrl::Error> {
10    let bin_path = std::env::var(format!("CARGO_BIN_EXE_{name}"))
11        .expect("missing CARGO_BIN_EXE path for binary");
12    let mut cmd = Command::new(bin_path);
13    cmd.args(args);
14    Session::spawn(cmd)
15}
