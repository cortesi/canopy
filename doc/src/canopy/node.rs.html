12/// Was an event handled or ignored?
13#[derive(Debug, PartialEq, Eq, Clone)]
14pub enum EventOutcome {
15    Handle,
16    Ignore,
17}
18
19/// The result of an event handler.
20#[derive(Debug, PartialEq, Eq, Clone)]
21pub struct EventResult {
22    /// Was the event handled or ignored?
23    pub outcome: EventOutcome,
24    /// If true, handling the event should not trigger a render sweep.
25    pub no_render: bool,
26}
27
28impl From<EventOutcome> for EventResult {
29    fn from(o: EventOutcome) -> Self {
30        EventResult {
31            outcome: o,
32            no_render: false,
33        }
34    }
35}
36
37impl EventResult {
38    /// Convenience to signal that the event was handled and should trigger a render.
39    pub fn handled() -> Self {
40        EventOutcome::Handle.into()
41    }
42
43    /// Convenience to signal that the event was ignored.
44    pub fn ignored() -> Self {
45        EventOutcome::Ignore.into()
46    }
47
48    /// Convenience to mark a handled event that should not trigger rendering.
49    pub fn no_render(mut self) -> Self {
50        self.no_render = true;
51        self
52    }
53}

56/// Nodes are the basic building-blocks of a Canopy UI. They are composed in a tree, with each node responsible for
57/// managing its own children.
58pub trait Node: StatefulNode + CommandNode {
59    /// Force the node to render in the next sweep. Over-riding this method should only be needed rarely, for instance
60    /// when a container node (e.g. a frame) needs to redraw if a child node changes.
61    fn force_render(&self, c: &dyn Context) -> bool {
62        false
63    }
64
65    /// Called for each node on the focus path, after each render sweep. The first node that returns a
66    /// ``cursor::Cursor`` specification controls the cursor. If no node returns a cursor, cursor display is disabled.
67    fn cursor(&self) -> Option<cursor::Cursor> {
68        None
69    }
70
71    /// Attempt to focus this node. If the node accepts focus, it should return true, and if not return false. The
72    /// default implementation returns false.
73    fn accept_focus(&mut self) -> bool {
74        false
75    }
76
77    /// Call a closure on this node's children. If any child handler returns an error, processing terminates without
78    /// visiting the remaining children. The order in which nodes are processed should match intuitive next/previous
79    /// relationships. The default implementation assumes this node has no children, and just returns.
80    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
81        Ok(())
82    }
83
84    /// Re-compute the size and view of the node if it had to be displayed in the target area. In practice, nodes will
85    /// usually either constrain themselves based on the width or the height of the target area, or neither, but not
86    /// both. The resulting size may be smaller or larger than the target. If non-trivial computation is done to compute
87    /// the size (e.g. reflowing text), it should be cached for use by future calls. This method may be called multiple
88    /// times for a given node during a render sweep, so re-fitting to the same size should be cheap and return
89    /// consistent results.
90    fn layout(&mut self, l: &Layout, target: Expanse) -> Result<()> {
91        Ok(())
92    }
93
94    /// Handle a key input event. This event is only called for nodes that are on the focus path. The default
95    /// implementation ignores input.
96    fn handle_key(&mut self, c: &mut dyn Context, k: key::Key) -> Result<EventResult> {
97        Ok(EventResult::ignored())
98    }
99
100    /// Handle a mouse input event. The default implementation ignores mouse input.
101    fn handle_mouse(&mut self, c: &mut dyn Context, k: mouse::MouseEvent) -> Result<EventResult> {
102        Ok(EventResult::ignored())
103    }
104
105    /// The scheduled poll endpoint. This function is called for every node the first time it is seen during the
106    /// pre-render sweep. Each time the function returns a duration, a subsequent call is scheduled. If the function
107    /// returns None, the `poll` function is never called again. The default implementation returns `None`.
108    fn poll(&mut self, c: &mut dyn Context) -> Option<Duration> {
109        None
110    }
111
112    /// Render this widget. The render method should:
113    ///
114    /// - Lay out any child nodes by manipulating their viewports. This will often involve calling the `fit` method on
115    ///   the child nodes to get their dimensions.
116    /// - Render itself to screen. This node's viewport will already have been set by a parent.
117    ///
118    /// The default implementation does nothing.
119    fn render(&mut self, c: &dyn Context, r: &mut Render) -> Result<()> {
120        Ok(())
121    }
122}
