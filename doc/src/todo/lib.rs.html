78pub struct Todo {

85#[derive_commands]
86impl Todo {
87    pub fn new() -> Result<Self> {
88        let mut r = Todo {
89            state: NodeState::default(),
90            content: frame::Frame::new(List::new(vec![])),
91            statusbar: StatusBar {
92                state: NodeState::default(),
93            },
94            adder: None,
95        };
96        r.load()?;
97        Ok(r)
98    }
99
100    #[command]
101    pub fn enter_item(&mut self, c: &mut dyn Context) -> canopy::Result<()> {
102        let mut adder = frame::Frame::new(Input::new(""));
103        c.set_focus(&mut adder.child);
104        self.adder = Some(adder);
105        c.taint(self);
106        Ok(())
107    }
108
109    #[command]
110    pub fn delete_item(&mut self, c: &mut dyn Context) -> canopy::Result<()> {
111        let lst = &mut self.content.child;
112        if let Some(t) = lst.selected() {
113            store::get().delete_todo(t.todo.id).unwrap();
114            lst.delete_selected(c);
115            c.taint_tree(self);
116        }
117        Ok(())
118    }
119
120    #[command]
121    pub fn accept_add(&mut self, c: &mut dyn Context) -> canopy::Result<()> {
122        if let Some(adder) = self.adder.take() {
123            let value = adder.child.textbuf.value;
124            if !value.is_empty() {
125                let item = store::get().add_todo(&value).unwrap();
126                self.content.child.append(TodoItem::new(item));
127                // Select the newly added item (which is the last one)
128                let new_index = self.content.child.len().saturating_sub(1);
129                self.content.child.select(new_index);
130            }
131        }
132        c.taint_tree(self);
133        c.set_focus(&mut self.content);
134        Ok(())
135    }
136
137    #[command]
138    pub fn cancel_add(&mut self, c: &mut dyn Context) -> canopy::Result<()> {
139        self.adder = None;
140        c.taint_tree(self);
141        c.focus_first(self);
142        Ok(())
143    }
144
145    fn load(&mut self) -> canopy::Result<()> {
146        let s = store::get().todos().unwrap();
147        let todos = s.iter().map(|x| TodoItem::new(x.clone()));
148        for i in todos {
149            self.content.child.append(i);
150        }
151        // Select the first item if any exist
152        if !self.content.child.is_empty() {
153            self.content.child.select(0);
154        }
155        Ok(())
156    }
157
158    #[command]
159    pub fn select_first(&mut self, c: &mut dyn Context) -> canopy::Result<()> {
160        self.content.child.select_first(c);
161        Ok(())
162    }

189impl Node for Todo {
190    fn layout(&mut self, l: &Layout, sz: Expanse) -> canopy::Result<()> {
191        l.fill(self, sz)?;
192        let vp = self.vp();
193        let (a, b) = vp.view().carve_vend(1);
194        l.place(&mut self.statusbar, vp, b)?;
195        l.place(&mut self.content, vp, a)?;
196
197        let a = self.vp().screen_rect();
198        if let Some(add) = &mut self.adder {
199            l.place(add, vp, Rect::new(a.tl.x + 2, a.tl.y + a.h / 2, a.w - 4, 3))?;
200        }
201        Ok(())
202    }
203
204    fn accept_focus(&mut self) -> bool {
205        true
206    }
207
208    fn children(
209        &mut self,
210        f: &mut dyn FnMut(&mut dyn Node) -> canopy::Result<()>,
211    ) -> canopy::Result<()> {
212        f(&mut self.statusbar)?;
213        f(&mut self.content)?;
214        if let Some(a) = &mut self.adder {
215            f(a)?;
216        }
217        Ok(())
218    }
219}

240pub fn bind_keys(cnpy: &mut Canopy) {
241    Binder::new(cnpy)
242        .with_path("todo/")
243        .key('q', "root::quit()")
244        .key('d', "todo::delete_item()")
245        .key('a', "todo::enter_item()")
246        .key('g', "todo::select_first()")
247        .key('j', "todo::select_next()")
248        .key(key::KeyCode::Down, "todo::select_next()")
249        .key('k', "todo::select_prev()")
250        .key(key::KeyCode::Up, "todo::select_prev()")
251        .key(' ', "todo::page_down()")
252        .key(key::KeyCode::PageDown, "todo::page_down()")
253        .key(key::KeyCode::PageUp, "todo::page_up()")
254        .mouse(mouse::Action::ScrollUp, "todo::select_prev()")
255        .mouse(mouse::Action::ScrollDown, "todo::select_next()")
256        .with_path("input")
257        .key(key::KeyCode::Left, "input::left()")
258        .key(key::KeyCode::Right, "input::right()")
259        .key(key::KeyCode::Backspace, "input::backspace()")
260        .key(key::KeyCode::Enter, "todo::accept_add()")
261        .key(key::KeyCode::Esc, "todo::cancel_add()");
262}
