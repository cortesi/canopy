47impl Widget for TodoEntry {
48    fn layout(&self) -> Layout {
49        // Flex horizontally but use Measure for height so scrolling works
50        Layout::column().flex_horizontal(1)
51    }
52
53    fn measure(&self, c: MeasureConstraints) -> Measurement {
54        let available_width = match c.width {
55            Constraint::Exact(n) | Constraint::AtMost(n) => n.max(1),
56            Constraint::Unbounded => 80,
57        };
58        let width = available_width as usize;
59        let lines = textwrap::wrap(&self.todo.item, width);
60        let height = lines.len().max(1) as u32;
61        c.clamp(Size::new(available_width, height))
62    }
63
64    fn render(&mut self, rndr: &mut Render, ctx: &dyn ViewContext) -> Result<()> {
65        let view = ctx.view();
66        let area = view.view_rect_local();
67
68        if area.w == 0 || area.h == 0 {
69            return Ok(());
70        }
71
72        // Column 0: Selection indicator (when selected)
73        if self.selected && area.w >= 1 {
74            let indicator_rect = Rect::new(area.tl.x, area.tl.y, 1, area.h);
75            rndr.fill("list/selected", indicator_rect, '\u{2588}')?;
76        }
77
78        // Column 1: Spacer
79        if area.w >= 2 {
80            let spacer = Rect::new(area.tl.x + 1, area.tl.y, 1, area.h);
81            rndr.fill("", spacer, ' ')?;
82        }
83
84        // Text content starts at column 2
85        let text_start_x = area.tl.x + 2;
86        let text_visible_width = area.w.saturating_sub(2);
87
88        if text_visible_width > 0 {
89            let width = text_visible_width as usize;
90            let lines = textwrap::wrap(&self.todo.item, width);
91            for (i, line) in lines.iter().enumerate().take(area.h as usize) {
92                let line_rect =
93                    Rect::new(text_start_x, area.tl.y + i as u32, text_visible_width, 1);
94                rndr.text("text", line_rect.line(0), line)?;
95            }
96        }
97
98        Ok(())
99    }
100
101    fn accept_focus(&self, _ctx: &dyn ViewContext) -> bool {
102        true
103    }
104
105    fn name(&self) -> NodeName {
106        NodeName::convert("todo_entry")
107    }
108}

140/// Root node for the todo demo.
141pub struct Todo {
142    pending: Vec<store::Todo>,
143    /// Container holding the main content (list + status).
144    main_content_id: Option<NodeId>,
145    list_id: Option<NodeId>,
146    status_id: Option<NodeId>,
147    /// Modal widget for the add dialog.
148    modal_id: Option<NodeId>,
149    input_id: Option<NodeId>,
150    adder_active: bool,
151}
152
153#[derive_commands]
154impl Todo {
155    pub fn new() -> AnyResult<Self> {
156        let pending = store::get().todos()?;
157        Ok(Self {
158            pending,
159            main_content_id: None,
160            list_id: None,
161            status_id: None,
162            modal_id: None,
163            input_id: None,
164            adder_active: false,
165        })
166    }
167
168    fn ensure_tree(&mut self, c: &mut dyn Context) -> Result<()> {
169        if self.main_content_id.is_some() {
170            return Ok(());
171        }
172
173        // Create the main content container (list + status bar in column layout)
174        let main_content_id = c.add_orphan(MainContent);
175        let list_id = c.add_orphan(List::<TodoEntry>::new());
176        let frame_id = c.add_orphan(frame::Frame::new());
177        c.mount_child_to(frame_id, list_id)?;
178
179        let status_id = c.add_orphan(StatusBar);
180        c.set_children_of(main_content_id, vec![frame_id, status_id])?;
181
182        // Set Todo (self) to use Stack direction for modal overlay support
183        c.with_layout(&mut |layout| {
184            *layout = Layout::fill().direction(Direction::Stack);
185        })?;
186
187        // Main content fills the space
188        c.with_layout_of(main_content_id, &mut |layout| {
189            *layout = Layout::column().flex_horizontal(1).flex_vertical(1);
190        })?;
191
192        c.with_layout_of(frame_id, &mut |layout| {
193            layout.width = Sizing::Flex(1);
194            layout.height = Sizing::Flex(1);
195        })?;
196        c.with_layout_of(list_id, &mut |layout| {
197            *layout = Layout::fill();
198        })?;
199
200        c.with_layout_of(status_id, &mut |layout| {
201            *layout = Layout::row().flex_horizontal(1).fixed_height(1);
202        })?;
203
204        // Initially only show main content
205        c.set_children(vec![main_content_id])?;
206
207        self.main_content_id = Some(main_content_id);
208        self.list_id = Some(list_id);
209        self.status_id = Some(status_id);
210
211        if !self.pending.is_empty() {
212            let pending = std::mem::take(&mut self.pending);
213            self.with_list(c, |list, ctx| {
214                for item in pending.iter().cloned() {
215                    list.append(ctx, TodoEntry::new(item))?;
216                }
217                Ok(())
218            })?;
219        }
220
221        Ok(())
222    }
223
224    fn ensure_modal(&mut self, c: &mut dyn Context) -> Result<()> {
225        if self.modal_id.is_some() {
226            return Ok(());
227        }
228
229        // Create the modal with an input frame
230        let modal_id = c.add_orphan(Modal::new());
231        let input_id = c.add_orphan(Input::new(""));
232        let adder_frame_id = c.add_orphan(frame::Frame::new());
233        c.mount_child_to(adder_frame_id, input_id)?;
234        c.mount_child_to(modal_id, adder_frame_id)?;
235
236        c.with_layout_of(adder_frame_id, &mut |layout| {
237            layout.width = Sizing::Flex(1);
238            layout.min_height = Some(3);
239            layout.max_height = Some(3);
240            layout.min_width = Some(30);
241            layout.max_width = Some(50);
242        })?;
243
244        c.with_layout_of(input_id, &mut |layout| {
245            *layout = Layout::fill();
246        })?;
247
248        self.modal_id = Some(modal_id);
249        self.input_id = Some(input_id);
250
251        Ok(())
252    }
253
254    fn sync_children(&mut self, c: &mut dyn Context) -> Result<()> {
255        let main_content_id = self.main_content_id.expect("main content not initialized");
256
257        // With Stack layout: main content is always first, modal overlays on top when active
258        let mut children = vec![main_content_id];
259        if self.adder_active {
260            self.ensure_modal(c)?;
261            if let Some(modal_id) = self.modal_id {
262                children.push(modal_id);
263                // Dim the background content
264                c.push_effect(main_content_id, effects::dim(0.5))?;
265            }
266        } else {
267            // Clear dimming when modal is not active
268            c.clear_effects(main_content_id)?;
269        }
270        c.set_children(children)?;
271        Ok(())
272    }
273
274    fn with_list<F>(&mut self, c: &mut dyn Context, mut f: F) -> Result<()>
275    where
276        F: FnMut(&mut List<TodoEntry>, &mut dyn Context) -> Result<()>,
277    {
278        let list_id = self.list_id.expect("list not initialized");
279        c.with_widget(list_id, |list: &mut List<TodoEntry>, ctx| f(list, ctx))
280    }
281
282    fn with_input<F>(&mut self, c: &mut dyn Context, mut f: F) -> Result<()>
283    where
284        F: FnMut(&mut Input) -> Result<()>,
285    {
286        let input_id = self.input_id.expect("input not initialized");
287        c.with_widget_mut(input_id, &mut |widget, _ctx| {
288            let any = widget as &mut dyn Any;
289            let input = any.downcast_mut::<Input>().expect("input type mismatch");
290            f(input)
291        })
292    }

294    #[command]
295    pub fn enter_item(&mut self, c: &mut dyn Context) -> Result<()> {
296        self.ensure_tree(c)?;
297        self.ensure_modal(c)?;
298        self.adder_active = true;
299        self.sync_children(c)?;
300        self.with_input(c, |input| {
301            input.set_value("");
302            Ok(())
303        })?;
304        if let Some(input_id) = self.input_id {
305            c.set_focus(input_id);
306        }
307        Ok(())
308    }

310    #[command]
311    pub fn delete_item(&mut self, c: &mut dyn Context) -> Result<()> {
312        // Get the selected item's todo id before deleting
313        let mut to_delete = None;
314
315        self.with_list(c, |list, ctx| {
316            if let Some(item_id) = list.selected_item() {
317                ctx.with_widget(item_id.into(), |entry: &mut TodoEntry, _| {
318                    to_delete = Some(entry.todo.id);
319                    Ok(())
320                })?;
321            }
322            list.delete_selected(ctx)?;
323            Ok(())
324        })?;
325
326        if let Some(id) = to_delete {
327            store::get().delete_todo(id).unwrap();
328        }
329        Ok(())
330    }

332    #[command]
333    pub fn accept_add(&mut self, c: &mut dyn Context) -> Result<()> {
334        let mut value = String::new();
335        self.with_input(c, |input| {
336            value = input.value().to_string();
337            Ok(())
338        })?;
339
340        if !value.is_empty() {
341            let item = store::get().add_todo(&value).unwrap();
342            self.with_list(c, |list, ctx| {
343                list.append(ctx, TodoEntry::new(item.clone()))?;
344                list.select_last(ctx);
345                Ok(())
346            })?;
347        }
348
349        self.adder_active = false;
350        self.sync_children(c)?;
351        c.set_focus(c.node_id());
352        Ok(())
353    }

480    let mut cnpy = Canopy::new();
