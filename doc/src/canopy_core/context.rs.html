3/// The API exposed to nodes by Canopy.
4pub trait Context {
5    /// Does the node need to render in the next sweep? Returns true for all visible nodes.
6    fn needs_render(&self, n: &dyn Node) -> bool;
7
8    /// Is the specified node on the focus path? A node is on the focus path if it
9    /// has focus, or if it's the ancestor of a node with focus.
10    fn is_on_focus_path(&self, n: &mut dyn Node) -> bool;
11
12    /// Does the node have focus?
13    fn is_focused(&self, n: &dyn Node) -> bool;
14
15    /// Move focus downward of the currently focused node within the subtree at root.
16    fn focus_down(&mut self, root: &mut dyn Node);
17
18    /// Focus the first node that accepts focus in the pre-order traversal of the subtree at root.
19    fn focus_first(&mut self, root: &mut dyn Node);
20
21    /// Move focus to the left of the currently focused node within the subtree at root.
22    fn focus_left(&mut self, root: &mut dyn Node);
23
24    /// Focus the next node in the pre-order traversal of root. If no node with focus is found, we focus the first node
25    /// we can find instead.
26    fn focus_next(&mut self, root: &mut dyn Node);
27
28    /// Return the focus path for the subtree under `root`.
29    fn focus_path(&self, root: &mut dyn Node) -> Path;
30
31    /// Focus the previous node in the pre-order traversal of `root`. If no node with focus is found, we focus the first
32    /// node we can find instead.
33    fn focus_prev(&mut self, root: &mut dyn Node);
34
35    /// Move focus to  right of the currently focused node within the subtree at root.
36    fn focus_right(&mut self, root: &mut dyn Node);
37
38    /// Move focus upward of the currently focused node within the subtree at root.
39    fn focus_up(&mut self, root: &mut dyn Node);
40
41    /// Focus a node. Returns `true` if focus changed.
42    fn set_focus(&mut self, n: &mut dyn Node) -> bool;
43
44    /// Move focus in a specified direction within the subtree at root.
45    fn focus_dir(&mut self, root: &mut dyn Node, dir: Direction);
46
47    /// Scroll the view to the specified position. The view is clamped within
48    /// the outer rectangle. Returns `true` if movement occurred.
49    fn scroll_to(&mut self, n: &mut dyn Node, x: u32, y: u32) -> bool {
50        let before = n.vp().view();
51        n.scroll_to(x, y);
52
53        before != n.vp().view()
54    }
55
56    /// Scroll the view by the given offsets. The view rectangle is clamped
57    /// within the outer rectangle. Returns `true` if movement occurred.
58    fn scroll_by(&mut self, n: &mut dyn Node, x: i32, y: i32) -> bool {
59        let before = n.vp().view();
60        n.scroll_by(x, y);
61
62        before != n.vp().view()
63    }
64
65    /// Scroll the view up by one page. Returns `true` if movement occurred.
66    fn page_up(&mut self, n: &mut dyn Node) -> bool {
67        let before = n.vp().view();
68        n.page_up();
69
70        before != n.vp().view()
71    }
72
73    /// Scroll the view down by one page. Returns `true` if movement occurred.
74    fn page_down(&mut self, n: &mut dyn Node) -> bool {
75        let before = n.vp().view();
76        n.page_down();
77
78        before != n.vp().view()
79    }
80
81    /// Scroll the view up by one line. Returns `true` if movement occurred.
82    fn scroll_up(&mut self, n: &mut dyn Node) -> bool {
83        let before = n.vp().view();
84        n.scroll_up();
85
86        before != n.vp().view()
87    }
88
89    /// Scroll the view down by one line. Returns `true` if movement occurred.
90    fn scroll_down(&mut self, n: &mut dyn Node) -> bool {
91        let before = n.vp().view();
92        n.scroll_down();
93
94        before != n.vp().view()
95    }
96
97    /// Scroll the view left by one line. Returns `true` if movement occurred.
98    fn scroll_left(&mut self, n: &mut dyn Node) -> bool {
99        let before = n.vp().view();
100        n.scroll_left();
101
102        before != n.vp().view()
103    }
104
105    /// Scroll the view right by one line. Returns `true` if movement occurred.
106    fn scroll_right(&mut self, n: &mut dyn Node) -> bool {
107        let before = n.vp().view();
108        n.scroll_right();
109
110        before != n.vp().view()
111    }
112
113    /// Start the backend renderer.
114    fn start(&mut self) -> Result<()>;
115
116    /// Stop the backend renderer, releasing control of the terminal.
117    fn stop(&mut self) -> Result<()>;
118
119    /// Stop the render backend and exit the process.
120    fn exit(&mut self, code: i32) -> !;
121
122    /// Current focus generation counter.

124        0
