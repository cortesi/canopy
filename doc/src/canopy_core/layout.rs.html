1//! Helper functions for `Node::layout` implementations.
2
3use crate::{Node, Result, geom::Rect};
4
5/// The Layout struct provides operations that a node can perform on children during its layout
6/// phase.
7pub struct Layout {}
8
9impl Layout {
10    /// Hides the element and all its descendants from rendering. The nodes are still included in
11    /// the tree.
12    pub fn hide(&self, child: &mut dyn Node) {
13        child.state_mut().hidden = true;
14    }
15
16    /// Unhides the element and all its descendants, allowing them to be rendered again.
17    pub fn unhide(&self, child: &mut dyn Node) {
18        child.state_mut().hidden = false;
19    }
20
21    /// Lay the child out and place it in a given sub-rectangle of a parent's canvas.
22    pub fn place(&self, child: &mut dyn Node, loc: Rect) -> Result<()> {
23        child.layout(self, loc.into())?;
24        child.state_mut().set_position(loc.tl);
25        Ok(())
26    }
27
28    /// Adjust a child node so that it fits a viewport. This lays the node out to the parent's view
29    /// rectangle, then adjusts the node's position to match.
30    pub fn fit(&self, n: &mut dyn Node, view: Rect) -> Result<()> {
31        n.layout(self, view.into())?;
32        n.state_mut().set_position(view.tl);
33        Ok(())
34    }
35}
