7/// The trait implemented by renderers.
8pub trait RenderBackend {
9    /// Apply a style to the following text output
10    fn style(&mut self, style: Style) -> Result<()>;
11    /// Output text to screen. This method is used for all text output.
12    fn text(&mut self, loc: geom::Point, txt: &str) -> Result<()>;
13    /// Flush output to the terminal.
14    fn flush(&mut self) -> Result<()>;
15    /// Exit the process, relinquishing screen control.
16    fn exit(&mut self, code: i32) -> !;
17    fn reset(&mut self) -> Result<()>;
18}

20/// A renderer that only renders to a specific rectangle within an expanse.
21pub struct Render<'a> {
22    /// The terminal buffer to render to.
23    buf: TermBuf,
24    /// The style manager used to apply styles.
25    pub style: &'a mut StyleManager,
26    /// The style map used to resolve style names to styles.
27    stylemap: &'a StyleMap,
28    /// The expanse that defines the total area.
29    expanse: geom::Expanse,
30    /// The rectangle within the expanse that we render to.
31    rect: geom::Rect,
32}
33
34impl<'a> Render<'a> {
35    pub fn new(
36        stylemap: &'a StyleMap,
37        style: &'a mut StyleManager,
38        canvas: geom::Expanse,
39        rect: geom::Rect,
40    ) -> Self {
41        let buf = TermBuf::new(
42            (rect.w, rect.h),
43            '\0',
44            Style {
45                fg: Color::White,
46                bg: Color::Black,
47                attrs: AttrSet::default(),
48            },
49        );
50        Render {
51            buf,
52            style,
53            stylemap,
54            expanse: canvas,
55            rect,
56        }
57    }
58
59    /// Fill a rectangle with a specified character.
60    pub fn fill(&mut self, style: &str, r: geom::Rect, c: char) -> Result<()> {
61        // Check if the rectangle is within the canvas bounds
62        let canvas_rect = self.expanse.rect();
63        if !canvas_rect.contains_rect(&r) {
64            return Err(Error::Geometry(
65                "Rectangle extends outside canvas bounds".to_string(),
66            ));
67        }
68
69        // Check if the rectangle intersects with our render rectangle
70        if let Some(intersection) = r.intersect(&self.rect) {
71            let style = self.style.get(self.stylemap, style);
72            // Adjust the intersection to be relative to our buffer's origin
73            let adjusted = geom::Rect::new(
74                intersection.tl.x - self.rect.tl.x,
75                intersection.tl.y - self.rect.tl.y,
76                intersection.w,
77                intersection.h,
78            );
79            self.buf.fill(style, adjusted, c);
80        }
81        Ok(())
82    }
83
84    /// Draw a solid frame
85    pub fn solid_frame(&mut self, style: &str, f: geom::Frame, c: char) -> Result<()> {
86        self.fill(style, f.top, c)?;
87        self.fill(style, f.left, c)?;
88        self.fill(style, f.right, c)?;
89        self.fill(style, f.bottom, c)?;
90        self.fill(style, f.topleft, c)?;
91        self.fill(style, f.topright, c)?;
92        self.fill(style, f.bottomleft, c)?;
93        self.fill(style, f.bottomright, c)?;
94        Ok(())
95    }
96
97    /// Print text in the specified line. If the text is wider than the
98    /// rectangle, it will be truncated; if it is shorter, it will be padded.
99    pub fn text(&mut self, style: &str, l: geom::Line, txt: &str) -> Result<()> {
100        // Convert line to rectangle to check intersection
101        let line_rect = geom::Rect::new(l.tl.x, l.tl.y, l.w, 1);
102
103        // Check if the line is within the canvas bounds
104        let canvas_rect = self.expanse.rect();
105        if !canvas_rect.contains_rect(&line_rect) {
106            return Err(Error::Geometry(
107                "Line extends outside canvas bounds".to_string(),
108            ));
109        }
110
111        if let Some(intersection) = line_rect.intersect(&self.rect) {
112            let style_res = self.style.get(self.stylemap, style);
113
114            // Calculate how much of the text to skip and take
115            let skip_amount = if l.tl.x < self.rect.tl.x {
116                (self.rect.tl.x - l.tl.x) as usize
117            } else {
118                0
119            };
120
121            let take_amount = intersection.w as usize;
122
123            let out = txt
124                .chars()
125                .skip(skip_amount)
126                .take(take_amount)
127                .collect::<String>();
128
129            // Adjust the line position relative to our buffer
130            let adjusted_line = geom::Line {
131                tl: geom::Point {
132                    x: intersection.tl.x - self.rect.tl.x,
133                    y: intersection.tl.y - self.rect.tl.y,
134                },
135                w: intersection.w,
136            };
137
138            self.buf.text(style_res.clone(), adjusted_line, &out);
139
140            // Pad with spaces if needed
141            if out.len() < adjusted_line.w as usize {
142                let pad_rect = geom::Rect::new(
143                    adjusted_line.tl.x + out.len() as u32,
144                    adjusted_line.tl.y,
145                    adjusted_line.w - out.len() as u32,
146                    1,
147                );
148                self.buf.fill(style_res, pad_rect, ' ');
149            }
150        }
151        Ok(())
152    }
153
154    /// Get a reference to the internal buffer
155    pub fn get_buffer(&self) -> &TermBuf {
156        &self.buf
157    }
158}
