1/// A terminal color value.
2#[derive(Copy, Clone, Debug, PartialEq, Eq, Ord, PartialOrd, Hash)]
3pub enum Color {
4    /// Black.
5    Black,
6    /// Dark grey.
7    DarkGrey,
8    /// Red.
9    Red,
10    /// Dark red.
11    DarkRed,
12    /// Green.
13    Green,
14    /// Dark green.
15    DarkGreen,
16    /// Yellow.
17    Yellow,
18    /// Dark yellow.
19    DarkYellow,
20    /// Blue.
21    Blue,
22    /// Dark blue.
23    DarkBlue,
24    /// Magenta.
25    Magenta,
26    /// Dark magenta.
27    DarkMagenta,
28    /// Cyan.
29    Cyan,
30    /// Dark cyan.
31    DarkCyan,
32    /// White.
33    White,
34    /// Grey.
35    Grey,
36    /// RGB color.
37    Rgb {
38        /// Red channel.
39        r: u8,
40        /// Green channel.
41        g: u8,
42        /// Blue channel.
43        b: u8,
44    },
45
46    /// An ANSI color. See [256 colors - cheat
47    /// sheet](https://jonasjacek.github.io/colors/) for more info.
48    AnsiValue(u8),
49}

51/// Macro to create a Color from a hex string at compile time
52#[macro_export]
53macro_rules! rgb {
54    ($hex:literal) => {{
55        const fn hex_char_to_num(c: u8) -> u8 {
56            match c {
57                b'0'..=b'9' => c - b'0',
58                b'a'..=b'f' => c - b'a' + 10,
59                b'A'..=b'F' => c - b'A' + 10,
60                _ => panic!("Invalid hex character"),
61            }
62        }
63
64        const fn parse_hex_byte(high: u8, low: u8) -> u8 {
65            hex_char_to_num(high) * 16 + hex_char_to_num(low)
66        }
67
68        let bytes = $hex.as_bytes();
69        let start = if bytes[0] == b'#' { 1 } else { 0 };
70
71        if bytes.len() - start != 6 {
72            panic!("Invalid hex color: must be 6 hex digits");
73        }
74
75        Color::Rgb {
76            r: parse_hex_byte(bytes[start], bytes[start + 1]),
77            g: parse_hex_byte(bytes[start + 2], bytes[start + 3]),
78            b: parse_hex_byte(bytes[start + 4], bytes[start + 5]),
79        }
80    }};
81}

83impl Color {
84    /// Construct a color from a hex RGB string.
85    /// Accepts "#RRGGBB" or "RRGGBB" and panics on invalid input.
86    pub fn rgb(hex: &str) -> Self {
87        let hex = hex.trim_start_matches('#');
88
89        if hex.len() != 6 {
90            panic!(
91                "Invalid hex color string: expected 6 hex digits, got {}",
92                hex.len()
93            );
94        }
95
96        let r = u8::from_str_radix(&hex[0..2], 16)
97            .expect("Invalid hex color: failed to parse red component");
98        let g = u8::from_str_radix(&hex[2..4], 16)
99            .expect("Invalid hex color: failed to parse green component");
100        let b = u8::from_str_radix(&hex[4..6], 16)
101            .expect("Invalid hex color: failed to parse blue component");
102
103        Self::Rgb { r, g, b }
104    }
105}

107#[cfg(test)]

111    #[test]
112    fn test_rgb_from_hex() {
113        let test_cases = vec![
114            // (input, expected_r, expected_g, expected_b)
115            ("#FF0000", 255, 0, 0),     // Red
116            ("FF0000", 255, 0, 0),      // Red without #
117            ("#00FF00", 0, 255, 0),     // Green
118            ("00FF00", 0, 255, 0),      // Green without #
119            ("#0000FF", 0, 0, 255),     // Blue
120            ("0000FF", 0, 0, 255),      // Blue without #
121            ("#FFFFFF", 255, 255, 255), // White
122            ("#000000", 0, 0, 0),       // Black
123            ("#123456", 18, 52, 86),    // Random color
124            ("ABCDEF", 171, 205, 239),  // Hex with letters
125            ("abcdef", 171, 205, 239),  // Lowercase hex
126        ];
127
128        for (input, expected_r, expected_g, expected_b) in test_cases {
129            let color = Color::rgb(input);
130            match color {
131                Color::Rgb { r, g, b } => {
132                    assert_eq!(r, expected_r, "Red component mismatch for input: {input}");
133                    assert_eq!(g, expected_g, "Green component mismatch for input: {input}");
134                    assert_eq!(b, expected_b, "Blue component mismatch for input: {input}");
135                }
136                _ => panic!("Expected Color::Rgb variant for input: {input}"),
137            }
138        }
139    }
