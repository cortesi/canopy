7/// Traverses a tree of nodes and returns a string showing the node names and
8/// viewports for each node for visual display. This is a debug function.
9pub fn dump(root: &mut dyn Node) -> Result<String> {
10    let mut buffer = Buffer::ansi();
11    dump_node(&mut buffer, root, 0, None)?;
12    Ok(String::from_utf8_lossy(buffer.as_slice()).into_owned())
13}

15/// Traverses a tree of nodes and returns a string showing the node names and
16/// viewports for each node for visual display, with focus information.
17/// This is a debug function.
18pub fn dump_with_focus(root: &mut dyn Node, focus_gen: u64) -> Result<String> {
19    let mut buffer = Buffer::ansi();
20    dump_node(&mut buffer, root, 0, Some(focus_gen))?;
21    Ok(String::from_utf8_lossy(buffer.as_slice()).into_owned())

24/// Helper to write an indented, colored label followed by a value.
25fn write_field(buffer: &mut Buffer, indent: &str, label: &str, value: &str) {
26    write!(buffer, "{indent}  ").unwrap();
27    buffer
28        .set_color(ColorSpec::new().set_fg(Some(Color::Green)))
29        .unwrap();
30    write!(buffer, "{label}").unwrap();
31    buffer.reset().unwrap();
32    writeln!(buffer, " {value}").unwrap();
33}

35/// Walk a node subtree and emit formatted debug output.
36fn dump_node(
37    buffer: &mut Buffer,
38    node: &mut dyn Node,
39    level: usize,
40    focus_gen: Option<u64>,
41) -> Result<()> {
42    // Create indentation based on the level
43    let indent = "    ".repeat(level);
44
45    // Get node information
46    let id = node.id();
47    let viewport = node.vp();
48    let is_hidden = node.is_hidden();
49    let is_focused = focus_gen
50        .map(|fg| node.state().focus_gen == fg)
51        .unwrap_or(false);
52
53    // Write indent
54    write!(buffer, "{indent}").unwrap();
55
56    // Format the node name with bold and color
57    buffer
58        .set_color(ColorSpec::new().set_fg(Some(Color::Cyan)).set_bold(true))
59        .unwrap();
60    write!(buffer, "{id}").unwrap();
61    buffer.reset().unwrap();
62
63    // Add status indicators
64    let mut indicators = Vec::new();
65    if is_focused {
66        indicators.push("FOCUSED");
67    }
68    if is_hidden {
69        indicators.push("hidden");
70    }
71
72    if !indicators.is_empty() {
73        write!(buffer, " ").unwrap();
74
75        // Write each indicator with its own color
76        for (i, indicator) in indicators.iter().enumerate() {
77            if i > 0 {
78                write!(buffer, ", ").unwrap();
79            }
80            let color = match *indicator {
81                "FOCUSED" => Color::Magenta,
82                "hidden" => Color::Yellow,
83                _ => Color::White,
84            };
85            buffer
86                .set_color(ColorSpec::new().set_fg(Some(color)))
87                .unwrap();
88            write!(buffer, "{indicator}").unwrap();
89            buffer.reset().unwrap();
90        }
91    }
92    writeln!(buffer).unwrap();
93
94    // Format position
95    let pos = viewport.position();
96    write_field(
97        buffer,
98        &indent,
99        "pos in parent canvas:",
100        &format!("({}, {})", pos.x, pos.y),
101    );
102
103    // Format view rectangle
104    let view = viewport.view();
105    write_field(
106        buffer,
107        &indent,
108        "view:",
109        &format!(
110            "x: {}, y: {}, w: {}, h: {}",
111            view.tl.x, view.tl.y, view.w, view.h
112        ),
113    );
114
115    // Format canvas size
116    let canvas = viewport.canvas();
117    write_field(
118        buffer,
119        &indent,
120        "canvas:",
121        &format!("{} Ã— {}", canvas.w, canvas.h),
122    );
123
124    // Recursively dump children (skip if node is hidden)
125    if !is_hidden {
126        node.children(&mut |child| dump_node(buffer, child, level + 1, focus_gen))?;
127    }
128
129    Ok(())
130}
