4use crate::geom::Rect;
5use crate::viewport::ViewPort;
6use crate::{Error, Result};

8/// A stack of viewports that manages nested view transformations.
9///
10/// Invariants:
11/// - The stack always contains at least one viewport, enforced by:
12///   - `new()` requiring an initial viewport
13///   - `pop()` preventing removal of the last item
14/// - The first viewport's view represents the physical screen dimensions
15///   (i.e., its view size defines the screen size for the entire stack)
16pub struct ViewStack {
17    views: Vec<ViewPort>,
18}
19
20impl ViewStack {
21    pub fn new(initial: ViewPort) -> Self {
22        Self {
23            views: vec![initial],
24        }
25    }
26
27    pub fn push(&mut self, view: ViewPort) {
28        // Ensure the new viewport is positioned within the parent's canvas
29        // We know views always has at least one item
30        let parent = self.views.last().unwrap();
31        let parent_canvas = parent.canvas().rect();
32
33        assert!(
34            parent_canvas.contains_point(view.position()),
35            "ViewPort position {:?} is outside parent's canvas {:?}",
36            view.position(),
37            parent_canvas
38        );
39
40        // Check that the child's actual placement in the parent overlaps with parent's canvas
41        // The child occupies a rectangle at its position with size equal to its view size
42        let child_placement_in_parent = Rect::new(
43            view.position().x,
44            view.position().y,
45            view.view().w,
46            view.view().h,
47        );
48
49        assert!(
50            parent_canvas
51                .intersect(&child_placement_in_parent)
52                .is_some(),
53            "ViewPort at position {:?} with size {}x{} does not overlap with parent's canvas {:?}",
54            view.position(),
55            view.view().w,
56            view.view().h,
57            parent_canvas
58        );
59
60        self.views.push(view);
61    }
62
63    pub fn pop(&mut self) -> Result<ViewPort> {
64        if self.views.len() <= 1 {
65            return Err(Error::Geometry(
66                "Cannot pop the last viewport from stack".into(),
67            ));
68        }
69        Ok(self.views.pop().unwrap())
70    }
71
72    /// Get the top viewport on the stack
73    pub fn top(&self) -> &ViewPort {
74        self.views.last().unwrap()
75    }
76
77    /// Returns the physical screen dimensions as a rectangle rooted at (0,0).
78    ///
79    /// The size is determined by the first viewport's view, which by convention
80    /// represents the actual screen size. The position of the first viewport itself
81    /// is ignored - only its view dimensions matter.
82    pub fn root_screen(&self) -> Rect {
83        let v = self.views[0].view();
84        Rect::new(0, 0, v.w, v.h)
85    }
86
87    /// Calculates the projection from the final viewport's canvas to the screen.
88    ///
89    /// Returns a tuple of (canvas_rect, screen_rect) where:
90    /// - canvas_rect: The region in the final viewport's canvas that we're drawing
91    /// - screen_rect: The corresponding region on the screen where it will be displayed
92    ///
93    /// These rectangles always have the same dimensions but different positions:
94    /// canvas_rect is in the final viewport's coordinate system, while screen_rect
95    /// is in absolute screen coordinates.
96    ///
97    /// Returns None if the viewport stack results in no visible area (e.g., when
98    /// viewports are positioned outside their parent's visible area).
99    pub fn projection(&self) -> Option<(Rect, Rect)> {
100        // Start with the first viewport's view as the screen
101        // The first viewport's position is ignored since it represents the physical screen
102        let first_view = self.views[0].view();
103        let mut current_screen = Rect::new(0, 0, first_view.w, first_view.h);
104
105        // For each subsequent viewport, we need to calculate where its view
106        // appears on screen, taking into account its position within the parent
107        for i in 1..self.views.len() {
108            let viewport = &self.views[i];
109            let parent = &self.views[i - 1];
110
111            // Calculate where the child viewport appears in the parent's canvas
112            let child_rect_in_parent = Rect::new(
113                viewport.position().x,
114                viewport.position().y,
115                viewport.view().w,
116                viewport.view().h,
117            );
118
119            // Project this rectangle through the parent to screen coordinates
120            if let Some(visible_in_parent) = parent.view().intersect(&child_rect_in_parent) {
121                // Rebase the visible portion relative to parent's view
122                let rebased = parent.view().rebase_rect(&visible_in_parent).unwrap();
123                // Translate to screen coordinates by adding the offset from current_screen
124                let rect_on_screen = Rect {
125                    tl: current_screen
126                        .tl
127                        .scroll(rebased.tl.x as i32, rebased.tl.y as i32),
128                    w: rebased.w,
129                    h: rebased.h,
130                };
131                // Intersect with current screen area
132                current_screen = current_screen.intersect(&rect_on_screen)?;
133            } else {
134                // No overlap with parent's view
135                return None;
136            }
137        }
138
139        let screen_rect = current_screen;
140
141        // Now calculate the canvas rect by tracking through the viewport transformations
142        let canvas_rect = if self.views.len() == 1 {
143            self.views[0].view()
144        } else {
145            // Track the visible region through each viewport transformation
146            let mut visible_in_parent = self.views[0].view();
147
148            for i in 1..self.views.len() {
149                let viewport = &self.views[i];
150                let _parent = &self.views[i - 1];
151
152                // Calculate where the child appears in parent's canvas
153                let child_rect_in_parent = Rect::new(
154                    viewport.position().x,
155                    viewport.position().y,
156                    viewport.view().w,
157                    viewport.view().h,
158                );
159
160                // Find the intersection with what's visible in the parent
161                if let Some(visible_part) = visible_in_parent.intersect(&child_rect_in_parent) {
162                    // Transform this visible part to child's canvas coordinates
163                    // by shifting back by the child's position
164                    visible_in_parent = Rect {
165                        tl: visible_part
166                            .tl
167                            .scroll(
168                                -(viewport.position().x as i32),
169                                -(viewport.position().y as i32),
170                            )
171                            .scroll(viewport.view().tl.x as i32, viewport.view().tl.y as i32),
172                        w: visible_part.w,
173                        h: visible_part.h,
174                    };
175                } else {
176                    // No intersection - nothing visible
177                    return None;
178                }
179            }
180
181            // The final visible_in_parent is our canvas rect
182            visible_in_parent
183        };
184
185        // Verify the invariant that canvas_rect and screen_rect have the same size
186        debug_assert_eq!(
187            (canvas_rect.w, canvas_rect.h),
188            (screen_rect.w, screen_rect.h),
189            "canvas_rect and screen_rect must have the same dimensions - canvas: {canvas_rect:?}, screen: {screen_rect:?}"
190        );
191
192        Some((canvas_rect, screen_rect))
193    }
194}

200    struct TestCase {
201        name: &'static str,
202        viewports: Vec<((u32, u32), (u32, u32, u32, u32), (u32, u32))>,
203        projections: Vec<Option<((u32, u32, u32, u32), (u32, u32, u32, u32))>>,
204    }

206    impl TestCase {
207        fn run(&self) {
208            assert!(
209                !self.viewports.is_empty(),
210                "Test case must have at least one viewport"
211            );
212            assert_eq!(
213                self.viewports.len(),
214                self.projections.len(),
215                "Number of viewports must match number of projections for '{}'",
216                self.name
217            );
218
219            let first = &self.viewports[0];
220            let view = ViewPort::new(first.0, first.1, first.2).unwrap();
221            let mut stack = ViewStack::new(view);
222
223            // Check projection after first viewport
224            let projection = stack.projection();
225            let expected = self.projections[0].map(|(canvas, screen)| {
226                (
227                    Rect::new(canvas.0, canvas.1, canvas.2, canvas.3),
228                    Rect::new(screen.0, screen.1, screen.2, screen.3),
229                )
230            });
231            assert_eq!(
232                projection, expected,
233                "projection failed for '{}' after viewport 0",
234                self.name
235            );
236
237            // For single viewport tests, verify root_screen
238            if self.viewports.len() == 1 && self.projections[0].is_some() {
239                let root = stack.root_screen();
240                let (_, expected_screen) = self.projections[0].unwrap();
241                assert_eq!(
242                    (root.w, root.h),
243                    (expected_screen.2, expected_screen.3),
244                    "root_screen size must match screen size for '{}'",
245                    self.name
246                );
247            }
248
249            // Push remaining viewports and check projections
250            for (i, viewport) in self.viewports[1..].iter().enumerate() {
251                let view = ViewPort::new(viewport.0, viewport.1, viewport.2).unwrap();
252                stack.push(view);
253
254                let projection = stack.projection();
255                let expected = self.projections[i + 1].map(|(canvas, screen)| {
256                    (
257                        Rect::new(canvas.0, canvas.1, canvas.2, canvas.3),
258                        Rect::new(screen.0, screen.1, screen.2, screen.3),
259                    )
260                });
261                assert_eq!(
262                    projection,
263                    expected,
264                    "projection failed for '{}' after viewport {}",
265                    self.name,
266                    i + 1
267                );
268            }
269        }
270    }

273    fn test_basic_operations() {
274        // Test new()
275        let view1 = ViewPort::new((100, 50), (0, 0, 100, 50), (0, 0)).unwrap();
276        let stack = ViewStack::new(view1);
277        assert_eq!(stack.views.len(), 1);
278
279        // Test push() and pop()
280        let view2 = ViewPort::new((80, 40), (10, 10, 60, 30), (10, 10)).unwrap();

282
283        let mut stack = ViewStack::new(view1);
284        stack.push(view2);
285        stack.push(view3);
286
287        // Pop should return in LIFO order
288        assert_eq!(stack.pop().unwrap(), view3);
289        assert_eq!(stack.pop().unwrap(), view2);
290
291        // Cannot pop the last item
292        let result = stack.pop();
293        assert!(result.is_err());
294        assert_eq!(result.unwrap_err().to_string(), "geometry");
