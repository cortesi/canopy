1/// This is the UI the interface that all nodes must implement in Canopy.
2///
3/// Some explanation is necessary to justify the split between .layout() and .render(). Why not
4/// just combine these two methods into a single .draw() method? The reason is that nodes need to
5/// be able to account for the size of their children when laying themselves out. Imagine, for
6/// instance, a text container that wraps text and resizes itself to fit. The parent needs to be
7/// aware of the resulting size BEFORE it can decide where to place the child and render itself.
8/// Our solution is to split rendering and layout. During the layout phase, no drawing is done, but
9/// all nodes have their size and position determined. We do this by asking each node what their
10/// size would be IF they had to draw within a given target area. The result can be larger or
11/// smaller than the target, but gives the node the opportunity to do things like reflow text with
12/// a width constraint, and then tell the parent how much space it needs. During the render phase,
13/// size and position are fixed, and nodes only draw themselves to the screen.
14///
15/// Another key aspect of the design is that all rendering and layout happens with respect to the
16/// node's own canvas. Nodes don't have to (and can't) know where they are being drawn on the
17/// physical screen.

29/// The result of an event handler.
30#[derive(Debug, PartialEq, Eq, Clone)]
31pub enum EventOutcome {
32    /// The event was processed and the node should be rendered.
33    Handle,
34    /// The event was processed, but nothing changed so rendering is skipped and propagation stops.
35    Consume,
36    /// The event was not handled and will bubble up the tree.
37    Ignore,
38}

40#[allow(unused_variables)]
41/// Nodes are the basic building-blocks of a Canopy UI. They are composed in a tree, with each node responsible for
42/// managing its own children.
43pub trait Node: StatefulNode + CommandNode {
44    /// Force the node to render in the next sweep. Over-riding this method should only be needed rarely, for instance
45    /// when a container node (e.g. a frame) needs to redraw if a child node changes.
46    fn force_render(&self, c: &dyn Context) -> bool {
47        false
48    }
49
50    /// Called for each node on the focus path, after each render sweep. The first node that returns a
51    /// ``cursor::Cursor`` specification controls the cursor. If no node returns a cursor, cursor display is disabled.
52    fn cursor(&self) -> Option<cursor::Cursor> {
53        None
54    }
55
56    /// Attempt to focus this node. If the node accepts focus, it should return true, and if not return false. The
57    /// default implementation returns false.
58    fn accept_focus(&mut self) -> bool {
59        false
60    }
61
62    /// Call a closure on this node's children. If any child handler returns an error, processing terminates without
63    /// visiting the remaining children. The order in which nodes are processed should match intuitive next/previous
64    /// relationships. The default implementation assumes this node has no children, and just returns.
65    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
66        Ok(())
67    }
68
69    /// Handle a key input event. This event is only called for nodes that are on the focus path. The default
70    /// implementation ignores input.
71    fn handle_key(&mut self, c: &mut dyn Context, k: key::Key) -> Result<EventOutcome> {
72        Ok(EventOutcome::Ignore)
73    }
74
75    /// Handle a mouse input event. The default implementation ignores mouse input.
76    fn handle_mouse(&mut self, c: &mut dyn Context, k: mouse::MouseEvent) -> Result<EventOutcome> {
77        Ok(EventOutcome::Ignore)
78    }
79
80    /// The scheduled poll endpoint. This function is called for every node the first time it is seen during the
81    /// pre-render sweep. Each time the function returns a duration, a subsequent call is scheduled. If the function
82    /// returns None, the `poll` function is never called again. The default implementation returns `None`.
83    fn poll(&mut self, c: &mut dyn Context) -> Option<Duration> {
84        None
85    }
86
87    /// Re-compute the size and view of the node to display in the target area. In practice, nodes
88    /// will either constrain themselves based on the width or the height of the target area, or
89    /// neither, but not both. The resulting size may be smaller or larger than the target. If
90    /// non-trivial computation is done (e.g. reflowing text), it should be cached for use by
91    /// future calls.
92    fn layout(&mut self, l: &Layout, target: Expanse) -> Result<()> {
93        Ok(())
94    }
95
96    /// Render this widget, only drawing itself with reference to its own canvas.
97    fn render(&mut self, c: &dyn Context, r: &mut Render) -> Result<()> {
98        Ok(())
99    }
100}
