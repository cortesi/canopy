12/// A node that has a pending callback.
13#[derive(Debug)]
14struct PendingNode {
15    time: SystemTime,
16    node_id: NodeId,
17}
18
19impl PartialEq for PendingNode {
20    fn eq(&self, other: &Self) -> bool {
21        self.time == other.time
22    }
23}
24
25impl Eq for PendingNode {}
26
27/// Reverse order so nodes with the closest callback time are at the top.
28impl PartialOrd for PendingNode {
29    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
30        Some(other.time.cmp(&self.time))
31    }
32}
33
34/// Reverse order so nodes with the closest callback time are at the top.
35impl Ord for PendingNode {
36    fn cmp(&self, other: &Self) -> Ordering {
37        other.time.cmp(&self.time)
38    }
39}

41/// A heap that tracks the current list of pending callbacks.
42#[derive(Default, Debug)]
43struct PendingHeap {
44    nodes: BinaryHeap<PendingNode>,
45}
46
47impl PendingHeap {
48    fn _add(&mut self, now: SystemTime, node_id: NodeId, duration: Duration) {
49        self.nodes.push(PendingNode {
50            time: now + duration,
51            node_id,
52        });
53    }
54
55    /// Add a node with a callback duration to the heap.
56    fn add(&mut self, node_id: NodeId, duration: Duration) {
57        self._add(SystemTime::now(), node_id, duration);
58    }
59
60    fn _current_wait(&self, now: SystemTime) -> Option<Duration> {
61        self.nodes
62            .peek()
63            .map(|top| top.time.duration_since(now).unwrap_or(Duration::new(0, 0)))
64    }
65
66    /// Retrieve the current shortest wait time. We return None if no nodes are
67    /// waiting, and a duration of 0 if the current top-most node has a
68    /// scheduled time in the past.
69    fn current_wait(&self) -> Option<Duration> {
70        self._current_wait(SystemTime::now())
71    }
72
73    fn _collect(&mut self, now: SystemTime) -> Vec<NodeId> {
74        let mut v = vec![];
75        while let Some(n) = self.nodes.pop() {
76            if n.time <= now {
77                v.push(n.node_id);
78            } else {
79                // Put it back on the heap.
80                self.nodes.push(n);
81                break;
82            }
83        }
84        v
85    }
86
87    /// Remove and return all the pending operations .
88    pub fn collect(&mut self) -> Vec<NodeId> {
89        self._collect(SystemTime::now())
90    }
91}

93/// The Poller is responsible for scheduling poll events for nodes.
94#[derive(Debug)]
95pub struct Poller {
96    /// Handle for the scheduler thread
97    handle: Option<thread::JoinHandle<()>>,
98    pending: Arc<Mutex<PendingHeap>>,
99    event_tx: mpsc::Sender<Event>,
100}
101
102impl Poller {
103    pub(crate) fn new(event_tx: mpsc::Sender<Event>) -> Self {
104        Self {
105            handle: None,
106            pending: Arc::new(Mutex::new(PendingHeap::default())),
107            event_tx,
108        }
109    }
110
111    /// Schedule a node to be polled. This function requires us to pass in the
112    /// tx channel, which means that a lock over the global state must already
113    /// be in place.
114    pub fn schedule(&mut self, node_id: NodeId, duration: Duration) {
115        let mut l = self.pending.lock().unwrap();
116        l.add(node_id, duration);
117        if let Some(h) = self.handle.as_mut() {
118            // The thread is running, let's wake it up.
119            h.thread().unpark();
120        } else {
121            let pending = self.pending.clone();
122            let tx = self.event_tx.clone();
123            self.handle = Some(thread::spawn(move || {
124                loop {
125                    // Caution: moving this into the statement below means that we
126                    // retain the lock over the thread park, causing deadlock.
127                    let d = pending.lock().unwrap().current_wait();
128                    if let Some(d) = d {
129                        thread::park_timeout(d);
130                    } else {
131                        // We have no current wait time, so we just park the thread.
132                        thread::park();
133                    };
134                    let ids = pending.lock().unwrap().collect();
135                    if !ids.is_empty() && tx.send(Event::Poll(ids)).is_err() {
136                        break;
137                    }
138                }
139            }));
140        }
141    }
142}

144#[cfg(test)]
145mod tests {
146    use super::*;
147    use crate::tutils::ttree::{BaLa, BaLb};
148    use crate::{Result, StatefulNode};
149    #[test]
150    fn pendingheap() -> Result<()> {
151        let now = SystemTime::now();
152
153        let mut ph = PendingHeap::default();
154        let n1 = BaLa::new();
155        let n2 = BaLb::new();
156
157        assert_eq!(ph._current_wait(now), None);
158        ph._add(now, n1.id(), Duration::from_secs(10));
159        assert_eq!(ph._current_wait(now).unwrap(), Duration::from_secs(10));
160        ph._add(now, n2.id(), Duration::from_secs(100));
161        assert!(ph._current_wait(now).unwrap() <= Duration::from_secs(10));
162        assert_eq!(
163            ph._collect(SystemTime::now() + Duration::from_secs(11)),
164            vec![n1.id()]
165        );
166        assert!(ph._current_wait(now).unwrap() <= Duration::from_secs(100));
167
168        Ok(())
169    }
