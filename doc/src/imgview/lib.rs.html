24/// Character used to render two vertical pixels per terminal cell.
25const HALF_BLOCK: char = '\u{2580}';
26/// Minimum zoom factor.
27const MIN_ZOOM: f32 = 0.1;
28/// Maximum zoom factor.
29const MAX_ZOOM: f32 = 16.0;
30/// Zoom multiplier applied per step.
31const ZOOM_STEP: f32 = 1.25;
32/// Pan step in terminal columns.
33const PAN_STEP_COLUMNS: i32 = 1;
34/// Pan step in terminal rows.
35const PAN_STEP_ROWS: i32 = 1;
36
37/// Current panning offset in image coordinates.
38#[derive(Clone, Copy, Debug, Default)]
39struct Pan {
40    /// Horizontal offset in image pixels.
41    column: f32,
42    /// Vertical offset in image pixels.
43    row: f32,
44}
45
46/// Widget that renders an image into terminal cells.
47pub struct ImageView {
48    /// Loaded image buffer.
49    image: RgbaImage,
50    /// Cached image width in pixels.
51    image_width: u32,
52    /// Cached image height in pixels.
53    image_height: u32,
54    /// Zoom factor in display subpixels per image pixel.
55    zoom: f32,
56    /// Current panning offset.
57    pan: Pan,
58    /// Whether the view should auto-fit the image to the terminal.
59    auto_fit: bool,
60}
61
62#[derive_commands]
63impl ImageView {
64    /// Convert the cached image width to a float.
65    fn image_width_f32(&self) -> f32 {
66        self.image_width as f32
67    }
68
69    /// Convert the cached image height to a float.
70    fn image_height_f32(&self) -> f32 {
71        self.image_height as f32
72    }
73
74    /// Convert the view width to display subpixels.
75    fn view_subpixel_width(view_size: Expanse) -> f32 {
76        view_size.w as f32
77    }
78
79    /// Convert the view height to display subpixels.
80    fn view_subpixel_height(view_size: Expanse) -> f32 {
81        view_size.h as f32 * 2.0
82    }
83
84    /// Clamp the pan offset so it stays within the image bounds.
85    fn clamp_pan(&mut self, view_size: Expanse) {
86        if self.image_width == 0 || self.image_height == 0 {
87            self.pan = Pan::default();
88            return;
89        }
90
91        let view_width = Self::view_subpixel_width(view_size) / self.zoom;
92        let view_height = Self::view_subpixel_height(view_size) / self.zoom;
93        let max_column = (self.image_width_f32() - view_width).max(0.0);
94        let max_row = (self.image_height_f32() - view_height).max(0.0);
95
96        self.pan.column = self.pan.column.clamp(0.0, max_column);
97        self.pan.row = self.pan.row.clamp(0.0, max_row);
98    }
99
100    /// Compute a zoom value that fits the entire image inside the view.
101    fn fit_zoom(&self, view_size: Expanse) -> f32 {
102        let image_width = self.image_width_f32();
103        let image_height = self.image_height_f32();
104        if image_width == 0.0 || image_height == 0.0 || view_size.w == 0 || view_size.h == 0 {
105            return 1.0;
106        }
107
108        let view_width = Self::view_subpixel_width(view_size);
109        let view_height = Self::view_subpixel_height(view_size);
110        let zoom_width = view_width / image_width;
111        let zoom_height = view_height / image_height;
112
113        zoom_width.min(zoom_height).clamp(0.0, MAX_ZOOM)
114    }
115
116    /// Apply automatic fit if enabled.
117    fn apply_auto_fit(&mut self, view_size: Expanse) {
118        if !self.auto_fit {
119            return;
120        }
121        if view_size.w == 0 || view_size.h == 0 {
122            return;
123        }
124
125        self.zoom = self.fit_zoom(view_size);
126        self.pan = Pan::default();
127        self.clamp_pan(view_size);
128    }
129
130    /// Zoom around the center of the current view.
131    fn zoom_by(&mut self, view_size: Expanse, factor: f32) {
132        let view_width = Self::view_subpixel_width(view_size);
133        let view_height = Self::view_subpixel_height(view_size);
134
135        let center_column = self.pan.column + view_width / (2.0 * self.zoom);
136        let center_row = self.pan.row + view_height / (2.0 * self.zoom);
137
138        let min_zoom = MIN_ZOOM.min(self.fit_zoom(view_size));
139        self.zoom = (self.zoom * factor).clamp(min_zoom, MAX_ZOOM);
140        self.pan.column = center_column - view_width / (2.0 * self.zoom);
141        self.pan.row = center_row - view_height / (2.0 * self.zoom);
142        self.clamp_pan(view_size);
143    }
144
145    /// Pan by a number of terminal cells.
146    fn pan_by_cells(&mut self, view_size: Expanse, delta_columns: i32, delta_rows: i32) {
147        let delta_subpixels_x = delta_columns as f32;
148        let delta_subpixels_y = delta_rows as f32 * 2.0;
149        self.pan_by_subpixels(view_size, delta_subpixels_x, delta_subpixels_y);
150    }
151
152    /// Pan by a number of display subpixels.
153    fn pan_by_subpixels(&mut self, view_size: Expanse, delta_columns: f32, delta_rows: f32) {
154        self.pan.column += delta_columns / self.zoom;
155        self.pan.row += delta_rows / self.zoom;
156        self.clamp_pan(view_size);
157    }
158
159    /// Compute the image-space bounds of a display subpixel.
160    fn subpixel_bounds(&self, subpixel_column: f32, subpixel_row: f32) -> (f32, f32, f32, f32) {
161        let inverse_zoom = 1.0 / self.zoom;
162        let left = self.pan.column + subpixel_column * inverse_zoom;
163        let right = self.pan.column + (subpixel_column + 1.0) * inverse_zoom;
164        let top = self.pan.row + subpixel_row * inverse_zoom;
165        let bottom = self.pan.row + (subpixel_row + 1.0) * inverse_zoom;
166        (left, top, right, bottom)
167    }
168
169    /// Sample a color from the image for a display subpixel.
170    fn sample_color(&self, subpixel_column: f32, subpixel_row: f32) -> Color {
171        let (left, top, right, bottom) = self.subpixel_bounds(subpixel_column, subpixel_row);
172        let center_column = (left + right) * 0.5;
173        let center_row = (top + bottom) * 0.5;
174        if center_column < 0.0
175            || center_row < 0.0
176            || center_column >= self.image_width_f32()
177            || center_row >= self.image_height_f32()
178        {
179            return Color::Black;
180        }
181
182        let Some((red, green, blue)) = self.sample_region(left, top, right, bottom) else {
183            return Color::Black;
184        };
185
186        Color::Rgb {
187            r: red,
188            g: green,
189            b: blue,
190        }
191    }
192
193    /// Compute the display subpixel offset to center the image in the view.
194    fn center_offset(&self, view_size: Expanse) -> (f32, f32) {
195        let view_width = Self::view_subpixel_width(view_size);
196        let view_height = Self::view_subpixel_height(view_size);
197        let image_width = self.image_width_f32() * self.zoom;
198        let image_height = self.image_height_f32() * self.zoom;
199
200        let offset_x = (view_width - image_width).max(0.0) / 2.0;
201        let offset_y = (view_height - image_height).max(0.0) / 2.0;
202
203        (offset_x, offset_y)
204    }
205
206    /// Sample a rectangular region in image space and return the average color.
207    fn sample_region(&self, left: f32, top: f32, right: f32, bottom: f32) -> Option<(u8, u8, u8)> {
208        if self.image_width == 0 || self.image_height == 0 {
209            return None;
210        }
211
212        let left_index = left.floor() as i32;
213        let right_index = right.ceil() as i32;
214        let top_index = top.floor() as i32;
215        let bottom_index = bottom.ceil() as i32;
216
217        let left_clamped = left_index.max(0);
218        let right_clamped = right_index.min(self.image_width as i32);
219        let top_clamped = top_index.max(0);
220        let bottom_clamped = bottom_index.min(self.image_height as i32);
221
222        if left_clamped >= right_clamped || top_clamped >= bottom_clamped {
223            return None;
224        }
225
226        let mut red_total: u32 = 0;
227        let mut green_total: u32 = 0;
228        let mut blue_total: u32 = 0;
229        let mut sample_count: u32 = 0;
230
231        for row_index in top_clamped..bottom_clamped {
232            for column_index in left_clamped..right_clamped {
233                let pixel = self.image.get_pixel(column_index as u32, row_index as u32);
234                let alpha = pixel[3] as u32;
235                red_total += (pixel[0] as u32 * alpha) / 255;
236                green_total += (pixel[1] as u32 * alpha) / 255;
237                blue_total += (pixel[2] as u32 * alpha) / 255;
238                sample_count += 1;
239            }
240        }
241
242        if sample_count == 0 {
243            return None;
244        }
245
246        let red = (red_total / sample_count) as u8;
247        let green = (green_total / sample_count) as u8;
248        let blue = (blue_total / sample_count) as u8;
249
250        Some((red, green, blue))
251    }

253    /// Render the image into the provided view rectangle.
254    fn render_cells(
255        &self,
256        render: &mut Render,
257        view: Rect,
258        offset: (f32, f32),
259    ) -> canopy_error::Result<()> {
260        let origin = view.tl;
261        let (offset_x, offset_y) = offset;
262
263        for row_index in 0..view.h {
264            let top_subpixel_row = row_index.saturating_mul(2);
265            let bottom_subpixel_row = top_subpixel_row.saturating_add(1);
266            let top_row = top_subpixel_row as f32 - offset_y;
267            let bottom_row = bottom_subpixel_row as f32 - offset_y;
268
269            for column_index in 0..view.w {
270                let column = column_index as f32 - offset_x;
271                let top_color = self.sample_color(column, top_row);
272                let bottom_color = self.sample_color(column, bottom_row);
273                let style = Style {
274                    fg: top_color,
275                    bg: bottom_color,
276                    attrs: AttrSet::default(),
277                };
278                let point = Point {
279                    x: origin.x + column_index,
280                    y: origin.y + row_index,
281                };
282                render.put_cell(style, point, HALF_BLOCK)?;
283            }
284        }
285
286        Ok(())
287    }

290        let image_width = image.width();
291        let image_height = image.height();
292        Self {
293            image,

302    /// Zoom in around the view center.
303    #[command]

305        let view_size = ctx.view().content_size();
306        self.auto_fit = false;

315        self.auto_fit = false;

357impl Widget for ImageView {
358    /// Fill the available space in the terminal view.
359    fn layout(&self) -> Layout {
360        Layout::fill()
361    }
362
363    /// Render the current image view into the terminal buffer.
364    fn render(&mut self, render: &mut Render, ctx: &dyn ViewContext) -> canopy_error::Result<()> {
365        let view = ctx.view().view_rect_local();
366        if view.w == 0 || view.h == 0 {
367            return Ok(());
368        }
369
370        let view_size = ctx.view().content_size();
371        self.apply_auto_fit(view_size);
372        self.clamp_pan(view_size);
373
374        let offset = self.center_offset(view_size);
375        self.render_cells(render, view, offset)
376    }
377
378    /// Accept focus so key bindings apply to this widget.
379    fn accept_focus(&self, _ctx: &dyn ViewContext) -> bool {
380        true
381    }
382}

391/// Configure key bindings for the image viewer.
392pub fn bind_keys(cnpy: &mut Canopy) {
393    Binder::new(cnpy)
394        .key('q', "root::quit()")
395        .with_path("image_view/")
396        .key('i', "image_view::zoom_in()")
397        .key('o', "image_view::zoom_out()")
398        .key('h', "image_view::pan_left()")
399        .key('j', "image_view::pan_down()")
400        .key('k', "image_view::pan_up()")
401        .key('l', "image_view::pan_right()")
402        .key(key::KeyCode::Left, "image_view::pan_left()")
403        .key(key::KeyCode::Right, "image_view::pan_right()")
404        .key(key::KeyCode::Up, "image_view::pan_up()")
405        .key(key::KeyCode::Down, "image_view::pan_down()");
406}

408/// Set up the Canopy app with the imgview widgets and bindings.
409pub fn setup_app(cnpy: &mut Canopy) {
410    Root::load(cnpy);
411    <ImageView as Loader>::load(cnpy);
412    bind_keys(cnpy);
413}

415/// Create a Canopy application for viewing the specified image.
416pub fn create_app(image_path: &Path) -> Result<Canopy> {
417    let image = image::open(image_path)?.to_rgba8();
418
419    let mut cnpy = Canopy::new();
420    setup_app(&mut cnpy);
421
422    let app_id = cnpy.core.add(ImageView::new(image));
423    Root::install(&mut cnpy.core, app_id)?;
424    Ok(cnpy)
425}

437    #[test]
438    fn fit_zoom_scales_down_below_min_zoom() {
439        let image = RgbaImage::new(2000, 1000);
440        let view = ImageView::new(image);
441        let zoom = view.fit_zoom(make_view(100, 25));
442        assert!(zoom < MIN_ZOOM);
443        assert!((zoom - 0.05).abs() < 0.0001);
444    }

447    fn fit_zoom_scales_up_when_view_is_larger() {
448        let image = RgbaImage::new(20, 10);
449        let view = ImageView::new(image);
450        let zoom = view.fit_zoom(make_view(100, 25));
451        assert!((zoom - 5.0).abs() < 0.0001);

456        let image = RgbaImage::new(2000, 1000);
457        let mut view = ImageView::new(image);
458        let view_size = make_view(100, 25);
459        view.zoom_by(view_size, 0.01);
460        let fit_zoom = view.fit_zoom(view_size);
461        assert!((view.zoom - fit_zoom).abs() < 0.0001);
462    }
