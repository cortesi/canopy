179/// Root widget for the stylegym demo.
180pub struct Stylegym {
181    /// Whether the modal is currently shown.
182    modal_visible: bool,
183    /// Current theme index.
184    current_theme: usize,
185    /// Available themes.
186    themes: Vec<ThemeOption>,
187}
188
189impl Default for Stylegym {
190    fn default() -> Self {
191        Self::new()
192    }
193}
194
195#[derive_commands]
196impl Stylegym {
197    /// Create a new stylegym instance.
198    pub fn new() -> Self {
199        Self {
200            modal_visible: false,
201            current_theme: 0,
202            themes: available_themes(),
203        }
204    }
205
206    /// Execute a closure with the right container widget.
207    fn with_right_container<F, R>(&self, c: &mut dyn Context, f: F) -> Result<R>
208    where
209        F: FnOnce(&mut Container, &mut dyn Context) -> Result<R>,
210    {
211        c.with_child::<RightContainerSlot, _>(f)
212    }
213
214    /// Execute a closure with the demo frame widget.
215    fn with_demo_frame<F, R>(&self, c: &mut dyn Context, f: F) -> Result<R>
216    where
217        F: FnOnce(&mut Frame, &mut dyn Context) -> Result<R>,
218    {
219        self.with_right_container(c, |_, ctx| ctx.with_child::<DemoFrameSlot, _>(f))
220    }
221
222    /// Show the modal overlay.
223    #[command]
224    pub fn show_modal(&mut self, c: &mut dyn Context) -> Result<()> {
225        if self.modal_visible {
226            return Ok(());
227        }
228        self.modal_visible = true;
229
230        self.with_right_container(c, |_, ctx| {
231            if ctx.has_child::<ModalSlot>() {
232                return Ok(());
233            }
234            let modal_id = ctx.add_keyed::<ModalSlot>(Modal::new())?;
235            let frame_id = ctx.add_child_to(modal_id, Frame::new().with_title("Demo Modal"))?;
236            ctx.add_child_to(frame_id, ModalContent)?;
237
238            let mut layout = Layout::fill().padding(Edges::all(1));
239            layout.min_width = Some(35);
240            layout.max_width = Some(40);
241            layout.min_height = Some(5);
242            layout.max_height = Some(7);
243            ctx.set_layout_of(frame_id, layout)?;
244            Ok(())
245        })?;
246
247        // Dim the demo content frame
248        self.with_demo_frame(c, |_frame, ctx| {
249            ctx.push_effect(ctx.node_id(), effects::dim(0.5))
250        })?;
251
252        Ok(())
253    }
254
255    /// Hide the modal overlay.
256    #[command]
257    pub fn hide_modal(&mut self, c: &mut dyn Context) -> Result<()> {
258        if !self.modal_visible {
259            return Ok(());
260        }
261        self.modal_visible = false;
262
263        self.with_right_container(c, |_, ctx| {
264            if let Some(modal_id) = ctx.child_keyed(ModalSlot::KEY) {
265                ctx.remove_subtree(modal_id)?;
266            }
267            Ok(())
268        })?;
269
270        // Re-apply user effects (clears dim, applies selected effects)
271        self.apply_effects(c)?;
272
273        Ok(())
274    }
275
276    /// Apply the selected theme from the dropdown.
277    #[command]
278    pub fn apply_theme(&mut self, c: &mut dyn Context) -> Result<()> {
279        let Some(selected_idx) =
280            c.try_with_unique_descendant::<Dropdown<ThemeOption>, _>(|dropdown, _ctx| {
281                Ok(dropdown.selected_index())
282            })?
283        else {
284            return Ok(());
285        };
286
287        if selected_idx != self.current_theme && selected_idx < self.themes.len() {
288            self.current_theme = selected_idx;
289            let theme_builder = self.themes[selected_idx].builder;
290            let new_style = theme_builder();
291            c.set_style(new_style);
292        }
293        Ok(())
294    }
295
296    /// Apply the selected effects from the selector to the demo pane.
297    #[command]
298    pub fn apply_effects(&mut self, c: &mut dyn Context) -> Result<()> {
299        let selected_indices = c
300            .try_with_unique_descendant::<Selector<EffectOption>, _>(|selector, _ctx| {
301                Ok(selector.selected_indices().to_vec())
302            })?
303            .unwrap_or_default();
304
305        self.with_demo_frame(c, |_frame, ctx| {
306            // Clear all existing effects on demo pane
307            ctx.clear_effects(ctx.node_id())?;
308
309            // Apply effects in selection order
310            let effect_list = available_effects();
311            for idx in selected_indices {
312                if let Some(effect_option) = effect_list.get(idx) {
313                    let effect = match effect_option.name {
314                        "Dim" => effects::dim(0.5),
315                        "Brighten" => effects::brighten(1.5),
316                        "Grayscale" => effects::saturation(0.0),
317                        "Invert" => effects::invert_rgb(),
318                        "Hue Shift" => effects::hue_shift(180.0),
319                        "Bold" => effects::bold(),
320                        "Italic" => effects::italic(),
321                        _ => continue,
322                    };
323                    ctx.push_effect(ctx.node_id(), effect)?;
324                }
325            }
326            Ok(())
327        })?;
328        Ok(())
329    }
330}

336impl Container {}

338impl Widget for Container {
339    fn render(&mut self, _r: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {

344        Layout::fill()

347
348impl Widget for Stylegym {
349    fn render(&mut self, _r: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {
350        Ok(())
351    }
352
353    fn layout(&self) -> Layout {
354        Layout::fill().direction(Direction::Row)
355    }
356
357    fn on_mount(&mut self, c: &mut dyn Context) -> Result<()> {
358        // Create left frame (controls) - preserve Frame's padding for border
359        let left_frame_id = c.add_keyed::<ControlsSlot>(Frame::new().with_title("Controls"))?;
360        c.set_layout_of(
361            left_frame_id,
362            Layout::column()
363                .fixed_width(32)
364                .flex_vertical(1)
365                .padding(Edges::all(1)),
366        )?;
367
368        // Create theme dropdown with its own frame - no fixed height so it can expand
369        let theme_frame_id =
370            c.add_keyed_to::<ThemeFrameSlot>(left_frame_id, Frame::new().with_title("Theme"))?;
371        c.add_keyed_to::<ThemeDropdownSlot>(theme_frame_id, Dropdown::new(available_themes()))?;
372        c.set_layout_of(
373            theme_frame_id,
374            Layout::column().flex_horizontal(1).padding(Edges::all(1)),
375        )?;
376
377        // Create effects selector with its own frame
378        let effects_frame_id =
379            c.add_keyed_to::<EffectsFrameSlot>(left_frame_id, Frame::new().with_title("Effects"))?;
380        c.add_keyed_to::<EffectsSelectorSlot>(
381            effects_frame_id,
382            Selector::new(available_effects()),
383        )?;
384        c.set_layout_of(effects_frame_id, Layout::fill().padding(Edges::all(1)))?;
385
386        // Create right container with Stack layout for modal overlay
387        let right_container_id = c.add_keyed::<RightContainerSlot>(Container)?;
388        c.set_layout_of(
389            right_container_id,
390            Layout::fill().direction(Direction::Stack),
391        )?;
392
393        // Create right frame (demo content)
394        let right_frame_id =
395            c.add_keyed_to::<DemoFrameSlot>(right_container_id, Frame::new().with_title("Demo"))?;
396        c.add_child_to(right_frame_id, DemoContent)?;
397        c.set_layout_of(right_frame_id, Layout::fill().padding(Edges::all(1)))?;
398
399        Ok(())
400    }
401}

403impl Loader for Stylegym {
404    fn load(c: &mut Canopy) -> Result<()> {
405        c.add_commands::<Self>()?;
406        c.add_commands::<DemoContent>()?;
407        c.add_commands::<ModalContent>()?;
408        c.add_commands::<Container>()?;
409        c.add_commands::<Dropdown<ThemeOption>>()?;
410        c.add_commands::<Selector<EffectOption>>()?;
411        Ok(())
412    }
413}

415/// Set up key bindings for the stylegym demo.
416pub fn setup_bindings(cnpy: &mut Canopy) -> Result<()> {
417    Binder::new(cnpy)
418        .defaults::<Root>()
419        .with_path("stylegym/")
420        .key('q', "root::quit()")
421        .key(key::KeyCode::Tab, "root::focus_next()")
422        .key(key::KeyCode::BackTab, "root::focus_prev()")
423        .key('m', "stylegym::show_modal()")
424        .key(key::KeyCode::Esc, "stylegym::hide_modal()")
425        // Global j/k for navigation between focusable items
426        .key('j', "root::focus_next()")
427        .key('k', "root::focus_prev()")
428        .with_path("dropdown")
429        .key(
430            key::KeyCode::Enter,
431            "dropdown::confirm(); stylegym::apply_theme()",
432        )
433        .key(' ', "dropdown::toggle()")
434        .key(key::KeyCode::Down, "dropdown::select_next()")
435        .key(key::KeyCode::Up, "dropdown::select_prev()")
436        // Mouse click on dropdown triggers theme application
437        .mouse(
438            mouse::Button::Left + mouse::Action::Down,
439            "stylegym::apply_theme()",
440        )
441        .with_path("selector")
442        .key(' ', "selector::toggle(); stylegym::apply_effects()")
443        .key(
444            key::KeyCode::Enter,
445            "selector::toggle(); stylegym::apply_effects()",
446        )
447        .key(key::KeyCode::Down, "selector::select_next()")
448        .key(key::KeyCode::Up, "selector::select_prev()")
449        // Mouse click on selector triggers effect application
450        .mouse(
451            mouse::Button::Left + mouse::Action::Down,
452            "stylegym::apply_effects()",
453        );
454    Ok(())
