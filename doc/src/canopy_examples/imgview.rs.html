10/// Configure key bindings for the image viewer.
11pub fn setup_bindings(cnpy: &mut Canopy) {
12    Binder::new(cnpy)
13        .key('q', "root::quit()")
14        .with_path("image_view/")
15        .key('i', "image_view::zoom_in()")
16        .key('o', "image_view::zoom_out()")
17        .key('h', "image_view::pan_left()")
18        .key('j', "image_view::pan_down()")
19        .key('k', "image_view::pan_up()")
20        .key('l', "image_view::pan_right()")
21        .key(key::KeyCode::Left, "image_view::pan_left()")
22        .key(key::KeyCode::Right, "image_view::pan_right()")
23        .key(key::KeyCode::Up, "image_view::pan_up()")
24        .key(key::KeyCode::Down, "image_view::pan_down()");
25}
26
27/// Create a Canopy application for viewing the specified image.
28pub fn create_app(image_path: &Path) -> Result<Canopy> {
29    let mut cnpy = Canopy::new();
30
31    Root::load(&mut cnpy);
32    ImageView::load(&mut cnpy);
33    setup_bindings(&mut cnpy);
34
35    let view = ImageView::from_path(image_path)?;
36    let app_id = cnpy.core.create_detached(view);
37    Root::install(&mut cnpy.core, app_id)?;
38    Ok(cnpy)
39}
