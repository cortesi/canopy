26/// Text block widget for the list gym demo.
27pub struct TextBlock {
28    /// Color layer name.
29    color: String,
30    /// Maximum line width (for horizontal scrolling).
31    max_line_width: u32,
32    /// Cached wrapped lines.
33    lines: Vec<String>,
34    /// Selection state.
35    selected: bool,
36}
37
38impl Selectable for TextBlock {
39    fn set_selected(&mut self, selected: bool) {
40        self.selected = selected;
41    }
42}
43
44#[derive_commands]
45impl TextBlock {
46    /// Construct a text block for the given index.
47    pub fn new(index: usize) -> Self {
48        let mut rng = rand::rng();
49        let wrap_width = rng.random_range(10..150);
50        let lines = Self::wrap_lines(TEXT, wrap_width);
51        let max_line_width = lines.iter().map(|l| l.len()).max().unwrap_or(0) as u32;
52        Self {
53            color: String::from(COLORS[index % 2]),
54            max_line_width,
55            lines,
56            selected: false,
57        }
58    }
59
60    /// Wrap text to the given width.
61    fn wrap_lines(text: &str, width: u32) -> Vec<String> {
62        let wrap_width = width.max(1) as usize;
63        textwrap::wrap(text, wrap_width)
64            .into_iter()
65            .map(|line| line.to_string())
66            .collect()
67    }
68}

70impl Widget for TextBlock {
71    fn layout(&self) -> Layout {
72        // Use intrinsic width for horizontal scrolling support.
73        // Items size to their content width, allowing parent scroll to reveal off-screen text.
74        Layout::column()
75    }
76
77    fn measure(&self, c: MeasureConstraints) -> Measurement {
78        let intrinsic_height = self.lines.len().max(1) as u32;
79        // Use intrinsic width for horizontal scrolling (bypass width constraint)
80        let width = self.max_line_width.saturating_add(2);
81        // Clamp height to constraint
82        let height = match c.height {
83            Constraint::Unbounded => intrinsic_height,
84            Constraint::AtMost(n) => intrinsic_height.min(n),
85            Constraint::Exact(n) => n,
86        };
87        Measurement::Fixed(Size::new(width, height))
88    }
89
90    fn canvas(&self, _view: Size<u32>, _ctx: &CanvasContext) -> Size<u32> {
91        // Canvas matches measurement for consistent sizing
92        let height = self.lines.len().max(1) as u32;
93        Size::new(self.max_line_width.saturating_add(2), height)
94    }
95
96    fn render(&mut self, rndr: &mut Render, ctx: &dyn ViewContext) -> Result<()> {
97        let view = ctx.view();
98        let area = view.view_rect_local();
99
100        if area.w == 0 || area.h == 0 {
101            return Ok(());
102        }
103
104        // Column 0: Selection indicator (when selected)
105        if self.selected {
106            let indicator_rect = Rect::new(0, 0, 1, self.lines.len().max(1) as u32);
107            rndr.fill("list/selected", indicator_rect, '\u{2588}')?;
108        }
109
110        // Column 1: Spacer
111        let spacer = Rect::new(1, 0, 1, self.lines.len().max(1) as u32);
112        rndr.fill("", spacer, ' ')?;
113
114        // Text content starts at column 2
115        // Render full content at local coordinates; the clip handles visibility
116        let style = format!("{}/text", self.color);
117
118        for (idx, line) in self.lines.iter().enumerate() {
119            if !line.is_empty() {
120                let line_rect = Rect::new(2, idx as u32, line.len() as u32, 1);
121                rndr.text(&style, line_rect.line(0), line)?;
122            }
123        }
124
125        Ok(())
126    }
127
128    fn accept_focus(&self, _ctx: &dyn ViewContext) -> bool {
129        true
130    }
131
132    fn name(&self) -> NodeName {
133        NodeName::convert("text_block")
134    }
135}

137/// Status bar widget for the list gym demo.
138pub struct StatusBar;
139
140#[derive_commands]
141impl StatusBar {
142    /// Build the status text based on the focused column.
143    fn label(&self, ctx: &dyn ViewContext) -> String {
144        let columns = Self::column_nodes(ctx);
145        let total = columns.len();
146        let focused = columns
147            .iter()
148            .position(|node| ctx.node_is_on_focus_path(*node));
149
150        match (focused, total) {
151            (Some(idx), total) if total > 0 => {
152                format!("listgym  col {}/{}", idx + 1, total)
153            }
154            _ => "listgym".to_string(),
155        }
156    }
157
158    /// Collect column container node ids from the panes widget.
159    fn column_nodes(ctx: &dyn ViewContext) -> Vec<NodeId> {
160        let root_children = ctx.children_of(ctx.root_id());
161        let Some(panes_id) = root_children.first().copied() else {
162            return Vec::new();
163        };
164        ctx.children_of(panes_id)
165    }
166}
167
168impl Widget for StatusBar {
169    fn render(&mut self, r: &mut Render, ctx: &dyn ViewContext) -> Result<()> {
170        r.push_layer("statusbar");
171        let label = self.label(ctx);
172        r.text("text", ctx.view().outer_rect_local().line(0), &label)?;
173        Ok(())
174    }
175}

193    /// Ensure the list, panes, and status bar are created.
194    fn ensure_tree(&self, c: &mut dyn Context) -> Result<()> {
195        if !c.children().is_empty() {
196            return Ok(());
197        }
198
199        let panes_id = c.add_child(Panes::new()).expect("Failed to mount panes");
200        let status_id = c.add_child(StatusBar).expect("Failed to mount statusbar");
201
202        c.with_layout(&mut |layout| {
203            *layout = Layout::column().flex_horizontal(1).flex_vertical(1);
204        })
205        .expect("Failed to configure layout");
206        c.with_layout_of(panes_id, &mut |layout| {
207            *layout = Layout::fill();
208        })
209        .expect("Failed to configure panes layout");
210        c.with_layout_of(status_id, &mut |layout| {
211            *layout = Layout::row().flex_horizontal(1).fixed_height(1);
212        })
213        .expect("Failed to configure status layout");
214
215        let (frame_id, list_id) = Self::create_column(c)?;
216        c.with_widget(panes_id, |panes: &mut Panes, ctx| {
217            panes.insert_col(ctx, frame_id)
218        })?;
219        // Focus the first focusable item in the list (List itself doesn't accept focus)
220        c.focus_first_in(list_id);
221
222        Ok(())
223    }
224
225    /// Create a framed list column and return (frame, list) node ids.
226    fn create_column(c: &mut dyn Context) -> Result<(NodeId, NodeId)> {
227        let list_id = c.add_orphan(List::<TextBlock>::new());
228        let frame_id = c.add_orphan(frame::Frame::new());
229        c.mount_child_to(frame_id, list_id)?;
230        c.with_layout_of(list_id, &mut |layout| {
231            *layout = Layout::fill().overflow_x();
232        })?;
233        // Add initial items
234        c.with_widget(list_id, |list: &mut List<TextBlock>, ctx| {
235            for i in 0..10 {
236                list.append(ctx, TextBlock::new(i))?;
237            }
238            Ok(())
239        })?;
240        Ok((frame_id, list_id))
241    }
242
243    /// Execute a closure with mutable access to the list widget.
244    fn with_list<F, R>(&self, c: &mut dyn Context, mut f: F) -> Result<R>
245    where
246        F: FnMut(&mut List<TextBlock>, &mut dyn Context) -> Result<R>,
247    {
248        self.ensure_tree(c)?;
249        let list_id = Self::list_id(c)?;
250        c.with_widget(list_id, |list: &mut List<TextBlock>, ctx| f(list, ctx))
251    }
252
253    /// Panes node id, if initialized.
254    fn panes_id(c: &dyn Context) -> Option<NodeId> {
255        c.children().first().copied()
256    }
257
258    /// Find the list to target for list commands.
259    fn list_id(c: &dyn Context) -> Result<NodeId> {
260        let panes_id =
261            Self::panes_id(c).ok_or_else(|| Error::Invalid("list not initialized".into()))?;
262        let lists = Self::column_lists(c, panes_id);
263        if lists.is_empty() {
264            return Err(Error::Invalid("list not initialized".into()));
265        }
266        if let Some(id) = lists
267            .iter()
268            .copied()
269            .find(|id| c.node_is_on_focus_path(*id))
270        {
271            return Ok(id);
272        }
273        Ok(lists[0])
274    }
275
276    /// Move focus between columns.
277    fn shift_column(&self, c: &mut dyn Context, forward: bool) -> Result<()> {
278        self.ensure_tree(c)?;
279        let panes_id = Self::panes_id(c).expect("panes not initialized");
280        let columns = Self::column_lists(c, panes_id);
281        if columns.is_empty() {
282            return Ok(());
283        }
284
285        let current = columns.iter().position(|id| c.node_is_on_focus_path(*id));
286        let next_idx = match (current, forward) {
287            (Some(idx), true) => (idx + 1) % columns.len(),
288            (Some(idx), false) => (idx + columns.len() - 1) % columns.len(),
289            (None, true) => 0,
290            (None, false) => columns.len() - 1,
291        };
292        c.set_focus(columns[next_idx]);
293        Ok(())
294    }
295
296    /// Collect the list nodes for each column, in column order.
297    fn column_lists(c: &dyn Context, panes_id: NodeId) -> Vec<NodeId> {
298        let mut lists = Vec::new();
299        for column_id in c.children_of(panes_id) {
300            let Some(frame_id) = c.children_of(column_id).first().copied() else {
301                continue;
302            };
303            let Some(list_id) = c.children_of(frame_id).first().copied() else {
304                continue;
305            };
306            lists.push(list_id);
307        }
308        lists
309    }
