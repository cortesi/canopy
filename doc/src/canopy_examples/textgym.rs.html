118/// Wrap a text widget in a titled frame.
119fn section(c: &mut dyn Context, title: &str, text: Text, layout: Layout) -> Result<NodeId> {
120    let text_id = c.create_detached(text);
121    c.set_layout_of(text_id, Layout::fill())?;
122    let frame_id = Frame::wrap_with(c, text_id, Frame::new().with_title(title))?;
123    let pad_id = Pad::wrap_with(c, frame_id, Pad::uniform(OUTER_PADDING))?;
124    let padded_layout = outer_layout(layout, OUTER_PADDING);
125    c.set_layout_of(pad_id, padded_layout)?;
126    Ok(pad_id)
127}
128
129/// Convert a frame layout into a padded container layout.
130fn outer_layout(layout: Layout, padding: u32) -> Layout {
131    let mut layout = layout.padding(Edges::all(padding));
132    let bump = padding.saturating_mul(2);
133    if let Some(min_width) = layout.min_width {
134        layout.min_width = Some(min_width.saturating_add(bump));
135    }
136    if let Some(max_width) = layout.max_width {
137        layout.max_width = Some(max_width.saturating_add(bump));
138    }
139    if let Some(min_height) = layout.min_height {
140        layout.min_height = Some(min_height.saturating_add(bump));
141    }
142    if let Some(max_height) = layout.max_height {
143        layout.max_height = Some(max_height.saturating_add(bump));
144    }
145    layout
146}
