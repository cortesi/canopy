13/// Simple editor wrapper for the cedit demo.
14pub struct Ed {
15    /// Initial contents for the editor.
16    contents: String,
17    /// File extension hint for syntax highlighting.
18    extension: String,
19}
20
21#[derive_commands]
22impl Ed {
23    /// Construct an editor with initial contents.
24    pub fn new(contents: &str, extension: &str) -> Self {
25        Self {
26            contents: contents.to_string(),
27            extension: extension.to_string(),
28        }
29    }
30}
31
32impl Widget for Ed {
33    fn on_mount(&mut self, c: &mut dyn Context) -> Result<()> {
34        let config = EditorConfig::new()
35            .with_mode(EditMode::Vi)
36            .with_wrap(WrapMode::None);
37        let mut editor = Editor::with_config(&self.contents, config);
38        editor.set_highlighter(Some(Box::new(SyntectHighlighter::new(
39            self.extension.as_str(),
40        ))));
41        let frame_id = c.add_child(Frame::new())?;
42        let editor_id = c.add_child_to(frame_id, editor)?;
43
44        c.set_layout_of(editor_id, Layout::fill().padding(Edges::all(1)))?;
45        c.set_layout_of(frame_id, Layout::fill().padding(Edges::all(1)))?;
46
47        c.set_layout(Layout::fill())?;
48        Ok(())
49    }
50
51    fn render(&mut self, _r: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {
52        Ok(())
53    }
54}
55
56impl Loader for Ed {
57    fn load(c: &mut Canopy) {
58        c.add_commands::<Self>();
59        c.add_commands::<Editor>();
60    }
61}
62
63/// Install key bindings for the cedit demo.
64pub fn setup_bindings(cnpy: &mut Canopy) {
65    Binder::new(cnpy)
66        .defaults::<Root>()
67        .with_path("ed/")
68        .key(key::KeyCode::Tab, "root::focus_next()")
69        .key('p', "print(\"cedit\")");
70}
