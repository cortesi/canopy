12/// A widget that renders a test pattern.
13#[derive(canopy::StatefulNode)]
14pub struct TestPattern {
15    /// Node state.
16    state: NodeState,
17    /// Virtual canvas size.
18    size: Expanse,
19}
20
21impl Default for TestPattern {
22    fn default() -> Self {
23        Self::new()
24    }
25}
26
27#[derive_commands]
28impl TestPattern {
29    /// Construct the test pattern node.
30    pub fn new() -> Self {
31        Self {
32            state: NodeState::default(),
33            size: Expanse::new(500, 500),
34        }
35    }
36
37    #[command]
38    /// Scroll to the top-left corner.
39    pub fn scroll_to_top(&mut self, c: &mut dyn Context) {
40        c.scroll_to(self, 0, 0);
41    }
42
43    #[command]
44    /// Scroll down by one line.
45    pub fn scroll_down(&mut self, c: &mut dyn Context) {
46        c.scroll_down(self);
47    }
48
49    #[command]
50    /// Scroll up by one line.
51    pub fn scroll_up(&mut self, c: &mut dyn Context) {
52        c.scroll_up(self);
53    }
54
55    #[command]
56    /// Scroll left by one column.
57    pub fn scroll_left(&mut self, c: &mut dyn Context) {
58        c.scroll_left(self);
59    }
60
61    #[command]
62    /// Scroll right by one column.
63    pub fn scroll_right(&mut self, c: &mut dyn Context) {
64        c.scroll_right(self);
65    }
66
67    #[command]
68    /// Page down in the viewport.
69    pub fn page_down(&mut self, c: &mut dyn Context) {
70        c.page_down(self);
71    }
72
73    #[command]
74    /// Page up in the viewport.
75    pub fn page_up(&mut self, c: &mut dyn Context) {
76        c.page_up(self);
77    }
78
79    /// Return the character for the test pattern at a position.
80    fn generate_pattern_char(x: u32, y: u32) -> char {
81        // Pattern: "abcdefghijklmnopqrstuvwxyz0123456789"
82        let pattern = "abcdefghijklmnopqrstuvwxyz0123456789";
83        let pattern_len = pattern.len() as u32;
84
85        // Offset each row by one more character than the previous
86        let index = ((x + y) % pattern_len) as usize;
87        pattern.chars().nth(index).unwrap_or(' ')
88    }
89}
90
91impl Node for TestPattern {
92    fn accept_focus(&mut self) -> bool {
93        true
94    }
95
96    fn layout(&mut self, _l: &Layout, sz: Expanse) -> Result<()> {
97        let canvas_size = self.size;
98        self.fit_size(canvas_size, sz);
99        Ok(())
100    }
101
102    fn render(&mut self, _c: &dyn Context, r: &mut Render) -> Result<()> {
103        let vp = self.vp();
104        let view = vp.view();
105
106        // The viewport automatically handles the visible window for us
107        // We just need to render the content that's visible
108        for y in 0..view.h {
109            let absolute_y = view.tl.y + y;
110            if absolute_y >= self.size.h {
111                break;
112            }
113
114            let mut line = String::new();
115            for x in 0..view.w {
116                let absolute_x = view.tl.x + x;
117                if absolute_x >= self.size.w {
118                    break;
119                }
120                let ch = Self::generate_pattern_char(absolute_x, absolute_y);
121                line.push(ch);
122            }
123
124            // Use different colors to make the pattern more visible
125            let color = match (absolute_y / 10) % 3 {
126                0 => "blue",
127                1 => "green",
128                _ => "yellow",
129            };
130
131            r.text(color, view.line(y), &line)?;
132        }
133
134        Ok(())
135    }
136}

138#[derive(canopy::StatefulNode)]
139/// Root node for the frame gym demo.
140pub struct FrameGym {
141    /// Node state.
142    state: NodeState,
143    /// Framed test pattern.
144    child: frame::Frame<TestPattern>,
145}
146
147#[derive_commands]
148impl Default for FrameGym {
149    fn default() -> Self {
150        Self::new()
151    }
152}
153
154impl FrameGym {
155    /// Construct a new frame gym.
156    pub fn new() -> Self {
157        Self {
158            state: NodeState::default(),
159            child: frame::Frame::new(TestPattern::new()).with_title("Frame Gym".to_string()),
160        }
161    }
162}
163
164impl Node for FrameGym {
165    fn accept_focus(&mut self) -> bool {
166        false // Don't accept focus, let it go to the child
167    }
168
169    fn layout(&mut self, l: &Layout, sz: Expanse) -> Result<()> {
170        self.child.layout(l, sz)?;
171        self.wrap(self.child.vp())?;
172        Ok(())
173    }
174
175    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
176        f(&mut self.child)?;
177        Ok(())
178    }
179}
180
181impl Loader for FrameGym {
182    fn load(c: &mut Canopy) {
183        c.add_commands::<Self>();
184        c.add_commands::<TestPattern>();
185    }
186}

188/// Install key bindings for the frame gym demo.
189pub fn setup_bindings(cnpy: &mut Canopy) {
190    Binder::new(cnpy)
191        .defaults::<Root<FrameGym>>()
192        .with_path("")
193        // Focus navigation
194        .key(key::KeyCode::Tab, "root::focus_next()")
195        // Arrow keys for scrolling
196        .key('g', "test_pattern::scroll_to_top()")
197        .key(key::KeyCode::Down, "test_pattern::scroll_down()")
198        .key(key::KeyCode::Up, "test_pattern::scroll_up()")
199        .key(key::KeyCode::Left, "test_pattern::scroll_left()")
200        .key(key::KeyCode::Right, "test_pattern::scroll_right()")
201        // Vim-style navigation
202        .key('j', "test_pattern::scroll_down()")
203        .key('k', "test_pattern::scroll_up()")
204        .key('h', "test_pattern::scroll_left()")
205        .key('l', "test_pattern::scroll_right()")
206        // Page navigation
207        .key(key::KeyCode::PageDown, "test_pattern::page_down()")
208        .key(key::KeyCode::PageUp, "test_pattern::page_up()")
209        .key(' ', "test_pattern::page_down()")
210        // Quit
211        .with_path("root")
212        .key('q', "root::quit()");
213}
