3/// A widget that renders a test pattern
4#[derive(StatefulNode)]
5pub struct TestPattern {

55    fn generate_pattern_char(x: u32, y: u32) -> char {
56        // Pattern: "abcdefghijklmnopqrstuvwxyz0123456789"
57        let pattern = "abcdefghijklmnopqrstuvwxyz0123456789";
58        let pattern_len = pattern.len() as u32;
59
60        // Offset each row by one more character than the previous
61        let index = ((x + y) % pattern_len) as usize;
62        pattern.chars().nth(index).unwrap_or(' ')
63    }
64}
65
66impl Node for TestPattern {
67    fn accept_focus(&mut self) -> bool {
68        true
69    }
70
71    fn layout(&mut self, l: &Layout, sz: Expanse) -> Result<()> {
72        let canvas_size = self.size;
73        l.size(self, canvas_size, sz)?;
74        Ok(())
75    }
76
77    fn render(&mut self, _c: &dyn Context, r: &mut Render) -> Result<()> {
78        let vp = self.vp();
79        let view = vp.view();
80
81        // The viewport automatically handles the visible window for us
82        // We just need to render the content that's visible
83        for y in 0..view.h {
84            let absolute_y = view.tl.y + y;
85            if absolute_y >= self.size.h {
86                break;
87            }
88
89            let mut line = String::new();
90            for x in 0..view.w {
91                let absolute_x = view.tl.x + x;
92                if absolute_x >= self.size.w {
93                    break;
94                }
95                let ch = Self::generate_pattern_char(absolute_x, absolute_y);
96                line.push(ch);
97            }
98
99            // Use different colors to make the pattern more visible
100            let color = match (absolute_y / 10) % 3 {
101                0 => "blue",
102                1 => "green",
103                _ => "yellow",
104            };
105
106            r.text(color, view.line(y), &line)?;
107        }
108
109        Ok(())
110    }
111}

113#[derive(StatefulNode)]
114pub struct FrameGym {
115    state: NodeState,
116    child: frame::Frame<TestPattern>,
117}
118
119#[derive_commands]
120impl Default for FrameGym {
121    fn default() -> Self {
122        Self::new()
123    }
124}
125
126impl FrameGym {
127    pub fn new() -> Self {
128        FrameGym {
129            state: NodeState::default(),
130            child: frame::Frame::new(TestPattern::new()).with_title("Frame Gym".to_string()),
131        }
132    }
133}
134
135impl Node for FrameGym {
136    fn accept_focus(&mut self) -> bool {
137        false // Don't accept focus, let it go to the child
138    }
139
140    fn layout(&mut self, l: &Layout, sz: Expanse) -> Result<()> {
141        self.child.layout(l, sz)?;
142        self.wrap(self.child.vp())?;
143        Ok(())
144    }
145
146    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
147        f(&mut self.child)?;
148        Ok(())
149    }
150}
151
152impl Loader for FrameGym {

154        c.add_commands::<FrameGym>();
155        c.add_commands::<TestPattern>();
156    }
157}

159pub fn setup_bindings(cnpy: &mut Canopy) {
160    Binder::new(cnpy)
161        .defaults::<Root<FrameGym>>()
162        .with_path("")
163        // Focus navigation
164        .key(key::KeyCode::Tab, "root::focus_next()")
165        // Arrow keys for scrolling
166        .key(key::KeyCode::Down, "test_pattern::scroll_down()")
167        .key(key::KeyCode::Up, "test_pattern::scroll_up()")
168        .key(key::KeyCode::Left, "test_pattern::scroll_left()")
169        .key(key::KeyCode::Right, "test_pattern::scroll_right()")
170        // Vim-style navigation
171        .key('j', "test_pattern::scroll_down()")
172        .key('k', "test_pattern::scroll_up()")
173        .key('h', "test_pattern::scroll_left()")
174        .key('l', "test_pattern::scroll_right()")
175        // Page navigation
176        .key(key::KeyCode::PageDown, "test_pattern::page_down()")
177        .key(key::KeyCode::PageUp, "test_pattern::page_up()")
178        .key(' ', "test_pattern::page_down()")
179        // Quit
180        .with_path("root")
181        .key('q', "root::quit()");
182}
