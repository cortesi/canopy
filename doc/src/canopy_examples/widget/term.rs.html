15/// Tab labels shown in the demo.
16const TAB_LABELS: [&str; 3] = ["claude", "codex", "gemini"];
17/// Height of each tab widget.
18const TAB_HEIGHT: u32 = 3;
19/// Button style prefix for tab widgets.
20const TAB_STYLE_PREFIX: &str = "term_demo/button";

73/// Terminal demo widget with three commands.
74pub struct TermDemo {
75    /// Current active index.
76    active: usize,
77    /// Tab button node ids.
78    tab_ids: Vec<TypedId<Button>>,
79    /// Terminal frame node ids.
80    frame_ids: Vec<NodeId>,
81    /// Terminal node ids.
82    terminal_ids: Vec<NodeId>,
83}
84
85impl Default for TermDemo {
86    fn default() -> Self {
87        Self::new()
88    }
89}
90
91#[derive_commands]
92impl TermDemo {
93    /// Construct a terminal demo.
94    pub fn new() -> Self {
95        Self {
96            active: 0,
97            tab_ids: Vec::new(),
98            frame_ids: Vec::new(),
99            terminal_ids: Vec::new(),
100        }
101    }
102
103    /// Activate a tab/terminal index.
104    fn set_active(&mut self, ctx: &mut dyn Context, index: usize) -> Result<()> {
105        if self.terminal_ids.is_empty() || self.tab_ids.is_empty() || self.frame_ids.is_empty() {
106            return Ok(());
107        }
108
109        let target = index.min(self.terminal_ids.len() - 1);
110        self.active = target;
111
112        for (idx, tab_id) in self.tab_ids.iter().enumerate() {
113            let active = idx == self.active;
114            ctx.with_typed(*tab_id, |tab: &mut Button, _ctx| {
115                tab.set_active(active);
116                Ok(())
117            })?;
118        }
119
120        for (idx, frame_id) in self.frame_ids.iter().enumerate() {
121            let active = idx == self.active;
122            ctx.with_layout_of(*frame_id, &mut |layout| {
123                *layout = if active {
124                    Layout::fill().padding(Edges::all(1))
125                } else {
126                    Layout::fill().none()
127                };
128            })?;
129        }
130
131        if let Some(active_id) = self.terminal_ids.get(self.active).copied() {
132            ctx.set_focus(active_id);
133        }
134
135        Ok(())
136    }
137
138    /// Cycle to the next tab.
139    #[command]
140    pub fn next_tab(&mut self, ctx: &mut dyn Context) -> Result<()> {
141        if self.terminal_ids.is_empty() {
142            return Ok(());
143        }
144        let next = (self.active + 1) % self.terminal_ids.len();
145        self.set_active(ctx, next)
146    }
147}
148
149impl Widget for TermDemo {
150    fn layout(&self) -> Layout {
151        Layout::fill().direction(Direction::Column)
152    }
153
154    fn on_mount(&mut self, ctx: &mut dyn Context) -> Result<()> {
155        let mut style = StyleMap::new();
156        style
157            .rules()
158            .prefix(TAB_STYLE_PREFIX)
159            .fg("active/border", Paint::solid(Color::rgb("#FF8C00")))
160            .fg("inactive/border", Paint::solid(Color::rgb("#6B6B6B")))
161            .fg("active/text", Paint::solid(Color::White))
162            .fg("inactive/text", Paint::solid(Color::White))
163            .apply();
164        ctx.set_style(style);
165
166        let tab_bar_id = ctx.add_child(TabBar::new())?;
167        for label in TAB_LABELS {
168            let tab_id = ctx.add_child_to(
169                tab_bar_id,
170                Button::new(label.to_string()).with_glyphs(ROUND),
171            )?;
172            ctx.set_layout_of(
173                tab_id,
174                Layout::fill().fixed_height(TAB_HEIGHT).flex_horizontal(1),
175            )?;
176            self.tab_ids.push(tab_id);
177        }
178
179        let stack_id = ctx.add_child(TerminalStack::new())?;
180
181        let cwd = env::current_dir().map_err(|err| Error::Internal(err.to_string()))?;
182        for label in TAB_LABELS {
183            let frame_id = ctx.add_child_to(stack_id, Frame::new())?;
184            ctx.set_layout_of(frame_id, Layout::fill().padding(Edges::all(1)))?;
185            let terminal_id = ctx.add_child_to(
186                frame_id,
187                Terminal::new(TerminalConfig {
188                    command: Some(vec![label.to_string()]),
189                    cwd: Some(cwd.clone()),
190                    ..TerminalConfig::default()
191                }),
192            )?;
193            ctx.set_layout_of(terminal_id, Layout::fill())?;
194            self.frame_ids.push(frame_id.into());
195            self.terminal_ids.push(terminal_id.into());
196        }
197
198        self.set_active(ctx, 0)?;
199        Ok(())
200    }
201
202    fn render(&mut self, rndr: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {
203        rndr.push_layer("term_demo");
204        Ok(())
205    }
206
207    fn name(&self) -> NodeName {
208        NodeName::convert("term_demo")
209    }
210}
