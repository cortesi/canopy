16/// Seed text for the single-line editor sample.

30/// Repeat the provided seed text a given number of times.
31fn repeated_line(seed: &str, repeats: usize) -> String {
32    seed.repeat(repeats.max(1))
33}
34
35/// Build a wrapped paragraph sample.
36fn wrap_text() -> String {
37    format!("{PARAGRAPH}\n\n{PARAGRAPH}")
38}
39
40/// Build a multi-line sample without wrapping.
41fn no_wrap_text() -> String {
42    let line = repeated_line(LONG_LINE_SEED, 2);
43    format!("{line}\n{line}\nShort line for contrast.")
44}

46/// Generate numbered sample lines.
47fn numbered_lines(prefix: &str, count: usize) -> String {
48    let mut lines = Vec::with_capacity(count);
49    for idx in 1..=count {
50        lines.push(format!("{prefix} {idx:02}: sample text for scrolling"));
51    }
52    lines.join("\n")
53}

55/// Create a framed editor node and return the frame node id.
56fn add_editor_frame(
57    c: &mut dyn Context,
58    title: &str,
59    text: impl Into<String>,
60    config: EditorConfig,
61    height: Option<u32>,
62    highlighter: Option<SyntectHighlighter>,
63) -> Result<NodeId> {
64    let mut editor = Editor::with_config(text, config);
65    if let Some(highlighter) = highlighter {
66        editor.set_highlighter(Some(Box::new(highlighter)));
67    }
68    let frame_id = c.create_detached(Frame::new().with_title(title));
69    let editor_id = c.add_child_to(frame_id, editor)?;
70
71    c.with_layout_of(editor_id, &mut |layout| {
72        *layout = Layout::fill();
73    })?;
74
75    let mut frame_layout = Layout::column().padding(Edges::all(1)).flex_horizontal(1);
76    if let Some(height) = height {
77        frame_layout = frame_layout.fixed_height(height);
78    }
79    c.with_layout_of(frame_id, &mut |layout| {
80        *layout = frame_layout;
81    })?;
82
83    Ok(frame_id)
84}

119    /// Scroll the outer pane by one line in the specified direction.
120    pub fn scroll(&mut self, c: &mut dyn Context, dir: ScrollDirection) {
121        match dir {
122            ScrollDirection::Up => c.scroll_up(),
123            ScrollDirection::Down => c.scroll_down(),
124            ScrollDirection::Left => c.scroll_left(),
125            ScrollDirection::Right => c.scroll_right(),
126        };
127    }
128
129    /// Page in the outer pane.
130    pub fn page(&mut self, c: &mut dyn Context, dir: VerticalDirection) {
131        match dir {
132            VerticalDirection::Up => c.page_up(),
133            VerticalDirection::Down => c.page_down(),
134        };
135    }
136

138    /// Scroll the outer pane up by one line.
139    pub fn scroll_up(&mut self, c: &mut dyn Context) {
140        self.scroll(c, ScrollDirection::Up);

143    #[command]
144    /// Scroll the outer pane down by one line.

179    /// Build the left column of editor samples.
180    fn build_left_column(&self, c: &mut dyn Context) -> Result<NodeId> {
181        let single_line = repeated_line(SINGLE_LINE_SEED, 2);
182        let soft_wrap = wrap_text();
183        let no_wrap = no_wrap_text();
184        let line_numbers = numbered_lines("Line", 30);
185        let auto_grow = numbered_lines("Auto", 4);
186
187        let frames = vec![
188            add_editor_frame(
189                c,
190                "Single line (text)",
191                single_line,
192                EditorConfig::new()
193                    .with_multiline(false)
194                    .with_wrap(WrapMode::None),
195                Some(3),
196                None,
197            )?,
198            add_editor_frame(
199                c,
200                "Soft wrap (multiline)",
201                soft_wrap,
202                EditorConfig::new().with_wrap(WrapMode::Soft),
203                Some(8),
204                None,
205            )?,
206            add_editor_frame(
207                c,
208                "No wrap (multiline)",
209                no_wrap,
210                EditorConfig::new().with_wrap(WrapMode::None),
211                Some(6),
212                None,
213            )?,
214            add_editor_frame(
215                c,
216                "Line numbers (absolute)",
217                line_numbers,
218                EditorConfig::new().with_line_numbers(LineNumbers::Absolute),
219                Some(8),
220                None,
221            )?,
222            add_editor_frame(
223                c,
224                "Auto grow (min 2 max 6)",
225                auto_grow,
226                EditorConfig::new()
227                    .with_auto_grow(true)
228                    .with_min_height(2)
229                    .with_max_height(Some(6)),
230                None,
231                None,
232            )?,
233            add_editor_frame(
234                c,
235                "Read only",
236                READ_ONLY_TEXT,
237                EditorConfig::new().with_read_only(true),
238                Some(6),
239                None,
240            )?,
241        ];
242
243        let column_id = c.create_detached(EditorColumn::new());
244        c.set_children_of(column_id, frames)?;
245        Ok(column_id)

307impl Widget for EditorGym {
308    fn layout(&self) -> Layout {
309        Layout::row()
310            .flex_horizontal(1)
311            .flex_vertical(1)
312            .gap(1)
313            .overflow_y()
314    }
315
316    fn canvas(&self, view: Size<u32>, ctx: &CanvasContext) -> Size<u32> {
317        let extent = ctx.children_extent();
318        Size::new(view.width.max(extent.width), view.height.max(extent.height))
319    }
320
321    fn on_mount(&mut self, c: &mut dyn Context) -> Result<()> {
322        let left = self.build_left_column(c)?;
323        let right = self.build_right_column(c)?;
324        c.set_children(vec![left, right])?;
325        Ok(())
326    }
327}

337/// Install key bindings for the editor gym demo.
338pub fn setup_bindings(cnpy: &mut Canopy) {
339    Binder::new(cnpy)
340        .defaults::<Root>()
341        .with_path("editor_gym")
342        .key(key::KeyCode::Tab, "root::focus_next()")
343        .key(key::KeyCode::BackTab, "root::focus_prev()")
344        .key(key::KeyCode::PageDown, "editor_gym::page_down()")
345        .key(key::KeyCode::PageUp, "editor_gym::page_up()")
346        .key(key::KeyCode::Home, "editor_gym::scroll_to(0, 0)")
347        .mouse(mouse::Action::ScrollDown, "editor_gym::scroll_down()")
348        .mouse(mouse::Action::ScrollUp, "editor_gym::scroll_up()")
349        .with_path("root")
350        .key('q', "root::quit()");
351}
