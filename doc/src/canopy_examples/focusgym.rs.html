8/// A focusable block that can split into children.
9pub struct Block {
10    /// True for horizontal layout.
11    horizontal: bool,
12}
13
14#[derive_commands]
15impl Block {
16    /// Construct a block with the requested orientation.
17    fn new(horizontal: bool) -> Self {
18        Self { horizontal }
19    }
20
21    /// Return true when the available area is too small to split.
22    fn size_limited(&self, a: Expanse) -> bool {
23        (self.horizontal && a.w <= 4) || (!self.horizontal && a.h <= 4)
24    }
25
26    /// Adjust flex factors by the requested deltas and apply the updated layout.
27    fn adjust_flex(&self, c: &mut dyn Context, delta: i32) -> Result<()> {
28        if let Some(view) = c.node_view(c.node_id())
29            && (view.outer.w <= 1 || view.outer.h <= 1)
30            && delta < 0
31        {
32            return Ok(());
33        }
34
35        let parent_dir = if let Some(parent) = c.parent_of(c.node_id()) {
36            let mut dir = None;
37            c.with_layout_of(parent, &mut |layout| {
38                dir = Some(layout.direction);
39            })?;
40            dir
41        } else {
42            None
43        };
44
45        let adjust_horizontal = match parent_dir {
46            Some(Direction::Row) => true,
47            Some(Direction::Column) => false,
48            Some(Direction::Stack) | None => self.horizontal,
49        };
50
51        let layout = c.layout();
52        let weight = if adjust_horizontal {
53            match layout.width {
54                Sizing::Flex(w) => w,
55                _ => 1,
56            }
57        } else {
58            match layout.height {
59                Sizing::Flex(w) => w,
60                _ => 1,
61            }
62        };
63        let next = weight.saturating_add_signed(delta).max(1);
64        c.with_layout(&mut |layout| {
65            if adjust_horizontal {
66                layout.width = Sizing::Flex(next);
67            } else {
68                layout.height = Sizing::Flex(next);
69            }
70        })
71    }
72
73    #[command]
74    /// Add a nested block if space permits.
75    fn add(&self, c: &mut dyn Context) -> Result<()> {
76        if let Some(first_child) = c.children().first().copied()
77            && let Some(view) = c.node_view(first_child)
78        {
79            let size = Expanse::new(view.outer.w, view.outer.h);
80            if self.size_limited(size) {
81                return Ok(());
82            }
83            c.add_child(Self::new(!self.horizontal))?;
84        }
85        Ok(())
86    }
87
88    #[command]
89    /// Split into two child blocks.
90    fn split(&self, c: &mut dyn Context) -> Result<()> {
91        let view = c.view();
92        let size = Expanse::new(view.outer.w, view.outer.h);
93        if !self.size_limited(size) && c.children().is_empty() {
94            c.add_child(Self::new(!self.horizontal))?;
95            c.add_child(Self::new(!self.horizontal))?;
96            c.focus_next();
97        }
98        Ok(())
99    }
100
101    #[command]
102    /// Increase this block's flex grow coefficient.
103    fn flex_grow_inc(&self, c: &mut dyn Context) -> Result<()> {
104        self.adjust_flex(c, 1)
105    }
106
107    #[command]
108    /// Decrease this block's flex grow coefficient.
109    fn flex_grow_dec(&self, c: &mut dyn Context) -> Result<()> {
110        self.adjust_flex(c, -1)
111    }
112
113    #[command]
114    /// Increase this block's flex shrink coefficient.
115    fn flex_shrink_inc(&self, c: &mut dyn Context) -> Result<()> {
116        self.adjust_flex(c, 1)
117    }
118
119    #[command]
120    /// Decrease this block's flex shrink coefficient.
121    fn flex_shrink_dec(&self, c: &mut dyn Context) -> Result<()> {
122        self.adjust_flex(c, -1)
123    }
124
125    #[command]
126    /// Focus this block.
127    fn focus(&self, c: &mut dyn Context) -> Result<()> {
128        c.set_focus(c.node_id());
129        Ok(())
130    }
131}
132
133impl Widget for Block {
134    fn accept_focus(&self, ctx: &dyn ReadContext) -> bool {
135        ctx.children().is_empty()
136    }
137
138    fn render(&mut self, r: &mut Render, ctx: &dyn ReadContext) -> Result<()> {
139        // Only render leaf blocks (those without children)
140        if ctx.children().is_empty() {
141            let bc = if ctx.is_focused() { "violet" } else { "blue" };
142            let rect = ctx.view().outer_rect_local();
143            if rect.is_zero() {
144                return Ok(());
145            }
146            r.fill(bc, rect, '\u{2588}')?;
147        }
148        Ok(())
149    }
150
151    fn layout(&self) -> Layout {
152        let base = if self.horizontal {
153            Layout::row()
154        } else {
155            Layout::column()
156        };
157        base.flex_horizontal(1)
158            .flex_vertical(1)
159            .min_width(1)
160            .min_height(1)
161            .gap(1)
162    }
163}

165/// Root node for the focus gym demo.
166#[derive(Default)]
167pub struct FocusGym;
168
169#[derive_commands]
170impl FocusGym {
171    /// Construct a new focus gym.
172    pub fn new() -> Self {
173        Self
174    }
175
176    #[command]
177    /// Delete the currently focused block.
178    fn delete_focused(&self, c: &mut dyn Context) -> Result<()> {
179        let Some(root_block) = c.unique_child::<Block>()? else {
180            return Ok(());
181        };
182        let root_block = NodeId::from(root_block);
183        let Some(focused) = c.focused_leaf(root_block) else {
184            return Ok(());
185        };
186        c.remove_subtree(focused)?;
187        c.focus_first_in(root_block);
188        Ok(())
189    }
190}
191
192impl Widget for FocusGym {
193    fn render(&mut self, _r: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {
194        Ok(())
195    }
196
197    fn on_mount(&mut self, c: &mut dyn Context) -> Result<()> {
198        c.set_layout(Layout::fill())?;
199        let root_block = c.add_child(Block::new(true))?;
200        c.add_child_to(root_block, Block::new(false))?;
201        c.add_child_to(root_block, Block::new(false))?;
202        Ok(())
203    }
204}
205
206impl Loader for FocusGym {
207    fn load(c: &mut Canopy) -> Result<()> {
208        c.add_commands::<Self>()?;
209        c.add_commands::<Block>()?;
210        Ok(())
211    }
212}
