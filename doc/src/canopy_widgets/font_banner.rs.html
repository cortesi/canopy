106    /// Return a cached layout for the provided size.
107    fn layout_for(&mut self, size: Expanse) -> &FontLayout {
108        let rebuild = match &self.cache {
109            Some(cache) => {
110                cache.text != self.text
111                    || cache.size != size
112                    || cache.options != self.options
113                    || cache.effects != self.effects
114            }
115            None => true,
116        };
117        if rebuild {
118            let layout = self
119                .renderer
120                .layout(&self.text, size, self.options, self.effects);
121            self.cache = Some(LayoutCache {
122                text: self.text.clone(),
123                size,
124                options: self.options,
125                effects: self.effects,
126                layout,
127            });
128        }
129        &self.cache.as_ref().expect("layout cached").layout
130    }

139impl Widget for FontBanner {
140    fn layout(&self) -> Layout {
141        Layout::fill()
142    }
143
144    fn render(&mut self, rndr: &mut Render, ctx: &dyn ReadContext) -> Result<()> {
145        let view = ctx.view();
146        let view_rect = view.view_rect_local();
147        if view_rect.w == 0 || view_rect.h == 0 {
148            return Ok(());
149        }
150        let size = Expanse::new(view_rect.w, view_rect.h);
151        let style = if self.selected {
152            self.selected_style
153                .as_deref()
154                .unwrap_or(&self.style)
155                .to_string()
156        } else {
157            self.style.clone()
158        };
159        let options = self.options;
160        let layout = self.layout_for(size);
161
162        let bounds = content_rect(view_rect, layout, options);
163        for (row_idx, row) in layout.cells.iter().enumerate() {
164            let y = view_rect.tl.y.saturating_add(row_idx as u32);
165            if y >= view_rect.tl.y.saturating_add(view_rect.h) {
166                break;
167            }
168            for (col_idx, cell) in row.iter().enumerate() {
169                if cell.fg_coverage == 0 && cell.bg_coverage == 0 {
170                    continue;
171                }
172                let x = view_rect.tl.x.saturating_add(col_idx as u32);
173                if x >= view_rect.tl.x.saturating_add(view_rect.w) {
174                    continue;
175                }
176                let point = Point { x, y };
177                let resolved = rndr.resolve_style_name_at(&style, bounds, point);
178                let blended = blend_style(resolved, cell.fg_coverage, cell.bg_coverage);
179                rndr.put_cell(blended, point, cell.ch)?;
180            }
181        }
182        Ok(())
183    }
184}

186/// Blend a resolved style by coverage weights.
187fn blend_style(resolved: ResolvedStyle, fg_cov: u8, bg_cov: u8) -> ResolvedStyle {
188    let fg_weight = f32::from(fg_cov) / 255.0;
189    let bg_weight = f32::from(bg_cov) / 255.0;
190    let fg = resolved.bg.blend(resolved.fg, fg_weight);
191    let bg = resolved.bg.blend(resolved.fg, bg_weight);
192    ResolvedStyle::new(fg, bg, resolved.attrs)
193}

195/// Compute a gradient bounds rect aligned to the rendered content.
196fn content_rect(view_rect: Rect, layout: &FontLayout, options: LayoutOptions) -> Rect {
197    if layout.content_size.w == 0 || layout.content_size.h == 0 {
198        return view_rect;
199    }
200
201    let offset_x = align_offset(layout.content_size.w, layout.size.w, options.h_align);
202    let offset_y = align_offset(layout.content_size.h, layout.size.h, options.v_align);
203
204    Rect::new(
205        view_rect.tl.x.saturating_add(offset_x),
206        view_rect.tl.y.saturating_add(offset_y),
207        layout.content_size.w,
208        layout.content_size.h,
209    )
210}

212/// Align content inside an available span.
213fn align_offset(content: u32, available: u32, align: Align) -> u32 {
214    if available <= content {
215        return 0;
216    }
217    match align {
218        Align::Start => 0,
219        Align::Center => (available - content) / 2,
220        Align::End => available - content,
221    }
