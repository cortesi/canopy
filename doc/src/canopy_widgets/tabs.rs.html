1use canopy_core as canopy;
2
3use canopy_core::{
4    Context, Node, NodeState, Render, Result, StatefulNode, command, derive_commands,
5};
6
7/// A tab control managing a set of nodes with titles.
8#[derive(canopy_core::StatefulNode)]
9pub struct Tabs {
10    pub state: NodeState,
11    pub tabs: Vec<String>,
12    pub active: usize,
13}
14
15#[derive_commands]
16impl Tabs {
17    pub fn new(tabs: Vec<String>) -> Self {
18        Tabs {
19            state: NodeState::default(),
20            active: 0,
21            tabs,
22        }
23    }
24
25    /// Select the next tab.
26    #[command]
27    pub fn next(&mut self, c: &mut dyn Context) {
28        self.active = (self.active + 1) % self.tabs.len();
29        c.taint(self);
30    }
31
32    /// Select the previous tab.
33    #[command]
34    pub fn prev(&mut self, c: &mut dyn Context) {
35        self.active = (self.active.wrapping_sub(1)) % self.tabs.len();
36        c.taint(self);
37    }
38}
39
40impl Node for Tabs {
41    fn render(&mut self, _c: &dyn Context, r: &mut Render) -> Result<()> {
42        for (i, rect) in self
43            .vp()
44            .view()
45            .split_horizontal(self.tabs.len() as u32)?
46            .iter()
47            .enumerate()
48        {
49            let styl = if i == self.active {
50                "tab/active"
51            } else {
52                "tab/inactive"
53            };
54            let (text, end) = rect.carve_hend(1);
55            r.text(styl, text.line(0), &self.tabs[i])?;
56            r.text("", end.line(0), " ")?;
57        }
58        Ok(())
59    }
60}
