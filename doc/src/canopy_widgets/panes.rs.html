3use canopy_core::{

7/// Panes manages a set of child nodes arranged in a 2d grid.
8#[derive(canopy_core::StatefulNode)]
9pub struct Panes<N: Node> {
10    pub children: Vec<Vec<N>>,
11    pub state: NodeState,
12}
13
14#[derive_commands]
15impl<N> Panes<N>
16where
17    N: Node,
18{
19    pub fn new(n: N) -> Self {
20        Panes {
21            children: vec![vec![n]],
22            state: NodeState::default(),
23        }
24    }
25
26    /// Get the offset of the current focus in the children vector.
27    pub fn focus_coords(&mut self, c: &dyn Context) -> Option<(usize, usize)> {
28        for (x, col) in self.children.iter_mut().enumerate() {
29            for (y, row) in col.iter_mut().enumerate() {
30                if c.is_on_focus_path(row) {
31                    return Some((x, y));
32                }
33            }
34        }
35        None
36    }
37
38    /// Delete the focus node. If a column ends up empty, it is removed.
39    pub fn delete_focus(&mut self, c: &mut dyn Context) -> Result<()> {
40        if let Some((x, y)) = self.focus_coords(c) {
41            c.focus_next(self);
42            self.children[x].remove(y);
43            if self.children[x].is_empty() {
44                self.children.remove(x);
45            }
46            c.taint_tree(self);
47        }
48        Ok(())
49    }
50
51    /// Insert a node, splitting vertically. If we have a focused node, the new
52    /// node is inserted in a row beneath it. If not, a new column is added.
53    pub fn insert_row(&mut self, c: &mut dyn Context, n: N)
54    where
55        N: Node,
56    {
57        if let Some((x, y)) = self.focus_coords(c) {
58            self.children[x].insert(y, n);
59        } else {
60            self.children.push(vec![n]);
61        }
62        c.taint_tree(self);
63    }
64
65    /// Insert a node in a new column. If we have a focused node, the new node
66    /// is added in a new column to the right.
67    pub fn insert_col(&mut self, c: &mut dyn Context, mut n: N) -> Result<()>
68    where
69        N: Node,
70    {
71        let coords = self.focus_coords(c);
72        c.focus_next(&mut n);
73        if let Some((x, _)) = coords {
74            self.children.insert(x + 1, vec![n])
75        } else {
76            self.children.push(vec![n])
77        }
78        c.taint_tree(self);
79        Ok(())
80    }
81
82    /// Returns the shape of the current child grid
83    fn shape(&self) -> Vec<u32> {
84        let mut ret = vec![];
85        for i in &self.children {
86            ret.push(i.len() as u32)
87        }
88        ret
89    }
90}
91
92impl<N: Node> Node for Panes<N> {
93    fn children(&mut self, f: &mut dyn FnMut(&mut dyn Node) -> Result<()>) -> Result<()> {
94        for col in &mut self.children {
95            for row in col {
96                f(row)?
97            }
98        }
99        Ok(())
100    }
101
102    fn layout(&mut self, l: &Layout, sz: Expanse) -> Result<()> {
103        l.fill(self, sz)?;
104        let vp = self.vp();
105        let lst = vp.view().split_panes(&self.shape())?;
106        for (ci, col) in self.children.iter_mut().enumerate() {
107            for (ri, row) in col.iter_mut().enumerate() {
108                l.place(row, vp, lst[ci][ri])?;
109            }
110        }
111        Ok(())
112    }
113}
