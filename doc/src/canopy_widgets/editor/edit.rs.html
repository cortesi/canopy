3/// A single text edit applied to the buffer.
4#[derive(Debug, Clone, PartialEq, Eq)]
5pub struct Edit {
6    /// Range replaced by the edit (before applying it).
7    pub range: TextRange,
8    /// Text removed from the range.
9    pub deleted: String,
10    /// Text inserted in place of the range.
11    pub inserted: String,
12}
13
14impl Edit {
15    /// Construct a new edit.
16    pub fn new(range: TextRange, deleted: String, inserted: String) -> Self {
17        Self {
18            range,
19            deleted,
20            inserted,
21        }
22    }
23}
24
25/// A group of edits that form a single undo/redo step.
26#[derive(Debug, Clone, PartialEq, Eq)]
27pub struct Transaction {
28    /// Edits in the order they were applied.
29    pub edits: Vec<Edit>,
30    /// Selection state before applying edits.
31    pub before: Selection,
32    /// Selection state after applying edits.
33    pub after: Selection,
34}
35
36impl Transaction {
37    /// Construct a new transaction with a starting selection.
38    pub fn new(before: Selection) -> Self {
39        Self {
40            edits: Vec::new(),
41            before,
42            after: before,
43        }
44    }
45
46    /// Return true when there are no edits recorded.
47    pub fn is_empty(&self) -> bool {
48        self.edits.is_empty()
49    }
50
51    /// Record the selection state after edits are applied.
52    pub fn finish(&mut self, after: Selection) {
53        self.after = after;
54    }
55}
