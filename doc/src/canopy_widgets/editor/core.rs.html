5/// The editor Core exposes the operations that can be performed on a text buffer. It's a facade over the state, with
6/// added operations to support a redo/undo stack.
7#[derive(Debug, Clone, PartialEq, Eq)]
8pub struct Core {
9    state: state::State,
10    /// The history of operations on this text buffer.
11    history: Vec<effect::Effect>,
12    redo: Vec<effect::Effect>,
13}
14
15impl Core {
16    pub fn new(text: &str) -> Self {
17        Core {
18            state: state::State::new(text),
19            history: Vec::new(),
20            redo: Vec::new(),
21        }
22    }
23
24    #[cfg(test)]
25    pub(crate) fn from_spec(spec: &str) -> Self {
26        Core {
27            state: state::State::from_spec(spec),
28            history: Vec::new(),
29            redo: Vec::new(),
30        }
31    }
32
33    /// Undo an operation. Return true if an operation was undone, false if the history is empty.
34    pub fn undo(&mut self) -> bool {
35        if let Some(op) = self.history.pop() {
36            op.revert(&mut self.state);
37            self.redo.push(op);
38            true
39        } else {
40            false
41        }
42    }
43
44    /// Redo an operation. Returne true if an operation was redone, false if redo history is empty.
45    pub fn redo(&mut self) -> bool {
46        if let Some(op) = self.redo.pop() {
47            op.apply(&mut self.state);
48            self.history.push(op);
49            true
50        } else {
51            false
52        }
53    }
54
55    fn action(&mut self, e: effect::Effect) {
56        e.apply(&mut self.state);
57        self.history.push(e);
58        self.redo.clear();
59    }
60
61    /// Insert text at the current cursor position.
62    pub fn insert_text(&mut self, text: &str) {
63        self.action(effect::Effect::Insert(effect::Insert::new(
64            &self.state,
65            self.state.cursor.insert(&self.state),
66            text.to_string(),
67        )));
68    }
69
70    /// Delete text in a given range.
71    pub fn delete<T>(&mut self, start: T, end: T)
72    where
73        T: Into<InsertPos>,
74    {
75        self.action(effect::Effect::Delete(effect::Delete::new(
76            &self.state,
77            start.into(),
78            end.into(),
79        )));
80    }
81
82    pub fn window_text(&self) -> Vec<Option<&str>> {
83        self.state.window_text()
84    }
85
86    pub fn wrapped_height(&self) -> usize {
87        self.state.line_height()
88    }
89
90    pub fn resize_window(&mut self, width: usize, height: usize) {
91        self.state.resize_window(width, height);
92    }
93
94    pub fn cursor_position(&self) -> Option<Point> {
95        self.state.cursor_position()
96    }
97
98    /// Move the cursor within the current chunk, moving to the next or previous wrapped line if needed. Won't move to
99    /// the next chunk.
100    pub fn cursor_shift(&mut self, n: isize) {
101        self.state.cursor_shift(n);
102    }
103
104    /// Move the up or down in the chunk list.
105    pub fn cursor_shift_chunk(&mut self, n: isize) {
106        self.state.cursor_shift_chunk(n);
107    }
108
109    /// Move the up or down along wrapped lines.
110    pub fn cursor_shift_lines(&mut self, n: isize) {
111        self.state.cursor_shift_line(n);
112    }
113}

119    /// Wee helper for effect tests. Applies a set of transformations to a starting state through a closure, then checks
120    /// that the changes achieved the expected end. We then undo all changes and test that we end up with the starting
121    /// state, and redo all changes to make sure we end up at the end again.
122    fn tundo<F>(start: &str, f: F, end: &str)
123    where
124        F: FnOnce(&mut Core),
125    {
126        let start = Core::from_spec(start);
127        let end = Core::from_spec(end);
128
129        let mut s = start.clone();
130
131        f(&mut s);
132        assert_eq!(s.state.text(), end.state.text());
133        loop {
134            if !s.undo() {
135                break;
136            }
137        }
138        assert_eq!(s.state, start.state);
139        loop {
140            if !s.redo() {
141                break;
142            }
143        }
144        assert_eq!(s.state, end.state);
145    }
146
147    #[test]
148    fn insert() {
149        tundo("_", |c| c.insert_text("hello"), "hello_");
150        tundo("<", |c| c.insert_text("hello"), "hello<");
