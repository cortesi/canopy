33/// Wrapping behavior for the editor.
34#[derive(Debug, Clone, Copy, PartialEq, Eq)]
35pub enum WrapMode {
36    /// No wrapping; horizontal scrolling is enabled.
37    None,
38    /// Soft wrapping at the view width.
39    Soft,
40}
41
42/// Editing mode for the editor widget.
43#[derive(Debug, Clone, Copy, PartialEq, Eq)]
44pub enum EditMode {
45    /// Text entry mode with direct insertion.
46    Text,
47    /// Vi-style modal editing.
48    Vi,
49}
50
51/// Line number rendering mode.
52#[derive(Debug, Clone, Copy, PartialEq, Eq)]
53pub enum LineNumbers {
54    /// Do not render line numbers.
55    None,
56    /// Render absolute line numbers.
57    Absolute,
58    /// Render relative line numbers (current line stays absolute).
59    Relative,
60}
61
62/// Configuration for the editor widget.
63#[derive(Debug, Clone)]
64pub struct EditorConfig {
65    /// Allow multi-line content.
66    pub multiline: bool,
67    /// Wrapping mode.
68    pub wrap: WrapMode,
69    /// Auto-grow height to fit contents.
70    pub auto_grow: bool,
71    /// Minimum height when auto-growing.
72    pub min_height: u32,
73    /// Maximum height when auto-growing.
74    pub max_height: Option<u32>,
75    /// Edit mode behavior.
76    pub mode: EditMode,
77    /// Whether the editor is read-only.
78    pub read_only: bool,
79    /// Line number rendering mode.
80    pub line_numbers: LineNumbers,
81    /// Tab stop width in columns.
82    pub tab_stop: usize,
83}
84
85impl Default for EditorConfig {
86    fn default() -> Self {
87        Self {
88            multiline: true,
89            wrap: WrapMode::Soft,
90            auto_grow: false,
91            min_height: 1,
92            max_height: None,
93            mode: EditMode::Text,
94            read_only: false,
95            line_numbers: LineNumbers::None,
96            tab_stop: 4,
97        }
98    }
99}
