5/// Default wrapping width for new states.
6const DEFAULT_WRAP: usize = 80;

8/// The current state of the editor
9#[derive(Debug, Clone, PartialEq, Eq, Hash)]
10pub struct State {
11    /// The underlying raw text being edited.
12    pub chunks: Vec<Chunk>,
13    /// The current cursor position.
14    pub cursor: Cursor,
15    /// The current wrap width
16    pub width: usize,
17    /// Visible window of wrapped lines.
18    pub window: Window,
19}

21impl State {
22    /// Create a new State from the specified text. The cursor begins at the start of the text, in visual mode.
23    pub fn new(text: &str) -> Self {
24        let cursor = Cursor::Char((0, 0).into());
25        let mut t: Vec<Chunk> = text
26            .split("\n")
27            .map(|x| Chunk::new(x, DEFAULT_WRAP))
28            .collect();
29        if t.is_empty() {
30            t.push(Chunk::new("", DEFAULT_WRAP))
31        }
32        Self {
33            chunks: t,
34            cursor,
35            width: DEFAULT_WRAP,
36            window: Window {
37                line: Line {
38                    chunk: 0,
39                    wrap_idx: 0,
40                },
41                height: 0,
42            },
43        }
44    }

46    /// Create a new State from a text specification. An Insert cursor position is indicated by an underscore "_"
47    /// character. A Character cursor position is indicated by a "<" character, which "points at" the character at the
48    /// offset. The cursor position indicator is removed from the final string.
49    #[cfg(test)]
50    pub(crate) fn from_spec(spec: &str) -> Self {
51        let mut txt = vec![];
52        let mut cursor = None;
53        for (cnt, i) in spec.lines().enumerate() {
54            if let Some(x) = i.find("_") {
55                cursor = Some(Cursor::Insert((cnt, x).into()));
56                txt.push(i.replace("_", ""))
57            } else if let Some(x) = i.find("<") {
58                cursor = Some(Cursor::Char((cnt, x.saturating_sub(1)).into()));
59                txt.push(i.replace("<", ""))
60            } else {
61                txt.push(i.into());
62            }
63        }
64        let mut n = Self::new(&txt.join("\n"));
65        if let Some(x) = cursor {
66            n.cursor = x;
67        }
68        n
69    }

71    /// Turns a state into a text specification.
72    #[cfg(test)]
73    pub(crate) fn to_spec(&self) -> String {
74        let mut buf = vec![];
75        let char = match self.cursor {
76            Cursor::Char(_) => '<',
77            Cursor::Insert(_) => '_',
78        };
79        let (chunk, offset) = match self.cursor {
80            Cursor::Char(x) => x.chunk_offset(),
81            Cursor::Insert(x) => x.chunk_offset(),
82        };
83        for (i, c) in self.chunks.iter().enumerate() {
84            let mut s = c.as_str().to_string();
85            if i == chunk {
86                if char == '<' {
87                    if s.is_empty() {
88                        s = "<".into();
89                    } else {
90                        s.insert(offset + 1, char);
91                    }
92                } else {
93                    s.insert(offset, char);
94                }
95            }
96            buf.push(s);
97        }
98        buf.join("\n")
99    }

101    /// Insert a set of lines at the cursor, then update the cursor to point just beyond the last inserted character.
102    pub fn insert_lines<T, S, I>(&mut self, pos: T, s: S)
103    where
104        S: AsRef<[I]>,
105        I: ToString,
106        T: Into<InsertPos>,
107    {
108        let pos = pos.into();
109        let s = s.as_ref();
110        if s.len() == 1 {
111            // Simple case - there are no newlines, we just insert the text in-place.
112            let s = &s[0].to_string();
113            self.chunks[pos.chunk].insert(pos.offset, s);
114
115            let cursor = self.cursor.insert(self);
116            if cursor >= pos {
117                // Adjust the cursor if it was after the insert point.
118                self.cursor = self.cursor.shift(self, s.len() as isize);
119            }
120        } else {
121            // We have a multi-line insert. Start by snipping the line at the insert point into start and end chunks.
122            let start = &self.chunks[pos.chunk].as_str()[..pos.offset];
123            let end = &self.chunks[pos.chunk].as_str()[pos.offset..].to_string();
124
125            // Now modify the start chunk to include the trailer
126            self.chunks[pos.chunk] =
127                Chunk::new(&format!("{}{}", start, s[0].to_string()), self.width);
128
129            // And generate and insert our trailing lines
130            let mut trailer = s[1..].iter().map(|x| x.to_string()).collect::<Vec<_>>();
131            let last = trailer.pop().unwrap();
132            trailer.push(format!("{last}{end}"));
133            self.chunks.splice(
134                pos.chunk + 1..pos.chunk + 1,
135                trailer.iter().map(|x| Chunk::new(x, self.width)),
136            );
137
138            let cursor = self.cursor.insert(self);
139            if cursor >= pos {
140                // The cursor was at or beyond the insert position, so we have to adjust it.
141                self.cursor = self
142                    .cursor
143                    .shift_chunk(self, s.len().saturating_sub(1) as isize);
144                if self.cursor.insert(self).chunk == pos.chunk + trailer.len() {
145                    self.cursor = self.cursor.shift(self, last.len() as isize);
146                }
147            }
148        }
149    }

152    pub fn insert<T>(&mut self, pos: T, s: &str)
153    where
154        T: Into<InsertPos>,
155    {
156        self.insert_lines(pos, s.split("\n").collect::<Vec<&str>>())
157    }

159    /// Insert the given text at the given position, and update the cursor if necessary.
160    pub fn delete<T>(&mut self, start: T, end: T)
161    where
162        T: Into<InsertPos>,
163    {
164        let start: InsertPos = start.into();
165        let end: InsertPos = end.into();
166        let cursor = self.cursor.insert(self);
167
168        if start.chunk > self.chunks.len() || end == start {
169            // Out of bounds, so this is a no-op
170        } else if start.chunk == end.chunk {
171            // We're doing a delete that doesn't cross chunk boundaries.
172            //
173            self.chunks[start.chunk].replace_range(start.offset..end.offset, "");
174            let ip = self.cursor.insert(self);
175            // We only need to adjust the cursor if it was beyond the deletion point
176            if ip > start && ip < end {
177                // If it was within the deleted text, the new cursor position is at the start of the deleted chunk.
178                self.cursor = self.cursor.at(self, start.chunk, start.offset);
179            } else if ip > start && ip.chunk == start.chunk {
180                // If it was beyond the deleted text, we shift the cursor back by the number of chars deleted.
181                self.cursor = self.cursor.at(
182                    self,
183                    ip.chunk,
184                    ip.offset.saturating_sub(end.offset - start.offset - 1),
185                );
186            } else {
187                self.cursor = self.cursor.constrain(self);
188            }
189        } else {
190            // We're doing a delete that crosses chunk boundaries.
191            //
192            // We begin by chopping off the trailer of the first chunk.
193            let mut m = self.chunks.remove(start.chunk);
194            m.replace_range(start.offset.., "");
195
196            // If our deletion range doesn't exceed the number of chunks we have (meaning we are deleting to the end of
197            // the text), we need to splice in the trailer of the last chunk.
198            if self.chunks.len() > end.chunk - 1 {
199                // Remove the last chunk, exract its trailer, and push it onto the end of the first chunk.
200                let mut n = self.chunks.remove(end.chunk - 1);
201                n.replace_range(..end.offset.min(n.len()), "");
202                m.push_str(n.as_str());
203                // Now remove all intermediate chunks - these are chunks that are deleted completely.
204                self.chunks.drain(start.chunk..end.chunk - 1);
205            }
206            self.chunks.insert(start.chunk, m);
207
208            // Now we need to adjust the cursor.
209            if cursor > start && cursor <= end {
210                // The cursor was within the deleted chunk, so the new position is just at deletion point.
211                self.cursor = self.cursor.at(self, start.chunk, start.offset);
212            } else if cursor > start && cursor.chunk == end.chunk {
213                // The cursor was within the trailer of the last chunk. Maintain the character position.
214                self.cursor = self.cursor.at(
215                    self,
216                    start.chunk,
217                    start.offset + cursor.offset.saturating_sub(end.offset),
218                );
219            } else {
220                // The cursor was beyond the deleted chunk. We only need to adjust the chunk offset.
221                self.cursor = self.cursor.at(
222                    self,
223                    cursor.chunk.saturating_sub(end.chunk - start.chunk),
224                    cursor.offset,
225                );
226            }
227        }
228    }

230    /// What's the last insert position in the text?
231    pub(super) fn last(&self) -> InsertPos {
232        let chunk = self.chunks.len().saturating_sub(1);
233        if self.chunks.is_empty() {
234            (0, 0)
235        } else {
236            (chunk, self.chunks[chunk].len().saturating_sub(1))
237        }
238        .into()
239    }

241    /// Retrieve lines of text from inclusive start to exclusive end. The first and last line returned may be partial if
242    /// the offsets are not on line boundaries.
243    pub fn line_range<T>(&self, start: T, end: T) -> Vec<String>
244    where
245        T: Into<InsertPos>,
246    {
247        let start = start.into().constrain(self);
248        let end = end.into().constrain(self);
249
250        let mut buf = vec![];
251        if start.chunk == end.chunk {
252            buf.push(self.chunks[start.chunk].as_str()[start.offset..end.offset].to_string());
253        } else {
254            buf.push(self.chunks[start.chunk].as_str()[start.offset..].to_string());
255            if end.chunk - start.chunk > 1 {
256                for l in &self.chunks[(start.chunk + 1)..(end.chunk - 1)] {
257                    buf.push(l.as_str().into());
258                }
259            }
260            buf.push(self.chunks[end.chunk].as_str()[..end.offset].to_string());
261        }
262        buf
263    }

283    /// Calculate the (x, y) co-ordinates of a cursor within a wrapped window. If the position is not in the window,
284    /// None is returned. Empty chunks are handled specially, with the
285    pub fn cursor_position(&self) -> Option<Point> {
286        let pos = self.cursor.insert(self);
287        let c = &self.chunks[pos.chunk];
288        for (y, l) in self.window.lines(self).iter().enumerate() {
289            if let Some(l) = l {
290                let (lstart, lend) = self.chunks[l.chunk].wraps[l.wrap_idx];
291                if c.len() == 0 && l.chunk == pos.chunk {
292                    // We're at the first character of an empty chunk.
293                    return Some((0, y as u32).into());
294                } else if pos.offset >= c.len() && l.chunk > pos.chunk {
295                    // We're beyond the end of the chunk, which means we must be an insertion cursor. Place the cursor
296                    // position at the first character of the next line.
297                    return Some((0, y as u32).into());
298                } else if l.chunk == pos.chunk && lstart <= pos.offset && lend > pos.offset {
299                    return Some(((pos.offset - lstart) as u32, y as u32).into());
300                }
301            }
302        }
303        None
304    }

306    /// Return the wrapped lines in the window. The start offset is in terms of the wrapped text. The returned Vec
307    /// may be shorter than length if the end of the text is reached.
308    pub fn window_text(&self) -> Vec<Option<&str>> {
309        let mut buf = vec![];
310        for l in self.window.lines(self) {
311            if let Some(l) = l {
312                buf.push(Some(self.chunks[l.chunk].wrapped_line(l.wrap_idx)));
313            } else {
314                buf.push(None);
315            }
316        }
317        buf
318    }

320    /// Return the total number of wrapped lines.
321    pub fn line_height(&self) -> usize {
322        self.chunks.iter().map(|x| x.wraps.len()).sum()
323    }

325    /// Set the width of the editor for wrapping, and return the total number of wrapped lines that resulted.
326    pub fn resize_window(&mut self, width: usize, height: usize) -> usize {
327        // This needs to be as cheap as possible if the width hasn't changed.
328        if self.width == width && self.window.height == height {
329            return self.line_height();
330        }
331        self.width = width;
332        self.window = self.window.with_height(height);
333        self.chunks.iter_mut().map(|x| x.wrap(width)).sum()
334    }

336    /// Move the cursor left or right within the current chunk, moving to the next or previous wrapped line if needed.
337    /// Won't move to the next chunk. Adjust the window to include the cursor if needed.
338    pub fn cursor_shift(&mut self, n: isize) {
339        self.cursor = self.cursor.shift(self, n);
340        self.window = self.window.adjust(self);
341    }

361    /// Take a state specification a, turn it into a State object, apply the transformation f, then check if the result
362    /// is equal to the state specification b.
363    fn seq<F>(a: &str, f: F, b: &str)
364    where
365        F: FnOnce(&mut State),
366    {
367        let mut a = State::from_spec(a);
368        let b = State::from_spec(b);
369        f(&mut a);
370        assert_eq!(a, b);
371    }

373    /// Verifies a text specification against the visible editor window. The window is resized to the specified width
374    /// and height before verification. If a cursor is present in the text specification, it is also validated. If no
375    /// cursor is specified, the current editor cursor is ignored.
376    fn assert_window(s: &mut State, w: usize, h: usize, offset: usize, t: &str) {
377        s.resize_window(w, h);
378        s.window = s.window.at_line(s, offset);
379        let split = if t.is_empty() {
380            vec![]
381        } else {
382            t.split("\n").collect::<Vec<_>>()
383        };
384        let cp = s.cursor_position();
385        for (i, w) in s.window_text().iter().enumerate() {
386            if i < split.len() {
387                let w = w.unwrap();
388                let s = if let Some(x) = split[i].find("_") {
389                    let cp = cp.unwrap();
390                    assert_eq!(cp.x, x as u32);
391                    assert_eq!(cp.y, i as u32);
392                    split[i].replace("_", "")
393                } else if let Some(x) = split[i].find("<") {
394                    let cp = cp.unwrap();
395                    assert_eq!(cp.x, (x - 1) as u32);
396                    assert_eq!(cp.y, i as u32);
397                    split[i].replace("<", "")
398                } else {
399                    split[i].into()
400                };
401                assert_eq!(w, s);
402            } else {
403                assert!(w.is_none());
404            }
405        }
406    }
