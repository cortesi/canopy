16#[derive_commands]
17impl EditorView {

19        EditorView {
20            state: NodeState::default(),
21            core: core::Core::new(txt),
22        }
23    }

26impl Node for EditorView {
27    fn cursor(&self) -> Option<cursor::Cursor> {
28        let p = self.core.cursor_position();
29        p.map(|p| cursor::Cursor {
30            location: p,
31            shape: cursor::CursorShape::Block,
32            blink: true,
33        })
34    }
35
36    fn accept_focus(&mut self) -> bool {
37        true
38    }
39
40    fn render(&mut self, _: &dyn Context, r: &mut Render) -> Result<()> {
41        let vo = self.vp().view();
42        let sr = self.vp().screen_rect();
43        self.core.resize_window(sr.w as usize, sr.h as usize);
44        for (i, s) in self.core.window_text().iter().enumerate() {
45            if let Some(t) = s {
46                r.text("text", vo.line(i as u32), t)?;
47            }
48        }
49        Ok(())
50    }
51
52    fn layout(&mut self, _l: &Layout, sz: Expanse) -> Result<()> {
53        let outer = Expanse::new(sz.w, self.core.wrapped_height() as u32);
54        self.fit_size(outer, sz);
55        Ok(())
56    }
57}

59/// A simple editor
60#[derive(canopy_core::StatefulNode)]
61pub struct Editor {
62    state: NodeState,
63    view: EditorView,
64}
65
66#[derive_commands]
67impl Editor {
68    pub fn new(txt: &str) -> Self {
69        Editor {
70            state: NodeState::default(),
71            view: EditorView::new(txt),
72        }
73    }
74
75    /// Move the cursor left or right.
76    #[command]
77    fn cursor_shift(&mut self, _: &dyn Context, n: isize) {
78        self.view.core.cursor_shift(n);
79    }
80
81    /// Move the cursor up or down in the chunk list.
82    #[command]
83    fn cursor_shift_chunk(&mut self, _: &dyn Context, n: isize) {
84        self.view.core.cursor_shift_chunk(n);
85    }
86
87    /// Move the cursor up or down by visual line.
88    #[command]
89    fn cursor_shift_lines(&mut self, _: &dyn Context, n: isize) {
90        self.view.core.cursor_shift_lines(n);
91    }
92}
