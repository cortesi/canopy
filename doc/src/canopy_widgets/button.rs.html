25/// Button widget that triggers a command when clicked.
26pub struct Button {
27    /// Button label.
28    label: String,
29    /// Command invocation to dispatch on click.
30    command: Option<CommandInvocation>,
31    /// Glyph set for the button border.
32    glyphs: BoxGlyphs,
33    /// Active state for the button.
34    active: bool,
35    /// Selection state for use in lists.
36    selected: bool,
37    /// Slot for the box container.
38    box_slot: Slot<BoxSlot>,
39    /// Slot for the centered label container.
40    center_slot: Slot<CenterSlot>,
41    /// Slot for the label text.
42    label_slot: Slot<LabelSlot>,
43}
44
45impl Selectable for Button {
46    fn set_selected(&mut self, selected: bool) {
47        self.selected = selected;
48    }
49}
50
51#[derive_commands]
52impl Button {
53    /// Construct a new button with a label.
54    pub fn new(label: impl Into<String>) -> Self {
55        Self {
56            label: label.into(),
57            command: None,
58            glyphs: SINGLE,
59            active: false,
60            selected: false,
61            box_slot: Slot::new(),
62            center_slot: Slot::new(),
63            label_slot: Slot::new(),
64        }
65    }
66
67    /// Build a button with a specified glyph set.
68    pub fn with_glyphs(mut self, glyphs: BoxGlyphs) -> Self {
69        self.glyphs = glyphs;
70        self
71    }
72
73    /// Build a button that dispatches a command when clicked.
74    pub fn with_command(mut self, command: CommandCall) -> Self {
75        self.command = Some(command.invocation());
76        self
77    }
78
79    /// Build a button with an active state.
80    pub fn with_active(mut self, active: bool) -> Self {
81        self.active = active;
82        self
83    }
84
85    /// Return the button label.
86    pub fn label(&self) -> &str {
87        &self.label
88    }
89
90    /// Set whether the button is active.
91    pub fn set_active(&mut self, active: bool) {
92        self.active = active;
93    }
94
95    /// Replace the button label.
96    pub fn set_label(&mut self, ctx: &mut dyn Context, label: impl Into<String>) -> Result<()> {
97        self.label = label.into();
98        self.sync_label(ctx)
99    }
100
101    /// Trigger the button action.
102    #[command]
103    pub fn press(&mut self, ctx: &mut dyn Context) -> Result<()> {
104        if let Some(command) = self.command.as_ref() {
105            ctx.dispatch_command(command)?;
106        }
107        Ok(())
108    }
109
110    /// Handle a mouse click event.
111    fn handle_click(&mut self, ctx: &mut dyn Context, event: mouse::MouseEvent) -> Result<bool> {
112        if event.button == mouse::Button::Left && event.action == mouse::Action::Down {
113            self.press(ctx)?;
114            return Ok(true);
115        }
116        Ok(false)
117    }
118
119    /// Compute the label width in terminal cells.
120    fn label_width(&self) -> u32 {
121        self.label
122            .lines()
123            .map(UnicodeWidthStr::width)
124            .max()
125            .unwrap_or(0)
126            .max(1) as u32
127    }
128
129    /// Sync the label text widget to the current label.
130    fn sync_label(&mut self, ctx: &mut dyn Context) -> Result<()> {
131        let box_id = self
132            .box_slot
133            .get_or_create(ctx, || Box::new().with_glyphs(self.glyphs).with_fill())?;
134        let center_id = self
135            .center_slot
136            .get_or_create_in(ctx, box_id, Center::new)?;
137        let label_id = self
138            .label_slot
139            .get_or_create_in(ctx, center_id, || Text::new(self.label.clone()))?;
140        ctx.with_typed(label_id, |text, _| {
141            text.set_raw(self.label.clone());
142            Ok(())
143        })?;
144        ctx.set_layout_of(label_id, Layout::column().max_width(self.label_width()))?;
145        Ok(())
146    }
147}
148
149impl Default for Button {
150    fn default() -> Self {
151        Self::new("")
152    }
153}
154
155impl Widget for Button {
156    fn layout(&self) -> Layout {
157        Layout::fill()
158    }
159
160    fn on_mount(&mut self, ctx: &mut dyn Context) -> Result<()> {
161        self.sync_label(ctx)
162    }
163
164    fn render(&mut self, rndr: &mut Render, _ctx: &dyn ReadContext) -> Result<()> {
165        rndr.push_layer("button");
166        if self.selected {
167            rndr.push_layer("selected");
168        }
169        if self.active {
170            rndr.push_layer("active");
171        } else {
172            rndr.push_layer("inactive");
173        }
174        Ok(())
175    }
176
177    fn on_event(&mut self, event: &Event, ctx: &mut dyn Context) -> Result<EventOutcome> {
178        if let Event::Mouse(mouse_event) = event
179            && self.handle_click(ctx, *mouse_event)?
180        {
181            return Ok(EventOutcome::Handle);
182        }
183        Ok(EventOutcome::Ignore)
184    }
185
186    fn name(&self) -> NodeName {
187        NodeName::convert("button")
188    }
189}
