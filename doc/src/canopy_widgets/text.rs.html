71impl Node for Text {
72    fn layout(&mut self, _l: &Layout, s: Expanse) -> Result<()> {
73        let w = if let Some(w) = self.fixed_width {
74            w
75        } else {
76            s.w
77        };
78        // Only resize if the width has changed
79        if self.current_size.w != w {
80            let mut split: Vec<String> = vec![];
81            for i in textwrap::wrap(&self.raw, w as usize) {
82                split.push(format!("{:width$}", i, width = w as usize))
83            }
84            self.current_size = Expanse {
85                w,
86                h: split.len() as u32,
87            };
88            self.lines = Some(split);
89        }
90        let cs = self.current_size;
91        self.fit_size(cs, s);
92        Ok(())
93    }
94
95    fn render(&mut self, _c: &dyn Context, rndr: &mut Render) -> Result<()> {
96        let vo = self.vp().view();
97        if let Some(lines) = self.lines.as_ref() {
98            for i in 0..vo.h {
99                let out = &lines[(vo.tl.y + i) as usize]
100                    .chars()
101                    .skip(vo.tl.x as usize)
102                    .collect::<String>();
103                rndr.text("text", vo.line(i), out)?;
104            }
105        }
106        Ok(())
107    }
