1use super::{PointI32, Rect};
2
3/// A rectangle with a signed origin and unsigned size.
4#[derive(Debug, Clone, Copy, Hash, PartialEq, Eq, Default)]
5pub struct RectI32 {
6    /// Top-left corner.
7    pub tl: PointI32,
8    /// Width.
9    pub w: u32,
10    /// Height.
11    pub h: u32,
12}
13
14impl RectI32 {
15    /// Construct a rectangle from coordinates and size.
16    pub fn new(x: i32, y: i32, w: u32, h: u32) -> Self {
17        Self {
18            tl: PointI32 { x, y },
19            w,
20            h,
21        }
22    }
23
24    /// Does this rect have a zero size?
25    pub fn is_zero(&self) -> bool {
26        self.w == 0 || self.h == 0
27    }
28
29    /// Check if the rectangle contains a point.
30    pub fn contains_point(&self, p: super::Point) -> bool {
31        let px = p.x as i64;
32        let py = p.y as i64;
33        let left = self.tl.x as i64;
34        let top = self.tl.y as i64;
35        let right = left + self.w as i64;
36        let bottom = top + self.h as i64;
37
38        px >= left && px < right && py >= top && py < bottom
39    }
40
41    /// Convert a screen point to local coordinates relative to this rect.
42    /// If the point is to the left/top of the rect, the result clamps to 0.
43    pub fn to_local_point(&self, p: super::Point) -> super::Point {
44        let px = p.x as i64;
45        let py = p.y as i64;
46        let left = self.tl.x as i64;
47        let top = self.tl.y as i64;
48        super::Point {
49            x: (px - left).max(0) as u32,
50            y: (py - top).max(0) as u32,
51        }
52    }
53
54    /// Intersect this signed rect with an unsigned rect in the same coordinate space.
55    pub fn intersect_rect(&self, other: Rect) -> Option<Rect> {
56        let left = self.tl.x;
57        let top = self.tl.y;
58        let right =
59            (self.tl.x as i64 + self.w as i64).clamp(i32::MIN as i64, i32::MAX as i64) as i32;
60        let bottom =
61            (self.tl.y as i64 + self.h as i64).clamp(i32::MIN as i64, i32::MAX as i64) as i32;
62
63        let other_left = other.tl.x as i32;
64        let other_top = other.tl.y as i32;
65        let other_right =
66            (other_left as i64 + other.w as i64).clamp(i32::MIN as i64, i32::MAX as i64) as i32;
67        let other_bottom =
68            (other_top as i64 + other.h as i64).clamp(i32::MIN as i64, i32::MAX as i64) as i32;
69
70        let inter_left = left.max(other_left);
71        let inter_top = top.max(other_top);
72        let inter_right = right.min(other_right);
73        let inter_bottom = bottom.min(other_bottom);
74
75        if inter_right <= inter_left || inter_bottom <= inter_top {
76            return None;
77        }
78
79        Some(Rect::new(
80            inter_left as u32,
81            inter_top as u32,
82            (inter_right - inter_left) as u32,
83            (inter_bottom - inter_top) as u32,
84        ))
85    }
86
87    /// Left edge of the rect.
88    pub fn left(&self) -> i64 {
89        self.tl.x as i64
90    }
91
92    /// Top edge of the rect.
93    pub fn top(&self) -> i64 {
94        self.tl.y as i64
95    }
96
97    /// Right edge of the rect.
98    pub fn right(&self) -> i64 {
99        self.tl.x as i64 + self.w as i64
100    }
101
102    /// Bottom edge of the rect.
103    pub fn bottom(&self) -> i64 {
104        self.tl.y as i64 + self.h as i64
105    }
106
107    /// Center point of the rect.
108    pub fn center(&self) -> (i64, i64) {
109        (
110            self.left() + self.w as i64 / 2,
111            self.top() + self.h as i64 / 2,
112        )
113    }
114
115    /// Return true if this rect overlaps another vertically.
116    pub fn overlaps_vertical(&self, other: Self) -> bool {
117        self.top() < other.bottom() && self.bottom() > other.top()
118    }
119
120    /// Return true if this rect overlaps another horizontally.
121    pub fn overlaps_horizontal(&self, other: Self) -> bool {
122        self.left() < other.right() && self.right() > other.left()
123    }
124}
125
126impl From<Rect> for RectI32 {
127    fn from(r: Rect) -> Self {
128        Self {
129            tl: PointI32 {
130                x: r.tl.x as i32,
131                y: r.tl.y as i32,
132            },
133            w: r.w,
134            h: r.h,
135        }
136    }
